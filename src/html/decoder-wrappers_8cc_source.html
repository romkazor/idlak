64 void DecodeUtteranceLatticeFasterClass::operator () () {
65  // Decoding and lattice determinization happens here.
66  computed_ = true; // Just means this function was called-- a check on the
67  // calling code.
68  success_ = true;
69  using fst::VectorFst;
70  if (!decoder_->Decode(decodable_)) {
71  KALDI_WARN << "Failed to decode file " << utt_;
72  success_ = false;
73  }
74  if (!decoder_->ReachedFinal()) {
75  if (allow_partial_) {
76  KALDI_WARN << "Outputting partial output for utterance " << utt_
77  << " since no final-state reached\n";
78  partial_ = true;
79  } else {
80  KALDI_WARN << "Not producing output for utterance " << utt_
81  << " since no final-state reached and "
82  << "--allow-partial=false.\n";
83  success_ = false;
84  }
85  }
86  if (!success_) return;
87 
88  // Get lattice, and do determinization if requested.
89  lat_ = new Lattice;
90  decoder_->GetRawLattice(lat_);
91  if (lat_->NumStates() == 0)
92  KALDI_ERR << "Unexpected problem getting lattice for utterance " << utt_;
93  fst::Connect(lat_);
94  if (determinize_) {
95  clat_ = new CompactLattice;
96  if (!DeterminizeLatticePhonePrunedWrapper(
97  *trans_model_,
98  lat_,
99  decoder_->GetOptions().lattice_beam,
100  clat_,
101  decoder_->GetOptions().det_opts))
102  KALDI_WARN << "Determinization finished earlier than the beam for "
103  << "utterance " << utt_;
104  delete lat_;
105  lat_ = NULL;
106  // We'll write the lattice without acoustic scaling.
107  if (acoustic_scale_ != 0.0)
108  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale_), clat_);
109  } else {
110  // We'll write the lattice without acoustic scaling.
111  if (acoustic_scale_ != 0.0)
112  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale_), lat_);
113  }
114 }

116 DecodeUtteranceLatticeFasterClass::~DecodeUtteranceLatticeFasterClass() {
117  if (!computed_)
118  KALDI_ERR << "Destructor called without operator (), error in calling code.";
119 
120  if (!success_) {
121  if (num_err_ != NULL) (*num_err_)++;
122  } else { // successful decode.
123  // Getting the one-best output is lightweight enough that we can do it in
124  // the destructor (easier than adding more variables to the class, and
125  // will rarely slow down the main thread.)
126  double likelihood;
127  LatticeWeight weight;
128  int32 num_frames;
129  { // First do some stuff with word-level traceback...
130  // This is basically for diagnostics.
131  fst::VectorFst<LatticeArc> decoded;
132  decoder_->GetBestPath(&decoded);
133  if (decoded.NumStates() == 0) {
134  // Shouldn't really reach this point as already checked success.
135  KALDI_ERR << "Failed to get traceback for utterance " << utt_;
136  }
137  std::vector<int32> alignment;
138  std::vector<int32> words;
139  GetLinearSymbolSequence(decoded, &alignment, &words, &weight);
140  num_frames = alignment.size();
141  if (words_writer_->IsOpen())
142  words_writer_->Write(utt_, words);
143  if (alignments_writer_->IsOpen())
144  alignments_writer_->Write(utt_, alignment);
145  if (word_syms_ != NULL) {
146  std::cerr << utt_ << ' ';
147  for (size_t i = 0; i < words.size(); i++) {
148  std::string s = word_syms_->Find(words[i]);
149  if (s == "")
150  KALDI_ERR << "Word-id " << words[i] << " not in symbol table.";
151  std::cerr << s << ' ';
152  }
153  std::cerr << '\n';
154  }
155  likelihood = -(weight.Value1() + weight.Value2());
156  }
157 
158  // Ouptut the lattices.
159  if (determinize_) { // CompactLattice output.
160  KALDI_ASSERT(compact_lattice_writer_ != NULL && clat_ != NULL);
161  if (clat_->NumStates() == 0) {
162  KALDI_WARN << "Empty lattice for utterance " << utt_;
163  } else {
164  compact_lattice_writer_->Write(utt_, *clat_);
165  }
166  delete clat_;
167  clat_ = NULL;
168  } else {
169  KALDI_ASSERT(lattice_writer_ != NULL && lat_ != NULL);
170  if (lat_->NumStates() == 0) {
171  KALDI_WARN << "Empty lattice for utterance " << utt_;
172  } else {
173  lattice_writer_->Write(utt_, *lat_);
174  }
175  delete lat_;
176  lat_ = NULL;
177  }
178 
179  // Print out logging information.
180  KALDI_LOG << "Log-like per frame for utterance " << utt_ << " is "
181  << (likelihood / num_frames) << " over "
182  << num_frames << " frames.";
183  KALDI_VLOG(2) << "Cost for utterance " << utt_ << " is "
184  << weight.Value1() << " + " << weight.Value2();
185 
186  // Now output the various diagnostic variables.
187  if (like_sum_ != NULL) *like_sum_ += likelihood;
188  if (frame_sum_ != NULL) *frame_sum_ += num_frames;
189  if (num_done_ != NULL) (*num_done_)++;
190  if (partial_ && num_partial_ != NULL) (*num_partial_)++;
191  }
192  // We were given ownership of these two objects that were passed in in
193  // the initializer.
194  delete decoder_;
195  delete decodable_;
196 }

428 // see comment in header.
429 void ModifyGraphForCarefulAlignment(
430  fst::VectorFst<fst::StdArc> *fst) {
431  typedef fst::StdArc Arc;
432  typedef Arc::StateId StateId;
433  typedef Arc::Label Label;
434  typedef Arc::Weight Weight;
435  StateId num_states = fst->NumStates();
436  if (num_states == 0) {
437  KALDI_WARN << "Empty FST input.";
438  return;
439  }
440  Weight zero = Weight::Zero();
441  // fst_rhs will be the right hand side of the Concat operation.
442  fst::VectorFst<fst::StdArc> fst_rhs(*fst);
443  // first remove the final-probs from fst_rhs.
444  for (StateId state = 0; state < num_states; state++)
445  fst_rhs.SetFinal(state, zero);
446  StateId pre_initial = fst_rhs.AddState();
447  Arc to_initial(0, 0, Weight::One(), fst_rhs.Start());
448  fst_rhs.AddArc(pre_initial, to_initial);
449  fst_rhs.SetStart(pre_initial);
450  // make the pre_initial state final with probability one;
451  // this is equivalent to keeping the final-probs of the first
452  // FST when we do concat (otherwise they would get deleted).
453  fst_rhs.SetFinal(pre_initial, Weight::One());
454  fst::VectorFst<fst::StdArc> fst_concat;
455  fst::Concat(fst, fst_rhs);
456 }

459 void AlignUtteranceWrapper(
460  const AlignConfig &config,
461  const std::string &utt,
462  BaseFloat acoustic_scale, // affects scores written to scores_writer, if
463  // present
464  fst::VectorFst<fst::StdArc> *fst, // non-const in case config.careful ==
465  // true.
466  DecodableInterface *decodable, // not const but is really an input.
467  Int32VectorWriter *alignment_writer,
468  BaseFloatWriter *scores_writer,
469  int32 *num_done,
470  int32 *num_error,
471  int32 *num_retried,
472  double *tot_like,
473  int64 *frame_count,
474  BaseFloatVectorWriter *per_frame_acwt_writer) {
475 
476  if ((config.retry_beam != 0 && config.retry_beam <= config.beam) ||
477  config.beam <= 0.0) {
478  KALDI_ERR << "Beams do not make sense: beam " << config.beam
479  << ", retry-beam " << config.retry_beam;
480  }
481 
482  if (fst->Start() == fst::kNoStateId) {
483  KALDI_WARN << "Empty decoding graph for " << utt;
484  if (num_error != NULL) (*num_error)++;
485  return;
486  }
487 
488  if (config.careful)
489  ModifyGraphForCarefulAlignment(fst);
490 
491  FasterDecoderOptions decode_opts;
492  decode_opts.beam = config.beam;
493 
494  FasterDecoder decoder(*fst, decode_opts);
495  decoder.Decode(decodable);
496 
497  bool ans = decoder.ReachedFinal(); // consider only final states.
498 
499  if (!ans && config.retry_beam != 0.0) {
500  if (num_retried != NULL) (*num_retried)++;
501  KALDI_WARN << "Retrying utterance " << utt << " with beam "
502  << config.retry_beam;
503  decode_opts.beam = config.retry_beam;
504  decoder.SetOptions(decode_opts);
505  decoder.Decode(decodable);
506  ans = decoder.ReachedFinal();
507  }
508 
509  if (!ans) { // Still did not reach final state.
510  KALDI_WARN << "Did not successfully decode file " << utt << ", len = "
511  << decodable->NumFramesReady();
512  if (num_error != NULL) (*num_error)++;
513  return;
514  }
515 
516  fst::VectorFst<LatticeArc> decoded; // linear FST.
517  decoder.GetBestPath(&decoded);
518  if (decoded.NumStates() == 0) {
519  KALDI_WARN << "Error getting best path from decoder (likely a bug)";
520  if (num_error != NULL) (*num_error)++;
521  return;
522  }
523 
524  std::vector<int32> alignment;
525  std::vector<int32> words;
526  LatticeWeight weight;
527 
528  GetLinearSymbolSequence(decoded, &alignment, &words, &weight);
529  BaseFloat like = -(weight.Value1()+weight.Value2()) / acoustic_scale;
530 
531  if (num_done != NULL) (*num_done)++;
532  if (tot_like != NULL) (*tot_like) += like;
533  if (frame_count != NULL) (*frame_count) += decodable->NumFramesReady();
534 
535  if (alignment_writer != NULL && alignment_writer->IsOpen())
536  alignment_writer->Write(utt, alignment);
537 
538  if (scores_writer != NULL && scores_writer->IsOpen())
539  scores_writer->Write(utt, -(weight.Value1()+weight.Value2()));
540 
541  Vector<BaseFloat> per_frame_loglikes;
542  if (per_frame_acwt_writer != NULL && per_frame_acwt_writer->IsOpen()) {
543  GetPerFrameAcousticCosts(decoded, &per_frame_loglikes);
544  per_frame_loglikes.Scale(-1 / acoustic_scale);
545  per_frame_acwt_writer->Write(utt, per_frame_loglikes);
546  }
547 }
