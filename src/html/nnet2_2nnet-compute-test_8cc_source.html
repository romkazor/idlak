29 void UnitTestNnetCompute() {
30  int32 input_dim = 10 + rand() % 40, output_dim = 100 + rand() % 500;
31  bool pad_input = (rand() % 2 == 0);
32 
33  Nnet *nnet = GenRandomNnet(input_dim, output_dim);
34  KALDI_LOG << "Left context = " << nnet->LeftContext() << ", right context = "
35  << nnet->RightContext() << ", pad-input = " << pad_input;
36  KALDI_LOG << "NNet info is " << nnet->Info();
37  int32 num_feats = 5 + rand() % 1000;
38  CuMatrix<BaseFloat> input(num_feats, input_dim);
39  input.SetRandn();
40 
41  int32 num_output_rows = num_feats -
42  (pad_input ? 0 : nnet->LeftContext() + nnet->RightContext());
43  if (num_output_rows <= 0)
44  return;
45  CuMatrix<BaseFloat> output1(num_output_rows, output_dim);
46  NnetComputation(*nnet, input, pad_input, &output1);
47 
48  CuMatrix<BaseFloat> output2(output1.NumRows(), output1.NumCols());
49  int32 cur_input_pos = 0, cur_output_pos = 0;
50 
51  NnetOnlineComputer computer(*nnet, pad_input);
52  while (cur_input_pos <= num_feats) {
53  int32 feats_left = num_feats - cur_input_pos;
54  CuMatrix<BaseFloat> output_part;
55  if (feats_left > 0) {
56  int32 chunk_size = std::min<int32>(1 + rand() % 10, feats_left);
57  CuSubMatrix<BaseFloat> input_part(input, cur_input_pos, chunk_size,
58  0, input_dim);
59  computer.Compute(input_part, &output_part);
60  cur_input_pos += chunk_size;
61  } else {
62  computer.Flush(&output_part);
63  cur_input_pos++; // will terminate the loop.
64  }
65  if (output_part.NumRows() != 0) {
66  output2.Range(cur_output_pos, output_part.NumRows(),
67  0, output_dim).CopyFromMat(output_part);
68  cur_output_pos += output_part.NumRows();
69  }
70  }
71  AssertEqual(output1, output2);
72  for (int32 i = 0; i < output1.NumRows(); i++) {
73  // just double-check that the frames near the end are right, in case
74  // the test above somehow passed despite that.
75  if (i < 10 || output1.NumRows() - i < 10) {
76  CuSubVector<BaseFloat> vec1(output1, i), vec2(output2, i);
77  AssertEqual(vec1, vec2);
78  }
79  }
80  KALDI_LOG << "OK";
81  delete nnet;
82 }
