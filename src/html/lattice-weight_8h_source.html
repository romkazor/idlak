95  bool Member() const {
96  // value1_ == value1_ tests for NaN.
97  // also test for no -inf, and either both or neither
98  // must be +inf, and
99  if (value1_ != value1_ || value2_ != value2_) return false; // NaN
100  if (value1_ == -numeric_limits<T>::infinity() ||
101  value2_ == -numeric_limits<T>::infinity()) return false; // -infty not allowed
102  if (value1_ == numeric_limits<T>::infinity() ||
103  value2_ == numeric_limits<T>::infinity()) {
104  if (value1_ != numeric_limits<T>::infinity() ||
105  value2_ != numeric_limits<T>::infinity()) return false; // both must be +infty;
106  // this is necessary so that the semiring has only one zero.
107  }
108  return true;
109  }

111  LatticeWeightTpl Quantize(float delta = kDelta) const {
112  if (value1_+value2_ == -numeric_limits<T>::infinity()) {
113  return LatticeWeightTpl(-numeric_limits<T>::infinity(), -numeric_limits<T>::infinity());
114  } else if (value1_+value2_ == numeric_limits<T>::infinity()) {
115  return LatticeWeightTpl(numeric_limits<T>::infinity(), numeric_limits<T>::infinity());
116  } else if (value1_+value2_ != value1_+value2_) { // NaN
117  return LatticeWeightTpl(value1_+value2_, value1_+value2_);
118  } else {
119  return LatticeWeightTpl(floor(value1_/delta + 0.5F)*delta, floor(value2_/delta + 0.5F) * delta);
120  }
121  }
122  static constexpr uint64 Properties() {
123  return kLeftSemiring | kRightSemiring | kCommutative |
124  kPath | kIdempotent;

173  // Internal helper function, used in ReadNoParen.
174  inline static void ReadFloatType(istream &strm, T &f) {
175  string s;
176  strm >> s;
177  if (s == "Infinity") {
178  f = numeric_limits<T>::infinity();
179  } else if (s == "-Infinity") {
180  f = -numeric_limits<T>::infinity();
181  } else if (s == "BadNumber") {
182  f = numeric_limits<T>::infinity();
183  f -= f; // get NaN
184  } else {
185  char *p;
186  f = strtod(s.c_str(), &p);
187  if (p < s.c_str() + s.size())
188  strm.clear(std::ios::badbit);
189  }

192  // Reads LatticeWeight when there are no parentheses around pair terms...
193  // currently the only form supported.
194  inline istream &ReadNoParen(
195  istream &strm, char separator) {
196  int c;
197  do {
198  c = strm.get();
199  } while (isspace(c));
200 
201  string s1;
202  while (c != separator) {
203  if (c == EOF) {
204  strm.clear(std::ios::badbit);
205  return strm;
206  }
207  s1 += c;
208  c = strm.get();
209  }
210  istringstream strm1(s1);
211  ReadFloatType(strm1, value1_); // ReadFloatType is class member function
212  // read second element
213  ReadFloatType(strm, value2_);
214  return strm;
215  }

228 /* ScaleTupleWeight is a function defined for LatticeWeightTpl and
229  CompactLatticeWeightTpl that mutliplies the pair (value1_, value2_) by a 2x2
230  matrix. Used, for example, in applying acoustic scaling.
231  */
232 template<class FloatType, class ScaleFloatType>
233 inline LatticeWeightTpl<FloatType> ScaleTupleWeight(
234  const LatticeWeightTpl<FloatType> &w,
235  const vector<vector<ScaleFloatType> > &scale) {
236  // Without the next special case we'd get NaNs from infinity * 0
237  if (w.Value1() == numeric_limits<FloatType>::infinity())
238  return LatticeWeightTpl<FloatType>::Zero();
239  return LatticeWeightTpl<FloatType>(scale[0][0] * w.Value1() + scale[0][1] * w.Value2(),
240  scale[1][0] * w.Value1() + scale[1][1] * w.Value2());
241 }

243 /* For testing purposes and in case it's ever useful, we define a similar
244  function to apply to LexicographicWeight and the like, templated on
245  TropicalWeight<float> etc.; we use PairWeight which is the base class of
246  LexicographicWeight.
247 */
248 template<class FloatType, class ScaleFloatType>
249 inline PairWeight<TropicalWeightTpl<FloatType>,
250  TropicalWeightTpl<FloatType> > ScaleTupleWeight(
251  const PairWeight<TropicalWeightTpl<FloatType>,
252  TropicalWeightTpl<FloatType> > &w,
253  const vector<vector<ScaleFloatType> > &scale) {
254  typedef TropicalWeightTpl<FloatType> BaseType;
255  typedef PairWeight<BaseType, BaseType> PairType;
256  const BaseType zero = BaseType::Zero();
257  // Without the next special case we'd get NaNs from infinity * 0
258  if (w.Value1() == zero || w.Value2() == zero)
259  return PairType(zero, zero);
260  FloatType f1 = w.Value1().Value(), f2 = w.Value2().Value();
261  return PairType(BaseType(scale[0][0] * f1 + scale[0][1] * f2),
262  BaseType(scale[1][0] * f1 + scale[1][1] * f2));
263 }

267 template<class FloatType>
268 inline bool operator==(const LatticeWeightTpl<FloatType> &wa,
269  const LatticeWeightTpl<FloatType> &wb) {
270  // Volatile qualifier thwarts over-aggressive compiler optimizations
271  // that lead to problems esp. with NaturalLess().
272  volatile FloatType va1 = wa.Value1(), va2 = wa.Value2(),
273  vb1 = wb.Value1(), vb2 = wb.Value2();
274  return (va1 == vb1 && va2 == vb2);
275 }

294 template<class FloatType>
295 inline int Compare (const LatticeWeightTpl<FloatType> &w1,
296  const LatticeWeightTpl<FloatType> &w2) {
297  FloatType f1 = w1.Value1() + w1.Value2(),
298  f2 = w2.Value1() + w2.Value2();
299  if (f1 < f2) { return 1; } // having smaller cost means you're larger
300  // in the semiring [higher probability]
301  else if (f1 > f2) { return -1; }
302  // mathematically we should be comparing (w1.value1_-w1.value2_ < w2.value1_-w2.value2_)
303  // in the next line, but add w1.value1_+w1.value2_ = w2.value1_+w2.value2_ to both sides and
304  // divide by two, and we get the simpler equivalent form w1.value1_ < w2.value1_.
305  else if (w1.Value1() < w2.Value1()) { return 1; }
306  else if (w1.Value1() > w2.Value1()) { return -1; }
307  else { return 0; }
308 }

318 // For efficiency, override the NaturalLess template class.
319 template<class FloatType>
320 class NaturalLess<LatticeWeightTpl<FloatType> > {
321  public:
322  typedef LatticeWeightTpl<FloatType> Weight;
323 
324  NaturalLess() {}
325 
326  bool operator()(const Weight &w1, const Weight &w2) const {
327  // NaturalLess is a negative order (opposite to normal ordering).
328  // This operator () corresponds to "<" in the negative order, which
329  // corresponds to the ">" in the normal order.
330  return (Compare(w1, w2) == 1);
331  }

368 // divide w1 by w2 (on left/right/any doesn't matter as
369 // commutative).
370 template<class FloatType>
371 inline LatticeWeightTpl<FloatType> Divide(const LatticeWeightTpl<FloatType> &w1,
372  const LatticeWeightTpl<FloatType> &w2,
373  DivideType typ = DIVIDE_ANY) {
374  typedef FloatType T;
375  T a = w1.Value1() - w2.Value1(), b = w1.Value2() - w2.Value2();
376  if (a != a || b != b || a == -numeric_limits<T>::infinity()
377  || b == -numeric_limits<T>::infinity()) {
378  KALDI_WARN << "LatticeWeightTpl::Divide, NaN or invalid number produced. "
379  << "[dividing by zero?] Returning zero";
380  return LatticeWeightTpl<T>::Zero();
381  }
382  if (a == numeric_limits<T>::infinity() ||
383  b == numeric_limits<T>::infinity())
384  return LatticeWeightTpl<T>::Zero(); // not a valid number if only one is infinite.
385  return LatticeWeightTpl<T>(a, b);
386 }

416 // CompactLattice will be an acceptor (accepting the words/output-symbols),
417 // with the weights and input-symbol-seqs on the arcs.
418 // There must be a total order on W. We assume for the sake of efficiency
419 // that there is a function
420 // Compare(W w1, W w2) that returns -1 if w1 < w2, +1 if w1 > w2, and
421 // zero if w1 == w2, and Plus for type W returns (Compare(w1,w2) >= 0 ? w1 : w2).

579 // Compare is not part of the standard for weight types, but used internally for
580 // efficiency. The comparison here first compares the weight; if this is the
581 // same, it compares the string. The comparison on strings is: first compare
582 // the length, if this is the same, use lexicographical order. We can't just
583 // use the lexicographical order because this would destroy the distributive
584 // property of multiplication over addition, taking into account that addition
585 // uses Compare. The string element of "Compare" isn't super-important in
586 // practical terms; it's only needed to ensure that Plus always give consistent
587 // answers and is symmetric. It's essentially for tie-breaking, but we need to
588 // make sure all the semiring axioms are satisfied otherwise OpenFst might
589 // break.
590 
591 template<class WeightType, class IntType>
592 inline int Compare (const CompactLatticeWeightTpl<WeightType, IntType> &w1,
593  const CompactLatticeWeightTpl<WeightType, IntType> &w2) {
594  int c1 = Compare(w1.Weight(), w2.Weight());
595  if (c1 != 0) return c1;
596  int l1 = w1.String().size(), l2 = w2.String().size();
597  // Use opposite order on the string lengths, so that if the costs are the same,
598  // the shorter string wins.
599  if (l1 > l2) return -1;
600  else if (l1 < l2) return 1;
601  for(int i = 0; i < l1; i++) {
602  if (w1.String()[i] < w2.String()[i]) return -1;
603  else if (w1.String()[i] > w2.String()[i]) return 1;
604  }
605  return 0;
606 }

671 template<class WeightType, class IntType>
672 inline CompactLatticeWeightTpl<WeightType, IntType> Times(
673  const CompactLatticeWeightTpl<WeightType, IntType> &w1,
674  const CompactLatticeWeightTpl<WeightType, IntType> &w2) {
675  WeightType w = Times(w1.Weight(), w2.Weight());
676  if (w == WeightType::Zero()) {
677  return CompactLatticeWeightTpl<WeightType, IntType>::Zero();
678  // special case to ensure zero is unique
679  } else {
680  vector<IntType> v;
681  v.resize(w1.String().size() + w2.String().size());
682  typename vector<IntType>::iterator iter = v.begin();
683  iter = std::copy(w1.String().begin(), w1.String().end(), iter); // returns end of first range.
684  std::copy(w2.String().begin(), w2.String().end(), iter);
685  return CompactLatticeWeightTpl<WeightType, IntType>(w, v);
686  }

689 template<class WeightType, class IntType>
690 inline CompactLatticeWeightTpl<WeightType, IntType> Divide(const CompactLatticeWeightTpl<WeightType, IntType> &w1,
691  const CompactLatticeWeightTpl<WeightType, IntType> &w2,
692  DivideType div = DIVIDE_ANY) {
693  if (w1.Weight() == WeightType::Zero()) {
694  if (w2.Weight() != WeightType::Zero()) {
695  return CompactLatticeWeightTpl<WeightType, IntType>::Zero();
696  } else {
697  KALDI_ERR << "Division by zero [0/0]";
698  }
699  } else if (w2.Weight() == WeightType::Zero()) {
700  KALDI_ERR << "Error: division by zero";
701  }
702  WeightType w = Divide(w1.Weight(), w2.Weight());
703 
704  const vector<IntType> v1 = w1.String(), v2 = w2.String();
705  if (v2.size() > v1.size()) {
706  KALDI_ERR << "Cannot divide, length mismatch";
707  }
708  typename vector<IntType>::const_iterator v1b = v1.begin(),
709  v1e = v1.end(), v2b = v2.begin(), v2e = v2.end();
710  if (div == DIVIDE_LEFT) {
711  if (!std::equal(v2b, v2e, v1b)) { // v2 must be identical to first part of v1.
712  KALDI_ERR << "Cannot divide, data mismatch";
713  }
714  return CompactLatticeWeightTpl<WeightType, IntType>(
715  w, vector<IntType>(v1b+(v2e-v2b), v1e)); // return last part of v1.
716  } else if (div == DIVIDE_RIGHT) {
717  if (!std::equal(v2b, v2e, v1e-(v2e-v2b))) { // v2 must be identical to last part of v1.
718  KALDI_ERR << "Cannot divide, data mismatch";
719  }
720  return CompactLatticeWeightTpl<WeightType, IntType>(
721  w, vector<IntType>(v1b, v1e-(v2e-v2b))); // return first part of v1.
722 
723  } else {
724  KALDI_ERR << "Cannot divide CompactLatticeWeightTpl with DIVIDE_ANY";
725  }
726  return CompactLatticeWeightTpl<WeightType,IntType>::Zero(); // keep compiler happy.
727 }

742 template <class WeightType, class IntType>
743 inline istream &operator >>(istream &strm, CompactLatticeWeightTpl<WeightType, IntType> &w) {
744  std::string s;
745  strm >> s;
746  if (strm.fail()) {
747  return strm;
748  }
749  CHECK(FLAGS_fst_weight_separator.size() == 1);
750  size_t pos = s.find_last_of(FLAGS_fst_weight_separator); // normally ","
751  if (pos == std::string::npos) {
752  strm.clear(std::ios::badbit);
753  return strm;
754  }
755  // get parts of str before and after the separator (default: ',');
756  std::string s1(s, 0, pos), s2(s, pos+1);
757  std::istringstream strm1(s1);
758  WeightType weight;
759  strm1 >> weight;
760  w.SetWeight(weight);
761  if (strm1.fail() || !strm1.eof()) {
762  strm.clear(std::ios::badbit);
763  return strm;
764  }
765  // read string part.
766  vector<IntType> string;
767  const char *c = s2.c_str();
768  while(*c != '\0') {
769  if (*c == kStringSeparator) // '_'
770  c++;
771  char *c2;
772  long int i = strtol(c, &c2, 10);
773  if (c2 == c || static_cast<long int>(static_cast<IntType>(i)) != i) {
774  strm.clear(std::ios::badbit);
775  return strm;
776  }
777  c = c2;
778  string.push_back(static_cast<IntType>(i));
779  }
780  w.SetString(string);
781  return strm;
782 }
