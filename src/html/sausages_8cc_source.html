27 void MinimumBayesRisk::MbrDecode() {
28 
29  for (size_t counter = 0; ; counter++) {
30  NormalizeEps(&R_);
31  AccStats(); // writes to gamma_
32  double delta_Q = 0.0; // change in objective function.
33 
34  one_best_times_.clear();
35  one_best_confidences_.clear();
36 
37  // Caution: q in the line below is (q-1) in the algorithm
38  // in the paper; both R_ and gamma_ are indexed by q-1.
39  for (size_t q = 0; q < R_.size(); q++) {
40  if (opts_.decode_mbr) { // This loop updates R_ [indexed same as gamma_].
41  // gamma_[i] is sorted in reverse order so most likely one is first.
42  const std::vector<std::pair<int32, BaseFloat> > &this_gamma = gamma_[q];
43  double old_gamma = 0, new_gamma = this_gamma[0].second;
44  int32 rq = R_[q], rhat = this_gamma[0].first; // rq: old word, rhat: new.
45  for (size_t j = 0; j < this_gamma.size(); j++)
46  if (this_gamma[j].first == rq) old_gamma = this_gamma[j].second;
47  delta_Q += (old_gamma - new_gamma); // will be 0 or negative; a bound on
48  // change in error.
49  if (rq != rhat)
50  KALDI_VLOG(2) << "Changing word " << rq << " to " << rhat;
51  R_[q] = rhat;
52  }
53  // build the outputs (time, confidences),
54  if (R_[q] != 0 || opts_.print_silence) {
55  one_best_times_.push_back(times_[q]);
56  BaseFloat confidence = 0.0;
57  for (int32 j = 0; j < gamma_[q].size(); j++)
58  if (gamma_[q][j].first == R_[q]) confidence = gamma_[q][j].second;
59  one_best_confidences_.push_back(confidence);
60  }
61  }
62  KALDI_VLOG(2) << "Iter = " << counter << ", delta-Q = " << delta_Q;
63  if (delta_Q == 0) break;
64  if (counter > 100) {
65  KALDI_WARN << "Iterating too many times in MbrDecode; stopping.";
66  break;
67  }
68  }
69  if (!opts_.print_silence) RemoveEps(&R_);
70 }

94 double MinimumBayesRisk::EditDistance(int32 N, int32 Q,
95  Vector<double> &alpha,
96  Matrix<double> &alpha_dash,
97  Vector<double> &alpha_dash_arc) {
98  alpha(1) = 0.0; // = log(1). Line 5.
99  alpha_dash(1, 0) = 0.0; // Line 5.
100  for (int32 q = 1; q <= Q; q++)
101  alpha_dash(1, q) = alpha_dash(1, q-1) + l(0, r(q)); // Line 7.
102  for (int32 n = 2; n <= N; n++) {
103  double alpha_n = kLogZeroDouble;
104  for (size_t i = 0; i < pre_[n].size(); i++) {
105  const Arc &arc = arcs_[pre_[n][i]];
106  alpha_n = LogAdd(alpha_n, alpha(arc.start_node) + arc.loglike);
107  }
108  alpha(n) = alpha_n; // Line 10.
109  // Line 11 omitted: matrix was initialized to zero.
110  for (size_t i = 0; i < pre_[n].size(); i++) {
111  const Arc &arc = arcs_[pre_[n][i]];
112  int32 s_a = arc.start_node, w_a = arc.word;
113  BaseFloat p_a = arc.loglike;
114  for (int32 q = 0; q <= Q; q++) {
115  if (q == 0) {
116  alpha_dash_arc(q) = // line 15.
117  alpha_dash(s_a, q) + l(w_a, 0, true);
118  } else { // a1,a2,a3 are the 3 parts of min expression of line 17.
119  int32 r_q = r(q);
120  double a1 = alpha_dash(s_a, q-1) + l(w_a, r_q),
121  a2 = alpha_dash(s_a, q) + l(w_a, 0, true),
122  a3 = alpha_dash_arc(q-1) + l(0, r_q);
123  alpha_dash_arc(q) = std::min(a1, std::min(a2, a3));
124  }
125  // line 19:
126  alpha_dash(n, q) += Exp(alpha(s_a) + p_a - alpha(n)) * alpha_dash_arc(q);
127  }
128  }
129  }
130  return alpha_dash(N, Q); // line 23.
131 }

134 void MinimumBayesRisk::AccStats() {
135  using std::map;
136 
137  int32 N = static_cast<int32>(pre_.size()) - 1,
138  Q = static_cast<int32>(R_.size());
139 
140  Vector<double> alpha(N+1); // index (1...N)
141  Matrix<double> alpha_dash(N+1, Q+1); // index (1...N, 0...Q)
142  Vector<double> alpha_dash_arc(Q+1); // index 0...Q
143  Matrix<double> beta_dash(N+1, Q+1); // index (1...N, 0...Q)
144  Vector<double> beta_dash_arc(Q+1); // index 0...Q
145  std::vector<char> b_arc(Q+1); // integer in {1,2,3}; index 1...Q
146  std::vector<map<int32, double> > gamma(Q+1); // temp. form of gamma.
147  // index 1...Q [word] -> occ.
148 
149  // The tau arrays below are the sums over words of the tau_b
150  // and tau_e timing quantities mentioned in Appendix C of
151  // the paper... we are using these to get averaged times for
152  // the sausage bins, not specifically for the 1-best output.
153  Vector<double> tau_b(Q+1), tau_e(Q+1);
154 
155  double Ltmp = EditDistance(N, Q, alpha, alpha_dash, alpha_dash_arc);
156  if (L_ != 0 && Ltmp > L_) { // L_ != 0 is to rule out 1st iter.
157  KALDI_WARN << "Edit distance increased: " << Ltmp << " > "
158  << L_;
159  }
160  L_ = Ltmp;
161  KALDI_VLOG(2) << "L = " << L_;
162  // omit line 10: zero when initialized.
163  beta_dash(N, Q) = 1.0; // Line 11.
164  for (int32 n = N; n >= 2; n--) {
165  for (size_t i = 0; i < pre_[n].size(); i++) {
166  const Arc &arc = arcs_[pre_[n][i]];
167  int32 s_a = arc.start_node, w_a = arc.word;
168  BaseFloat p_a = arc.loglike;
169  alpha_dash_arc(0) = alpha_dash(s_a, 0) + l(w_a, 0, true); // line 14.
170  for (int32 q = 1; q <= Q; q++) { // this loop == lines 15-18.
171  int32 r_q = r(q);
172  double a1 = alpha_dash(s_a, q-1) + l(w_a, r_q),
173  a2 = alpha_dash(s_a, q) + l(w_a, 0, true),
174  a3 = alpha_dash_arc(q-1) + l(0, r_q);
175  if (a1 <= a2) {
176  if (a1 <= a3) { b_arc[q] = 1; alpha_dash_arc(q) = a1; }
177  else { b_arc[q] = 3; alpha_dash_arc(q) = a3; }
178  } else {
179  if (a2 <= a3) { b_arc[q] = 2; alpha_dash_arc(q) = a2; }
180  else { b_arc[q] = 3; alpha_dash_arc(q) = a3; }
181  }
182  }
183  beta_dash_arc.SetZero(); // line 19.
184  for (int32 q = Q; q >= 1; q--) {
185  // line 21:
186  beta_dash_arc(q) += Exp(alpha(s_a) + p_a - alpha(n)) * beta_dash(n, q);
187  switch (static_cast<int>(b_arc[q])) { // lines 22 and 23:
188  case 1:
189  beta_dash(s_a, q-1) += beta_dash_arc(q);
190  // next: gamma(q, w(a)) += beta_dash_arc(q)
191  AddToMap(w_a, beta_dash_arc(q), &(gamma[q]));
192  // next: accumulating times, see decl for tau_b,tau_e
193  tau_b(q) += state_times_[s_a] * beta_dash_arc(q);
194  tau_e(q) += state_times_[n] * beta_dash_arc(q);
195  break;
196  case 2:
197  beta_dash(s_a, q) += beta_dash_arc(q);
198  break;
199  case 3:
200  beta_dash_arc(q-1) += beta_dash_arc(q);
201  // next: gamma(q, epsilon) += beta_dash_arc(q)
202  AddToMap(0, beta_dash_arc(q), &(gamma[q]));
203  // next: accumulating times, see decl for tau_b,tau_e
204  // WARNING: there was an error in Appendix C. If we followed
205  // the instructions there the next line would say state_times_[sa], but
206  // it would be wrong. I will try to publish an erratum.
207  tau_b(q) += state_times_[n] * beta_dash_arc(q);
208  tau_e(q) += state_times_[n] * beta_dash_arc(q);
209  break;
210  default:
211  KALDI_ERR << "Invalid b_arc value"; // error in code.
212  }
213  }
214  beta_dash_arc(0) += Exp(alpha(s_a) + p_a - alpha(n)) * beta_dash(n, 0);
215  beta_dash(s_a, 0) += beta_dash_arc(0); // line 26.
216  }
217  }
218  beta_dash_arc.SetZero(); // line 29.
219  for (int32 q = Q; q >= 1; q--) {
220  beta_dash_arc(q) += beta_dash(1, q);
221  beta_dash_arc(q-1) += beta_dash_arc(q);
222  AddToMap(0, beta_dash_arc(q), &(gamma[q]));
223  // the statements below are actually redundant because
224  // state_times_[1] is zero.
225  tau_b(q) += state_times_[1] * beta_dash_arc(q);
226  tau_e(q) += state_times_[1] * beta_dash_arc(q);
227  }
228  for (int32 q = 1; q <= Q; q++) { // a check (line 35)
229  double sum = 0.0;
230  for (map<int32, double>::iterator iter = gamma[q].begin();
231  iter != gamma[q].end(); ++iter) sum += iter->second;
232  if (fabs(sum - 1.0) > 0.1)
233  KALDI_WARN << "sum of gamma[" << q << ",s] is " << sum;
234  }
235  // The next part is where we take gamma, and convert
236  // to the class member gamma_, which is using a different
237  // data structure and indexed from zero, not one.
238  gamma_.clear();
239  gamma_.resize(Q);
240  for (int32 q = 1; q <= Q; q++) {
241  for (map<int32, double>::iterator iter = gamma[q].begin();
242  iter != gamma[q].end(); ++iter)
243  gamma_[q-1].push_back(std::make_pair(iter->first, static_cast<BaseFloat>(iter->second)));
244  // sort gamma_[q-1] from largest to smallest posterior.
245  GammaCompare comp;
246  std::sort(gamma_[q-1].begin(), gamma_[q-1].end(), comp);
247  }
248  // We do the same conversion for the state times tau_b and tau_e:
249  // they get turned into the times_ data member, which has zero-based
250  // indexing.
251  times_.clear();
252  times_.resize(Q);
253  for (int32 q = 1; q <= Q; q++) {
254  times_[q-1].first = tau_b(q);
255  times_[q-1].second = tau_e(q);
256  if (times_[q-1].first > times_[q-1].second) // this is quite bad.
257  KALDI_WARN << "Times out of order";
258  if (q > 1 && times_[q-2].second > times_[q-1].first) {
259  // We previously had a warning here, but now we'll just set both
260  // those values to their average. It's quite possible for this
261  // condition to happen, but it seems like it would have a bad effect
262  // on the downstream processing, so we fix it.
263  double avg = 0.5 * (times_[q-2].second + times_[q-1].first);
264  times_[q-2].second = times_[q-1].first = avg;
265  }
266  }
267 }

269 void MinimumBayesRisk::PrepareLatticeAndInitStats(CompactLattice *clat) {
270  KALDI_ASSERT(clat != NULL);
271 
272  CreateSuperFinal(clat); // Add super-final state to clat... this is
273  // one of the requirements of the MBR algorithm, as mentioned in the
274  // paper (i.e. just one final state).
275 
276  // Topologically sort the lattice, if not already sorted.
277  kaldi::uint64 props = clat->Properties(fst::kFstProperties, false);
278  if (!(props & fst::kTopSorted)) {
279  if (fst::TopSort(clat) == false)
280  KALDI_ERR << "Cycles detected in lattice.";
281  }
282  CompactLatticeStateTimes(*clat, &state_times_); // work out times of
283  // the states in clat
284  state_times_.push_back(0); // we'll convert to 1-based numbering.
285  for (size_t i = state_times_.size()-1; i > 0; i--)
286  state_times_[i] = state_times_[i-1];
287 
288  // Now we convert the information in "clat" into a special internal
289  // format (pre_, post_ and arcs_) which allows us to access the
290  // arcs preceding any given state.
291  // Note: in our internal format the states will be numbered from 1,
292  // which involves adding 1 to the OpenFst states.
293  int32 N = clat->NumStates();
294  pre_.resize(N+1);
295 
296  // Careful: "Arc" is a class-member struct, not an OpenFst type of arc as one
297  // would normally assume.
298  for (int32 n = 1; n <= N; n++) {
299  for (fst::ArcIterator<CompactLattice> aiter(*clat, n-1);
300  !aiter.Done();
301  aiter.Next()) {
302  const CompactLatticeArc &carc = aiter.Value();
303  Arc arc; // in our local format.
304  arc.word = carc.ilabel; // == carc.olabel
305  arc.start_node = n;
306  arc.end_node = carc.nextstate + 1; // convert to 1-based.
307  arc.loglike = - (carc.weight.Weight().Value1() +
308  carc.weight.Weight().Value2());
309  // loglike: sum graph/LM and acoustic cost, and negate to
310  // convert to loglikes. We assume acoustic scaling is already done.
311 
312  pre_[arc.end_node].push_back(arcs_.size()); // record index of this arc.
313  arcs_.push_back(arc);
314  }
315  }
316 }

318 MinimumBayesRisk::MinimumBayesRisk(const CompactLattice &clat_in,
319  MinimumBayesRiskOptions opts) : opts_(opts) {
320  CompactLattice clat(clat_in); // copy.
321 
322  PrepareLatticeAndInitStats(&clat);
323 
324  // We don't need to look at clat.Start() or clat.Final(state):
325  // we know clat.Start() == 0 since it's topologically sorted,
326  // and clat.Final(state) is Zero() except for One() at the last-
327  // numbered state, thanks to CreateSuperFinal and the topological
328  // sorting.
329 
330  { // Now set R_ to one best in the FST.
331  RemoveAlignmentsFromCompactLattice(&clat); // will be more efficient
332  // in best-path if we do this.
333  Lattice lat;
334  ConvertLattice(clat, &lat); // convert from CompactLattice to Lattice.
335  fst::VectorFst<fst::StdArc> fst;
336  ConvertLattice(lat, &fst); // convert from lattice to normal FST.
337  fst::VectorFst<fst::StdArc> fst_shortest_path;
338  fst::ShortestPath(fst, &fst_shortest_path); // take shortest path of FST.
339  std::vector<int32> alignment, words;
340  fst::TropicalWeight weight;
341  GetLinearSymbolSequence(fst_shortest_path, &alignment, &words, &weight);
342  KALDI_ASSERT(alignment.empty()); // we removed the alignment.
343  R_ = words;
344  L_ = 0.0; // Set current edit-distance to 0 [just so we know
345  // when we're on the 1st iter.]
346  }
347 
348  MbrDecode();
349 
350 }
