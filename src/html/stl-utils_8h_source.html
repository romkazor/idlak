46 template<typename T>
47 inline bool IsSorted(const std::vector<T> &vec) {
48  typename std::vector<T>::const_iterator iter = vec.begin(), end = vec.end();
49  if (iter == end) return true;
50  while (1) {
51  typename std::vector<T>::const_iterator next_iter = iter;
52  ++next_iter;
53  if (next_iter == end) return true; // end of loop and nothing out of order
54  if (*next_iter < *iter) return false;
55  iter = next_iter;
56  }
57 }

87  // copies members of s into v, in sorted order from lowest to highest
88  // (because the set was in sorted order).

217 template<typename Int>
218 struct VectorHasher { // hashing function for vector<Int>.
219  size_t operator()(const std::vector<Int> &x) const noexcept {
220  size_t ans = 0;
221  typename std::vector<Int>::const_iterator iter = x.begin(), end = x.end();
222  for (; iter != end; ++iter) {
223  ans *= kPrime;
224  ans += *iter;
225  }
226  return ans;
227  }
228  VectorHasher() { // Check we're instantiated with an integer type.
229  KALDI_ASSERT_IS_INTEGER_TYPE(Int);
230  }
231  private:
232  static const int kPrime = 7853;
233 };

289 template<typename I, typename F>
290 inline void MergePairVectorSumming(std::vector<std::pair<I, F> > *vec) {
291  KALDI_ASSERT_IS_INTEGER_TYPE(I);
292  CompareFirstMemberOfPair<I, F> c;
293  std::sort(vec->begin(), vec->end(), c); // sort on 1st element.
294  typename std::vector<std::pair<I, F> >::iterator out = vec->begin(),
295  in = vec->begin(), end = vec->end();
296  // special case: while there is nothing to be changed, skip over
297  // initial input (avoids unnecessary copying).
298  while (in + 1 < end && in[0].first != in[1].first && in[0].second != 0.0) {
299  in++;
300  out++;
301  }
302  while (in < end) {
303  // We reach this point only at the first element of
304  // each stretch of identical .first elements.
305  *out = *in;
306  ++in;
307  while (in < end && in->first == out->first) {
308  out->second += in->second; // this is the merge operation.
309  ++in;
310  }
311  if (out->second != static_cast<F>(0)) // Don't keep zero elements.
312  out++;
313  }
314  vec->erase(out, end);
315 }
