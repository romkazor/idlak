102 void TestableArpaFileParser::ConsumeNGram(const NGram& ngram) {
103  KALDI_ASSERT(header_available_);
104  KALDI_ASSERT(!read_complete_);
105  KALDI_ASSERT(ngram.words.size() <= NgramCounts().size());
106  KALDI_ASSERT(ngram.words.size() >= last_order_);
107  last_order_ = ngram.words.size();
108 
109  NGramTestData entry = { 0 };
110  entry.line_number = LineNumber();
111  entry.logprob = ngram.logprob;
112  entry.backoff = ngram.backoff;
113  std::copy(ngram.words.begin(), ngram.words.end(), entry.words);
114  ngrams_.push_back(entry);
115 }

123 bool CompareNgrams(const NGramTestData &actual,
124  NGramTestData expected) {
125  expected.logprob *= Log(10.0);
126  expected.backoff *= Log(10.0);
127  if (actual.line_number != expected.line_number
128  || !std::equal(actual.words, actual.words + kMaxOrder,
129  expected.words)
130  || !ApproxEqual(actual.logprob, expected.logprob)
131  || !ApproxEqual(actual.backoff, expected.backoff)) {
132  KALDI_WARN << "Actual n-gram [" << actual
133  << "] differs from expected [" << expected << "]";
134  return false;
135  }
136  return true;
137 }
