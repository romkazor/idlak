151  MapType::iterator iter = map_.find(tuple);
152  if (iter == map_.end()) { // not in map.

154  map_[tuple] = output_state;

163  void ProcessFinal(Tuple tuple, StateId output_state) {
164  // ProcessFinal is only called if the input_state has
165  // final-prob of One(). [else it should be zero. This
166  // is because we called CreateSuperFinal().]
167 
168  if (tuple.comp_state.IsEmpty()) { // computation state doesn't have
169  // anything pending.
170  std::vector<int32> empty_vec;
171  CompactLatticeWeight cw(tuple.comp_state.FinalWeight(), empty_vec);
172  lat_out_->SetFinal(output_state, Plus(lat_out_->Final(output_state), cw));
173  } else {
174  // computation state has something pending, i.e. input or
175  // output symbols that need to be flushed out. Note: OutputArc() would
176  // have returned false or we wouldn't have been called, so we have to
177  // force it out.
178  CompactLatticeArc lat_arc;
179  tuple.comp_state.OutputArcForce(tmodel_, opts_, &lat_arc, &error_);
180  lat_arc.nextstate = GetStateForTuple(tuple, true); // true == add to queue.
181  // The final-prob stuff will get called again from ProcessQueueElement().
182  // Note: because we did CreateSuperFinal(), this final-state on the input
183  // lattice will have no output arcs (and unit final-prob), so there will be
184  // no complications with processing the arcs from this state (there won't
185  // be any).
186  KALDI_ASSERT(output_state != lat_arc.nextstate);
187  lat_out_->AddArc(output_state, lat_arc);
188  }
189  }

192  void ProcessQueueElement() {
193  KALDI_ASSERT(!queue_.empty());
194  Tuple tuple = queue_.back().first;
195  StateId output_state = queue_.back().second;
196  queue_.pop_back();
197 
198  // First thing is-- we see whether the computation-state has something
199  // pending that it wants to output. In this case we don't do
200  // anything further. This is a chosen behavior similar to the
201  // epsilon-sequencing rules encoded by the filters in
202  // composition.
203  CompactLatticeArc lat_arc;
204  Tuple tuple2(tuple); // temp
205  if (tuple.comp_state.OutputPhoneArc(tmodel_, opts_, &lat_arc, &error_) ||
206  tuple.comp_state.OutputWordArc(tmodel_, opts_, &lat_arc, &error_)) {
207  // note: this function changes the tuple (when it returns true).
208  lat_arc.nextstate = GetStateForTuple(tuple, true); // true == add to queue,
209  // if not already present.
210  KALDI_ASSERT(output_state != lat_arc.nextstate);
211  lat_out_->AddArc(output_state, lat_arc);
212  } else {
213  // when there's nothing to output, we'll process arcs from the input-state.
214  // note: it would in a sense be valid to do both (i.e. process the stuff
215  // above, and also these), but this is a bit like the epsilon-sequencing
216  // stuff in composition: we avoid duplicate arcs by doing it this way.
217 
218  if (lat_.Final(tuple.input_state) != CompactLatticeWeight::Zero()) {
219  KALDI_ASSERT(lat_.Final(tuple.input_state) == CompactLatticeWeight::One());
220  // ... since we did CreateSuperFinal.
221  ProcessFinal(tuple, output_state);
222  }
223  // Now process the arcs. Note: final-state shouldn't have any arcs.
224  for(fst::ArcIterator<CompactLattice> aiter(lat_, tuple.input_state);
225  !aiter.Done(); aiter.Next()) {
226  const CompactLatticeArc &arc = aiter.Value();
227  Tuple next_tuple(tuple);
228  LatticeWeight weight;
229  next_tuple.comp_state.Advance(arc, opts_, &weight);
230  next_tuple.input_state = arc.nextstate;
231  StateId next_output_state = GetStateForTuple(next_tuple, true); // true == add to queue,
232  // if not already present.
233  // We add an epsilon arc here (as the input and output happens
234  // separately)... the epsilons will get removed later.
235  KALDI_ASSERT(next_output_state != output_state);
236  lat_out_->AddArc(output_state,
237  CompactLatticeArc(0, 0,
238  CompactLatticeWeight(weight, std::vector<int32>()),
239  next_output_state));
240  }
241  }
242  }

262  bool AlignLattice() {
263  lat_out_->DeleteStates();
264  if (lat_.Start() == fst::kNoStateId) {
265  KALDI_WARN << "Trying to word-align empty lattice.";
266  return false;
267  }
268  ComputationState initial_comp_state;
269  Tuple initial_tuple(lat_.Start(), initial_comp_state);
270  StateId start_state = GetStateForTuple(initial_tuple, true); // True = add this to queue.
271  lat_out_->SetStart(start_state);
272 
273  while (!queue_.empty())
274  ProcessQueueElement();
275 
276  if (opts_.remove_epsilon)
277  RemoveEpsilonsFromLattice();
278 
279  return !error_;
280  }

292 bool LatticePhoneAligner::ComputationState::OutputPhoneArc(
293  const TransitionModel &tmodel,
294  const PhoneAlignLatticeOptions &opts,
295  CompactLatticeArc *arc_out,
296  bool *error) {
297  if (transition_ids_.empty()) return false;
298  int32 phone = tmodel.TransitionIdToPhone(transition_ids_[0]);
299  // we assume the start of transition_ids_ is the start of the phone;
300  // this is a precondition.
301  size_t len = transition_ids_.size(), i;
302  // Keep going till we reach a "final" transition-id; note, if
303  // reorder==true, we have to go a bit further after this.
304  for (i = 0; i < len; i++) {
305  int32 tid = transition_ids_[i];
306  int32 this_phone = tmodel.TransitionIdToPhone(tid);
307  if (this_phone != phone && ! *error) { // error condition: should have
308  // reached final transition-id first.
309  *error = true;
310  KALDI_WARN << phone << " -> " << this_phone;
311  KALDI_WARN << "Phone changed before final transition-id found "
312  "[broken lattice or mismatched model or wrong --reorder option?]";
313  }
314  if (tmodel.IsFinal(tid))
315  break;
316  }
317  if (i == len) return false; // fell off loop.
318  i++; // go past the one for which IsFinal returned true.
319  if (opts.reorder) // we have to consume the following self-loop transition-ids.
320  while (i < len && tmodel.IsSelfLoop(transition_ids_[i])) i++;
321  if (i == len) return false; // we don't know if it ends here... so can't output arc.
322 
323  // interpret i as the number of transition-ids to consume.
324  std::vector<int32> tids_out(transition_ids_.begin(),
325  transition_ids_.begin()+i);
326 
327  Label output_label = 0;
328  if (!word_labels_.empty()) {
329  output_label = word_labels_[0];
330  word_labels_.erase(word_labels_.begin(), word_labels_.begin()+1);
331  }
332  if (opts.replace_output_symbols)
333  output_label = phone;
334  *arc_out = CompactLatticeArc(output_label, output_label,
335  CompactLatticeWeight(weight_, tids_out),
336  fst::kNoStateId);
337  transition_ids_.erase(transition_ids_.begin(), transition_ids_.begin()+i);
338  weight_ = LatticeWeight::One(); // we just output the weight.
339  return true;
340 }

361 void LatticePhoneAligner::ComputationState::OutputArcForce(
362  const TransitionModel &tmodel,
363  const PhoneAlignLatticeOptions &opts,
364  CompactLatticeArc *arc_out,
365  bool *error) {
366  KALDI_ASSERT(!IsEmpty());
367 
368  int32 phone = -1; // This value -1 will never be used,
369  // although it might not be obvious from superficially checking
370  // the code. IsEmpty() would be true if we had transition_ids_.empty()
371  // and opts.replace_output_symbols, so we would already die by assertion;
372  // in fact, this function would neve be called.
373 
374  if (!transition_ids_.empty()) { // Do some checking here.
375  int32 tid = transition_ids_[0];
376  phone = tmodel.TransitionIdToPhone(tid);
377  int32 num_final = 0;
378  for (int32 i = 0; i < transition_ids_.size(); i++) { // A check.
379  int32 this_tid = transition_ids_[i];
380  int32 this_phone = tmodel.TransitionIdToPhone(this_tid);
381  bool is_final = tmodel.IsFinal(this_tid); // should be exactly one.
382  if (is_final) num_final++;
383  if (this_phone != phone && ! *error) {
384  KALDI_WARN << "Mismatch in phone: error in lattice or mismatched transition model?";
385  *error = true;
386  }
387  }
388  if (num_final != 1 && ! *error) {
389  KALDI_WARN << "Problem phone-aligning lattice: saw " << num_final
390  << " final-states in last phone in lattice (forced out?) "
391  << "Producing partial lattice.";
392  *error = true;
393  }
394  }
395 
396  Label output_label = 0;
397  if (!word_labels_.empty()) {
398  output_label = word_labels_[0];
399  word_labels_.erase(word_labels_.begin(), word_labels_.begin()+1);
400  }
401  if (opts.replace_output_symbols)
402  output_label = phone;
403  *arc_out = CompactLatticeArc(output_label, output_label,
404  CompactLatticeWeight(weight_, transition_ids_),
405  fst::kNoStateId);
406  transition_ids_.clear();
407  weight_ = LatticeWeight::One(); // we just output the weight.
408 }
