47  const CuVectorBase<Real> &b) {
48  //MatrixIndexT a_dim = a.Dim();
49  KALDI_ASSERT(a.Dim() == b.Dim());

63 }

81 template<typename Real>
82 Real VecMatVec(const CuVectorBase<Real> &v1, const CuMatrixBase<Real> &M,
83  const CuVectorBase<Real> &v2) {
84  KALDI_ASSERT(v1.Dim() == M.NumRows() && M.NumCols() == v2.Dim());
85  if (v1.Dim() > v2.Dim()) { // do v2*M first
86  CuVector<Real> v2M(v1.Dim());
87  v2M.AddMatVec(1.0, M, kNoTrans, v2, 0.0);
88  return VecVec(v2M, v1);
89  } else { // do v1*M first
90  CuVector<Real> v1M(v2.Dim());
91  v1M.AddMatVec(1.0, M, kTrans, v1, 0.0);
92  return VecVec(v1M, v2);
93  }
94 }

290 template<typename Real>
291 Real CuVectorBase<Real>::Sum() const {
292  if (dim_ == 0)
293  return 0.0;
294 #if HAVE_CUDA == 1
295  if (CuDevice::Instantiate().Enabled()) {
296  Real result;
297  CuTimer tim;
298 
299  // Small vectors are copied to RAM and reduced on CPU.
300  // The length is chosen by cu-vector-speed-test
301  if (dim_ < 4096) {
302  Vector<Real> ans_cpu(*this);
303  result = ans_cpu.Sum();
304  } else {
305  // Use no more than 256 blocks (still too many?)
306  int dimBlock = CU1DBLOCK;
307  int dimGrid = n_blocks(dim_, dimBlock);
308  if (dimGrid > 256) {
309  dimGrid = 256;
310  }
311  CuVector<Real> ans(dimGrid, kUndefined);
312  cuda_vec_sum(dimGrid, dimBlock, data_, ans.Data(), dim_, 1);
313  CU_SAFE_CALL(cudaGetLastError());
314  Vector<Real> ans_cpu(ans);
315  result = ans_cpu.Sum();
316  }
317 
318  CuDevice::Instantiate().AccuProfile(__func__, tim);
319  return result;
320  } else
321 #endif
322  {
323  return Vec().Sum();
324  }

477 template<typename Real>
478 void CuVectorBase<Real>::AddMatVec(const Real alpha,
479  const CuMatrixBase<Real> &M,
480  MatrixTransposeType trans,
481  const CuVectorBase<Real> &v,
482  const Real beta) {
483  KALDI_ASSERT((trans == kNoTrans && M.NumCols() == v.dim_ && M.NumRows() == dim_) ||
484  (trans == kTrans && M.NumRows() == v.dim_ && M.NumCols() == dim_));
485  KALDI_ASSERT(&v != this);
486 #if HAVE_CUDA == 1
487  if (CuDevice::Instantiate().Enabled()) {
488  if (dim_ == 0) return;
489  CuTimer tim;
490 
491  // Everything is backwards in CuBlas. We need to reverse rows, columns,
492  // transpose-ness.
493  CUBLAS_SAFE_CALL(cublas_gemv(GetCublasHandle(),
494  (trans==kTrans? CUBLAS_OP_N:CUBLAS_OP_T),
495  M.NumCols(), M.NumRows(), alpha, M.Data(),
496  M.Stride(), v.Data(), 1, beta, data_, 1));
497 
498  CuDevice::Instantiate().AccuProfile(__func__, tim);
499  } else
500 #endif
501  {
502  Vec().AddMatVec(alpha,M.Mat(),trans,v.Vec(),beta);
503  }
504 }

582 template<typename Real>
583 void CuVectorBase<Real>::AddDiagMatMat(Real alpha, const CuMatrixBase<Real> &M,
584  MatrixTransposeType transM,
585  const CuMatrixBase<Real> &N,
586  MatrixTransposeType transN, Real beta) {
587 #if HAVE_CUDA == 1
588  if (CuDevice::Instantiate().Enabled()) {
589  CuTimer tim;
590 
591  if (transM != transN) {
592  KALDI_ASSERT(M.NumCols() == N.NumCols());
593  KALDI_ASSERT(M.NumRows() == N.NumRows());
594  if (transM == kNoTrans) {
595  // Case 1: diag(M*N') == sum(M.*N, 2)
596  // 1D grid and 1D block. One block per row of N.
597  // 1D grid expands along the column of N.
598  int dimBlock(CU1DBLOCK);
599  int dimGrid(M.NumRows());
600  cuda_add_diag_mat_mat_MNT(dimGrid, dimBlock, alpha, M.Data(), M.Dim(),
601  N.Data(), N.Stride(), beta, data_);
602  } else {
603  // Case 2: diag(M'*N) == sum(M.*N, 1)
604  // 16x16 or 8x32 2D block for coalesced memory access.
605  // Grid shape is designed as follows,
606  // 1. for small matrices, use 1D grid with only 1 row of 16x16 block,
607  // to avoid multiple kernel launch;
608  // 2. for large enough matrices (no matter thin or fat),
609  // use 1- or 2-D grid so that the grid contains
610  // at least and not much larger than 'kOptNumBlocks' blocks
611  // to fully utilize the GPU;
612  const int32 warpSize = 32;
613  const int32 kOptNumBlocks = 512;
614  const int32 tile_dim =
615  (N.NumRows() < 4096 && N.NumCols() < kOptNumBlocks * warpSize) ?
616  16 : 32;
617  dim3 dimBlock(tile_dim, CU1DBLOCK / tile_dim);
618  dim3 dimGrid(n_blocks(N.NumCols(), dimBlock.x),
619  n_blocks(N.NumRows(), dimBlock.y));
620  dimGrid.y = std::min(dimGrid.y, (kOptNumBlocks - 1) / dimGrid.x + 1);
621  dimGrid.y = tile_dim == 16 ? 1 : dimGrid.y;
622  if (dimGrid.y > 1) {
623  CuMatrix<Real> buf(dimGrid.y, N.NumCols());
624  cuda_add_diag_mat_mat_MTN(dimGrid, dimBlock, Real(1), M.Data(),
625  M.Stride(), N.Data(), N.Dim(), Real(0),
626  buf.Data(), buf.Stride());
627  this->AddRowSumMat(alpha, buf, beta);
628  } else {
629  cuda_add_diag_mat_mat_MTN(dimGrid, dimBlock, alpha, M.Data(),
630  M.Stride(), N.Data(), N.Dim(), beta, data_,
631  dim_);
632  }
633  }
634  } else {
635  KALDI_ASSERT(M.NumCols() == N.NumRows());
636  KALDI_ASSERT(N.NumCols() == M.NumRows());
637  if (transM == kNoTrans) {
638  // Case 3: diag(M*N) == sum(M'.*N, 1)
639  // 16x16 or 8x32 2D block for matrix transpose and coalesced memory access.
640  // One block per 'tile_dim' columns of N.
641  // 1D grid expands along the row of N.
642  int tile_dim =
643  sizeof(Real) == sizeof(float) && N.NumCols() >= 2048 ? 32 : 16;
644  dim3 dimBlock(tile_dim, CU1DBLOCK / tile_dim);
645  dim3 dimGrid(n_blocks(N.NumCols(), tile_dim));
646  cuda_add_diag_mat_mat_MN(dimGrid, dimBlock, alpha, M.Data(), M.Stride(),
647  N.Data(), N.Dim(), beta, data_);
648  } else {
649  // Case 4: diag(M'*N') == sum(N'.*M, 1)
650  // Same kernel and config as case 3 except M and N are swapped.
651  int tile_dim =
652  sizeof(Real) == sizeof(float) && N.NumCols() >= 2048 ? 32 : 16;
653  dim3 dimBlock(tile_dim, CU1DBLOCK / tile_dim);
654  dim3 dimGrid(n_blocks(M.NumCols(), tile_dim));
655  cuda_add_diag_mat_mat_MN(dimGrid, dimBlock, alpha, N.Data(), N.Stride(),
656  M.Data(), M.Dim(), beta, data_);
657  }
658  }
659  CU_SAFE_CALL(cudaGetLastError());
660 
661  CuDevice::Instantiate().AccuProfile(__func__, tim);
662  } else
663 #endif
664  {
665  Vec().AddDiagMatMat(alpha, M.Mat(), transM, N.Mat(), transN, beta);
666  }
