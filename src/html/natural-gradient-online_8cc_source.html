46 void OnlineNaturalGradient::InitOrthonormalSpecial(CuMatrixBase<BaseFloat> *R) {
47  int32 num_rows = R->NumRows(), num_cols = R->NumCols();
48  KALDI_ASSERT(num_cols >= num_rows);
49  R->SetZero();
50  std::vector<MatrixElement<BaseFloat> > elems;
51  elems.reserve(num_cols);
52  BaseFloat first_elem = 1.1;
53  for (int32 r = 0; r < num_rows; r++) {
54  std::vector<int32> cols; // columns that have an entry for this row
55  for (int32 c = r; c < num_cols; c += num_rows)
56  cols.push_back(c);
57  BaseFloat normalizer = 1.0 / sqrt(first_elem * first_elem +
58  cols.size() - 1);
59  for (size_t i = 0; i < cols.size(); i++) {
60  int32 c = cols[i];
61  MatrixElement<BaseFloat> e = { r, c,
62  normalizer * (i == 0 ? first_elem :
63  BaseFloat(1.0)) };
64  elems.push_back(e);
65  }
66  }
67  R->AddElements(1.0, elems);
68 }

71 void OnlineNaturalGradient::InitDefault(int32 D) {
72  if (rank_ >= D) {
73  KALDI_WARN << "Rank " << rank_ << " of online preconditioner is >= dim " << D
74  << ", setting it to "
75  << (D - 1) << " (but this is probably still too high)";
76  rank_ = D - 1;
77  }
78  if (rank_ == 0) {
79  // Dimension of input data was 1, so the natural gradient preconditioner
80  // would always be the unit matrix.
81  // We'll handle this as a special case, for generality.
82  return;
83  }
84  KALDI_ASSERT(num_samples_history_ > 0.0 && num_samples_history_ <= 1.0e+06);
85  KALDI_ASSERT((num_minibatches_history_ == 0.0 ||
86  num_minibatches_history_ > 1.0) &&
87  num_minibatches_history_ < 1.0e+06);
88  KALDI_ASSERT(alpha_ >= 0.0);
89  KALDI_ASSERT(rank_ > 0);
90  KALDI_ASSERT(epsilon_ > 0.0 && epsilon_ <= 1.0e-05); // plausible values.
91  KALDI_ASSERT(delta_ > 0.0 && delta_ <= 1.0e-02); // plausible values.
92 
93  // to initialize, in the equation
94  // F_t =(def) R_t^T D_t R_t + \rho_t I
95  // we will set the orthogonal R_t to a special orthogonal matrix with no zero
96  // rows or columns (see the function), rho_t to epsilon,
97  // and D_t to epsilon. But we don't store R_t directly. Instead, we store
98  // W_t =(def) E_t^{0.5} R_t,
99  // where E_t =(def) 1/\beta_t (D_t^{-1} + 1/\beta_t I)^{-1}
100  // from (eqn:tii),
101  // e_{tii} = 1/(\beta_t/d_{tii} + 1),
102  // where
103  // \beta_t =(def) \rho_t + \alpha/D tr(F_t)
104  // = epsilon + alpha/D * (epsilon * D + epsilon * rank)
105  // = epsilon * (1 + alpha * (D + rank) / D)
106  // And d_{tii} is epsilon, so
107  // e_{tii} = 1/((1 + alpha * (D + rank) / D) + 1) [for each i.]
108  // = 1/(2 + alpha * (D + rank) / D)).
109  BaseFloat epsilon = epsilon_; // we could make this a bit more.
110  rho_t_ = epsilon;
111  d_t_.Resize(rank_, kUndefined);
112  d_t_.Set(epsilon);
113  W_t_.Resize(rank_, D, kUndefined);
114  // after the next line, W_ will store the orthogonal matrix R_t.
115  InitOrthonormalSpecial(&W_t_);
116  BaseFloat E_tii = 1.0 / ( 2.0 + (D + rank_) * alpha_ / D );
117  // W_t =(def) E_t^{0.5} R_t.
118  W_t_.Scale(sqrt(E_tii));
119  t_ = 0;
120 }

122 void OnlineNaturalGradient::Init(const CuMatrixBase<BaseFloat> &R0) {
123  int32 D = R0.NumCols();
124  // for locking reasons it's better to use a different object.
125  OnlineNaturalGradient this_copy(*this);
126  this_copy.InitDefault(D);
127  this_copy.t_ = 1; // Prevent recursion to Init() again.
128 
129  CuMatrix<BaseFloat> R0_copy(R0.NumRows(), R0.NumCols(), kUndefined);
130  // 'num_iters' is number of iterations with the same data from a pseudorandom
131  // start. this is a faster way of starting than doing eigenvalue
132  // decomposition.
133  //
134  // Note: we only do three iterations of initialization if we have enough data
135  // that it's reasonably possible to estimate the subspace of dimension
136  // this_copy.rank_. If we don't have more than that many rows in our initial
137  // minibatch R0, we just do one iteration... this gives us almost exactly
138  // (barring small effects due to epsilon_ > 0) the row subspace of R0 after
139  // one iteration anyway.
140  int32 num_init_iters;
141  if (R0.NumRows() <= this_copy.rank_)
142  num_init_iters = 1;
143  else
144  num_init_iters = 3;
145 
146  this_copy.frozen_ = false; // un-freeze if it was frozen, so we can
147  // initialize.
148  for (int32 i = 0; i < num_init_iters; i++) {
149  BaseFloat scale;
150  R0_copy.CopyFromMat(R0);
151  this_copy.PreconditionDirections(&R0_copy, &scale);
152  }
153  rank_ = this_copy.rank_;
154  W_t_.Swap(&this_copy.W_t_);
155  d_t_.Swap(&this_copy.d_t_);
156  rho_t_ = this_copy.rho_t_;
157 }

159 void OnlineNaturalGradient::PreconditionDirections(
160  CuMatrixBase<BaseFloat> *X_t,
161  BaseFloat *scale) {
162  if (X_t->NumCols() == 1) {
163  // If the dimension of the space equals one then our natural gradient update
164  // with rescaling becomes a no-op, but the code wouldn't naturally handle it
165  // because rank would be zero. Support this as a special case.
166  if (scale)
167  *scale = 1.0;
168  return;
169  }
170 
171  if (t_ == 0) // not initialized
172  Init(*X_t);
173 
174  int32 R = W_t_.NumRows(), D = W_t_.NumCols();
175  // space for W_t, J_t, K_t, L_t.
176  CuMatrix<BaseFloat> WJKL_t(2 * R, D + R);
177  WJKL_t.Range(0, R, 0, D).CopyFromMat(W_t_);
178  BaseFloat rho_t(rho_t_);
179  Vector<BaseFloat> d_t(d_t_);
180 
181  bool updating = Updating();
182 
183  BaseFloat initial_product;
184  initial_product = TraceMatMat(*X_t, *X_t, kTrans);
185 
186  PreconditionDirectionsInternal(rho_t, initial_product,
187  updating, d_t, &WJKL_t, X_t);
188 
189  if (scale) {
190  if (initial_product <= 0.0) {
191  *scale = 1.0;
192  } else {
193  BaseFloat final_product = TraceMatMat(*X_t, *X_t, kTrans);
194  *scale = sqrt(initial_product / final_product);
195  }
196  }
197  t_ += 1;
198 }

200 void OnlineNaturalGradient::ReorthogonalizeXt1(
201  const VectorBase<BaseFloat> &d_t1,
202  BaseFloat rho_t1,
203  CuMatrixBase<BaseFloat> *W_t1,
204  CuMatrixBase<BaseFloat> *temp_W,
205  CuMatrixBase<BaseFloat> *temp_O) {
206  // threshold is a configuration value: a desired threshold on orthogonality,
207  // below which we won't reorthogonalize.
208  const BaseFloat threshold = 1.0e-03;
209 
210  int32 R = W_t1->NumRows(), D = W_t1->NumCols();
211  BaseFloat beta_t1 = rho_t1 * (1.0 + alpha_) + alpha_ * d_t1.Sum() / D;
212  Vector<BaseFloat> e_t1(R, kUndefined), sqrt_e_t1(R, kUndefined),
213  inv_sqrt_e_t1(R, kUndefined);
214  ComputeEt(d_t1, beta_t1, &e_t1, &sqrt_e_t1, &inv_sqrt_e_t1);
215 
216  temp_O->SymAddMat2(1.0, *W_t1, kNoTrans, 0.0);
217  // O_t = E_t^{-0.5} W_t W_t^T E_t^{-0.5}
218  Matrix<BaseFloat> O_mat(*temp_O);
219  SpMatrix<BaseFloat> O(O_mat, kTakeLower);
220  for (int32 i = 0; i < R; i++) {
221  BaseFloat i_factor = inv_sqrt_e_t1(i);
222  for (int32 j = 0; j <= i; j++) {
223  BaseFloat j_factor = inv_sqrt_e_t1(j);
224  O(i, j) *= i_factor * j_factor;
225  }
226  }
227  if (O.IsUnit(threshold)) {
228  if (self_debug_) {
229  KALDI_WARN << "Not reorthogonalizing since already orthognoal: " << O;
230  }
231  return;
232  }
233  TpMatrix<BaseFloat> C(R);
234  bool cholesky_ok = true;
235  try {
236  // one of the following two calls may throw an exception.
237  C.Cholesky(O);
238  C.Invert(); // Now it's C^{-1}.
239  if (!(C.Max() < 100.0)) {
240  KALDI_WARN << "Cholesky out of expected range, "
241  << "reorthogonalizing with Gram-Schmidt";
242  cholesky_ok = false;
243  }
244  } catch (...) {
245  // We do a Gram-Schmidt orthogonalization, which is a bit less efficient but
246  // more robust than the method using Cholesky.
247  KALDI_WARN << "Cholesky or Invert() failed while re-orthogonalizing R_t. "
248  << "Re-orthogonalizing on CPU.";
249  cholesky_ok = false;
250  }
251  if (!cholesky_ok) {
252  Matrix<BaseFloat> cpu_W_t1(*W_t1);
253  cpu_W_t1.OrthogonalizeRows();
254  W_t1->CopyFromMat(cpu_W_t1);
255  // at this point cpu_W_t1 represents R_{t+1}- it has orthonormal
256  // rows. Do: W_{t+1} = E_{t+1}^{0.5} R_{t+1}
257  CuVector<BaseFloat> sqrt_e_t1_gpu(sqrt_e_t1);
258  W_t1->MulRowsVec(sqrt_e_t1_gpu);
259  return;
260  }
261  // Next, compute (E_t^{0.5} C^{-1} E_t^{-0.5})
262  // but it's really t+1, not t.
263  for (int32 i = 0; i < R; i++) {
264  BaseFloat i_factor = sqrt_e_t1(i);
265  for (int32 j = 0; j < i; j++) {
266  // skip j == i because i_factor * j_factor == 1 for j == i.
267  BaseFloat j_factor = inv_sqrt_e_t1(j);
268  C(i, j) *= i_factor * j_factor;
269  }
270  }
271  O_mat.CopyFromTp(C);
272  temp_O->CopyFromMat(O_mat);
273  temp_W->CopyFromMat(*W_t1);
274  W_t1->AddMatMat(1.0, *temp_O, kNoTrans, *temp_W, kNoTrans, 0.0);
275 }

277 // makes sure certain invariants are being preserved
278 void OnlineNaturalGradient::SelfTest() const {
279  KALDI_ASSERT(rho_t_ >= epsilon_);
280  BaseFloat d_t_max = d_t_.Max(), d_t_min = d_t_.Min();
281  KALDI_ASSERT(d_t_min >= epsilon_);
282  KALDI_ASSERT(d_t_min > 0.9 * delta_ * d_t_max);
283  KALDI_ASSERT(rho_t_ > 0.9 * delta_ * d_t_max);
284 
285  int32 D = W_t_.NumCols(), R = W_t_.NumRows();
286  BaseFloat beta_t = rho_t_ * (1.0 + alpha_) + alpha_ * d_t_.Sum() / D;
287  Vector<BaseFloat> e_t(R, kUndefined), sqrt_e_t(R, kUndefined),
288  inv_sqrt_e_t(R, kUndefined);
289  ComputeEt(d_t_, beta_t, &e_t, &sqrt_e_t, &inv_sqrt_e_t);
290 
291  CuSpMatrix<BaseFloat> S(R);
292  S.AddMat2(1.0, W_t_, kNoTrans, 0.0);
293  SpMatrix<BaseFloat> O(S);
294  for (int32 i = 0; i < R; i++) {
295  BaseFloat i_factor = inv_sqrt_e_t(i);
296  for (int32 j = 0; j <= i; j++) {
297  BaseFloat j_factor = inv_sqrt_e_t(j);
298  O(i, j) *= i_factor * j_factor;
299  }
300  }
301  if (!O.IsUnit(1.0e-04) || O(0, 0) != O(0, 0)) {
302  BaseFloat worst_error = 0.0;
303  int32 worst_i = 0, worst_j = 0;
304  for (int32 i = 0; i < R; i++) {
305  for (int32 j = 0; j < R; j++) {
306  BaseFloat elem = O(i, j);
307  BaseFloat error = fabs(elem - (i == j ? 1.0 : 0.0));
308  if (error > worst_error || error != error) {
309  worst_error = error;
310  worst_i = i;
311  worst_j = j;
312  }
313  }
314  }
315  if (worst_error > 1.0e-02 || worst_error != worst_error) {
316  KALDI_WARN << "Failed to verify W_t (worst error: O[" << worst_i << ','
317  << worst_j << "] = " << O(worst_i, worst_j)
318  << ", d_t = " << d_t_;
319  }
320  }
321 }

323 void OnlineNaturalGradient::PreconditionDirectionsInternal(
324  const BaseFloat rho_t,
325  const BaseFloat tr_X_Xt,
326  bool updating,
327  const Vector<BaseFloat> &d_t,
328  CuMatrixBase<BaseFloat> *WJKL_t,
329  CuMatrixBase<BaseFloat> *X_t) {
330  int32 N = X_t->NumRows(), // Minibatch size.
331  D = X_t->NumCols(), // Dimensions of vectors we're preconditioning
332  R = rank_; // Rank of correction to unit matrix.
333  KALDI_ASSERT(R > 0 && R < D);
334  BaseFloat eta = Eta(N);
335 
336  CuMatrix<BaseFloat> H_t(N, R);
337  const CuSubMatrix<BaseFloat> W_t(*WJKL_t, 0, R, 0, D);
338  // Below, WJ_t and LK_t are combinations of two matrices,
339  // which we define in order to combine two separate multiplications into one.
340  CuSubMatrix<BaseFloat> J_t(*WJKL_t, R, R, 0, D),
341  L_t(*WJKL_t, 0, R, D, R),
342  K_t(*WJKL_t, R, R, D, R),
343  WJ_t(*WJKL_t, 0, 2 * R, 0, D),
344  LK_t(*WJKL_t, 0, 2 * R, D, R);
345 
346  H_t.AddMatMat(1.0, *X_t, kNoTrans, W_t, kTrans, 0.0); // H_t = X_t W_t^T
347 
348  if (!updating) {
349  // We're not updating the estimate of the Fisher matrix; we just apply the
350  // preconditioning and return.
351  // X_hat_t = X_t - H_t W_t
352  X_t->AddMatMat(-1.0, H_t, kNoTrans, W_t, kNoTrans, 1.0);
353  return;
354  }
355  J_t.AddMatMat(1.0, H_t, kTrans, *X_t, kNoTrans, 0.0); // J_t = H_t^T X_t
356 
357  bool compute_lk_together = (N > D);
358 
359  if (compute_lk_together) {
360  // do the following two multiplies in one operation...
361  // note
362  // L_t = W_t J_t^T
363  // K_t = J_t J_t^T
364  // Note: L_t was defined as L_t = J_t W_t^T, but it's actually symmetric,
365  // so we can compute it as L_t = W_t J_t^T.
366  LK_t.AddMatMat(1.0, WJ_t, kNoTrans, J_t, kTrans, 0.0);
367  } else {
368  K_t.SymAddMat2(1.0, J_t, kNoTrans, 0.0);
369  L_t.SymAddMat2(1.0, H_t, kTrans, 0.0);
370  }
371 
372  Matrix<BaseFloat> LK_cpu(LK_t); // contains L and K on the CPU.
373  SubMatrix<BaseFloat> L_t_cpu(LK_cpu, 0, R, 0, R),
374  K_t_cpu(LK_cpu, R, R, 0, R);
375  if (!compute_lk_together) {
376  // the SymAddMat2 operations only set the lower triangle and diagonal.
377  L_t_cpu.CopyLowerToUpper();
378  K_t_cpu.CopyLowerToUpper();
379  }
380 
381  // beta_t = \rho_t(1+\alpha) + \alpha/D tr(D_t)
382  BaseFloat beta_t = rho_t * (1.0 + alpha_) + alpha_ * d_t.Sum() / D;
383  Vector<BaseFloat> e_t(R), sqrt_e_t(R), inv_sqrt_e_t(R);
384  ComputeEt(d_t, beta_t, &e_t, &sqrt_e_t, &inv_sqrt_e_t);
385  KALDI_VLOG(5) << "e_t = " << e_t;
386 
387  // The double-precision Z_t here, and the scaling, is to avoid potential
388  // overflow, because Z_t is proportional to the fourth power of data.
389  SpMatrix<double> Z_t_double(R);
390  ComputeZt(N, rho_t, d_t, inv_sqrt_e_t, K_t_cpu, L_t_cpu, &Z_t_double);
391  BaseFloat z_t_scale = std::max<double>(1.0, Z_t_double.Trace());
392  Z_t_double.Scale(1.0 / z_t_scale);
393  SpMatrix<BaseFloat> Z_t_scaled(Z_t_double);
394 
395  Matrix<BaseFloat> U_t(R, R);
396  Vector<BaseFloat> c_t(R);
397  // do the symmetric eigenvalue decomposition Z_t = U_t C_t U_t^T.
398  Z_t_scaled.Eig(&c_t, &U_t);
399  SortSvd(&c_t, &U_t);
400  c_t.Scale(z_t_scale);
401 
402  const BaseFloat condition_threshold = 1.0e+06;
403  // must_reorthogonalize will be true if the last diagonal element of c_t is
404  // negative, since we don't take the absolute value, but this is the right
405  // thing anyway.
406  bool must_reorthogonalize = (c_t(0) > condition_threshold * c_t(R - 1));
407 
408  BaseFloat c_t_floor = pow(rho_t * (1 - eta), 2);
409  int32 nf;
410  c_t.ApplyFloor(c_t_floor, &nf);
411  if (nf > 0)
412  must_reorthogonalize = true;
413  if (nf > 0 && self_debug_) {
414  KALDI_WARN << "Floored " << nf << " elements of C_t.";
415  }
416 
417  X_t->AddMatMat(-1.0, H_t, kNoTrans, W_t, kNoTrans, 1.0); // X_hat_t = X_t - H_t W_t
418 
419  Vector<BaseFloat> sqrt_c_t(c_t);
420  sqrt_c_t.ApplyPow(0.5);
421 
422  // \rho_{t+1} = 1/(D - R) (\eta/N tr(X_t X_t^T) + (1-\eta)(D \rho_t + tr(D_t)) - tr(C_t^{0.5})).
423  BaseFloat rho_t1 = 1.0 / (D - R) * (eta / N * tr_X_Xt
424  + (1-eta)*(D * rho_t + d_t.Sum())
425  - sqrt_c_t.Sum());
426  // D_{t+1} = C_t^{0.5} - \rho_{t+1} I
427  Vector<BaseFloat> d_t1(sqrt_c_t);
428  d_t1.Add(-rho_t1);
429  BaseFloat floor_val = std::max(epsilon_, delta_ * sqrt_c_t.Max());
430  if (rho_t1 < floor_val)
431  rho_t1 = floor_val;
432  d_t1.ApplyFloor(floor_val);
433 
434  CuMatrix<BaseFloat> W_t1(R, D); // W_{t+1}
435  ComputeWt1(N, d_t, d_t1, rho_t, rho_t1, U_t, sqrt_c_t, inv_sqrt_e_t,
436  W_t, &J_t, &W_t1);
437 
438  if (must_reorthogonalize) {
439  if (self_debug_) {
440  KALDI_WARN << "Reorthogonalizing.";
441  }
442  ReorthogonalizeXt1(d_t1,
443  rho_t1,
444  &W_t1,
445  &J_t,
446  &L_t);
447  }
448 
449  W_t_.Swap(&W_t1);
450  d_t_.CopyFromVec(d_t1);
451  rho_t_ = rho_t1;
452 
453  if (self_debug_)
454  SelfTest();
455 }

457 bool OnlineNaturalGradient::Updating() const {
458  // Just hard-code it here that we do 10 initial updates before skipping any.
459  // This must be > 'num_init_iters = 3' from Init().
460  const int num_initial_updates = 10;
461 
462  return (!frozen_ &&
463  (t_ <= num_initial_updates ||
464  (t_ - num_initial_updates) % update_period_ == 0));
465 }

468 BaseFloat OnlineNaturalGradient::Eta(int32 N) const {
469  if (num_minibatches_history_ > 0.0) {
470  KALDI_ASSERT(num_minibatches_history_ > 1.0);
471  return 1.0 / num_minibatches_history_;
472  } else {
473  KALDI_ASSERT(num_samples_history_ > 0.0);
474  BaseFloat ans = 1.0 - exp(-N / num_samples_history_);
475  // Don't let eta approach 1 too closely, as it can lead to NaN's appearing if
476  // the input is all zero.
477  if (ans > 0.9) ans = 0.9;
478  return ans;
479  }
480 }

482 void OnlineNaturalGradient::ComputeWt1(int32 N,
483  const VectorBase<BaseFloat> &d_t,
484  const VectorBase<BaseFloat> &d_t1,
485  BaseFloat rho_t,
486  BaseFloat rho_t1,
487  const MatrixBase<BaseFloat> &U_t,
488  const VectorBase<BaseFloat> &sqrt_c_t,
489  const VectorBase<BaseFloat> &inv_sqrt_e_t,
490  const CuMatrixBase<BaseFloat> &W_t,
491  CuMatrixBase<BaseFloat> *J_t,
492  CuMatrixBase<BaseFloat> *W_t1) const {
493 
494  int32 R = d_t.Dim(), D = W_t.NumCols();
495  BaseFloat eta = Eta(N);
496 
497  // \beta_{t+1} = \rho_{t+1} (1+\alpha) + \alpha/D tr(D_{t+1})
498  BaseFloat beta_t1 = rho_t1 * (1.0 + alpha_) + alpha_ * d_t1.Sum() / D;
499  KALDI_ASSERT(beta_t1 > 0.0);
500  Vector<BaseFloat> e_t1(R, kUndefined), sqrt_e_t1(R, kUndefined),
501  inv_sqrt_e_t1(R, kUndefined);
502  ComputeEt(d_t1, beta_t1, &e_t1, &sqrt_e_t1, &inv_sqrt_e_t1);
503  Vector<BaseFloat> inv_sqrt_c_t(sqrt_c_t);
504  inv_sqrt_c_t.InvertElements();
505 
506  Vector<BaseFloat> w_t_coeff(R);
507  for (int32 i = 0; i < R; i++)
508  w_t_coeff(i) = (1.0 - eta) / (eta/N) * (d_t(i) + rho_t);
509  CuVector<BaseFloat> w_t_coeff_gpu(w_t_coeff);
510  // B_t = J_t + (1-\eta)/(\eta/N) (D_t + \rho_t I) W_t
511  J_t->AddDiagVecMat(1.0, w_t_coeff_gpu, W_t, kNoTrans, 1.0);
512 
513  // A_t = (\eta/N) E_{t+1}^{0.5} C_t^{-0.5} U_t^T E_t^{-0.5} B_t
514  Matrix<BaseFloat> A_t(U_t, kTrans);
515  for (int32 i = 0; i < R; i++) {
516  BaseFloat i_factor = (eta / N) * sqrt_e_t1(i) * inv_sqrt_c_t(i);
517  for (int32 j = 0; j < R; j++) {
518  BaseFloat j_factor = inv_sqrt_e_t(j);
519  A_t(i, j) *= i_factor * j_factor;
520  }
521  }
522  // W_{t+1} = A_t B_t
523  CuMatrix<BaseFloat> A_t_gpu(A_t);
524  W_t1->AddMatMat(1.0, A_t_gpu, kNoTrans, *J_t, kNoTrans, 0.0);
525 }

527 void OnlineNaturalGradient::ComputeZt(int32 N,
528  BaseFloat rho_t,
529  const VectorBase<BaseFloat> &d_t,
530  const VectorBase<BaseFloat> &inv_sqrt_e_t,
531  const MatrixBase<BaseFloat> &K_t,
532  const MatrixBase<BaseFloat> &L_t,
533  SpMatrix<double> *Z_t) const {
534  // Use doubles because the range of quantities in Z_t can get large (fourth
535  // power of data), and we want to avoid overflow. This routine is fast.
536  BaseFloat eta = Eta(N);
537  Vector<BaseFloat> d_t_rho_t(d_t);
538  d_t_rho_t.Add(rho_t); // now d_t_rho_t is diag(D_t + \rho_t I).
539  double etaN = eta / N, eta1 = 1.0 - eta,
540  etaN_sq = etaN * etaN, eta1_sq = eta1 * eta1,
541  etaN_eta1 = etaN * eta1;
542  int32 R = d_t.Dim();
543  for (int32 i = 0; i < R; i++) {
544  double inv_sqrt_e_t_i = inv_sqrt_e_t(i), d_t_rho_t_i = d_t_rho_t(i);
545  for (int32 j = 0; j <= i; j++) {
546  double inv_sqrt_e_t_j = inv_sqrt_e_t(j), d_t_rho_t_j = d_t_rho_t(j),
547  L_t_i_j = 0.5 * (L_t(i, j) + L_t(j, i)),
548  K_t_i_j = 0.5 * (K_t(i, j) + K_t(j, i));
549  // See (eqn:Zt) in header.
550  (*Z_t)(i, j) = etaN_sq * inv_sqrt_e_t_i * K_t_i_j * inv_sqrt_e_t_j
551  + etaN_eta1 * inv_sqrt_e_t_i * L_t_i_j * inv_sqrt_e_t_j * d_t_rho_t_j
552  + etaN_eta1 * d_t_rho_t_i * inv_sqrt_e_t_i * L_t_i_j * inv_sqrt_e_t_j
553  + (i == j ? eta1_sq * d_t_rho_t_i * d_t_rho_t_i : 0.0);
554  }
555  }
556 }

558 void OnlineNaturalGradient::ComputeEt(const VectorBase<BaseFloat> &d_t,
559  BaseFloat beta_t,
560  VectorBase<BaseFloat> *e_t,
561  VectorBase<BaseFloat> *sqrt_e_t,
562  VectorBase<BaseFloat> *inv_sqrt_e_t) const {
563  // e_{tii} = 1/(\beta_t/d_{tii} + 1)
564  int32 D = d_t.Dim();
565  const BaseFloat *d = d_t.Data();
566  BaseFloat *e = e_t->Data();
567  for (int32 i = 0; i < D; i++)
568  e[i] = 1.0 / (beta_t / d[i] + 1);
569  sqrt_e_t->CopyFromVec(*e_t);
570  sqrt_e_t->ApplyPow(0.5);
571  inv_sqrt_e_t->CopyFromVec(*sqrt_e_t);
572  inv_sqrt_e_t->InvertElements();
573 }
