30 void AccCmvnStats(const VectorBase<BaseFloat> &feats, BaseFloat weight, MatrixBase<double> *stats) {
31  int32 dim = feats.Dim();
32  KALDI_ASSERT(stats != NULL);
33  KALDI_ASSERT(stats->NumRows() == 2 && stats->NumCols() == dim + 1);
34  // Remove these __restrict__ modifiers if they cause compilation problems.
35  // It's just an optimization.
36  double *__restrict__ mean_ptr = stats->RowData(0),
37  *__restrict__ var_ptr = stats->RowData(1),
38  *__restrict__ count_ptr = mean_ptr + dim;
39  const BaseFloat * __restrict__ feats_ptr = feats.Data();
40  *count_ptr += weight;
41  // Careful-- if we change the format of the matrix, the "mean_ptr < count_ptr"
42  // statement below might become wrong.
43  for (; mean_ptr < count_ptr; mean_ptr++, var_ptr++, feats_ptr++) {
44  *mean_ptr += *feats_ptr * weight;
45  *var_ptr += *feats_ptr * *feats_ptr * weight;
46  }
47 }

49 void AccCmvnStats(const MatrixBase<BaseFloat> &feats,
50  const VectorBase<BaseFloat> *weights,
51  MatrixBase<double> *stats) {
52  int32 num_frames = feats.NumRows();
53  if (weights != NULL) {
54  KALDI_ASSERT(weights->Dim() == num_frames);
55  }
56  for (int32 i = 0; i < num_frames; i++) {
57  SubVector<BaseFloat> this_frame = feats.Row(i);
58  BaseFloat weight = (weights == NULL ? 1.0 : (*weights)(i));
59  if (weight != 0.0)
60  AccCmvnStats(this_frame, weight, stats);
61  }
62 }

64 void ApplyCmvn(const MatrixBase<double> &stats,
65  bool var_norm,
66  MatrixBase<BaseFloat> *feats,
67  bool mean_norm) {
68  KALDI_ASSERT(feats != NULL);
69  int32 dim = stats.NumCols() - 1;
70  if (stats.NumRows() > 2 || stats.NumRows() < 1 || feats->NumCols() != dim) {
71  KALDI_ERR << "Dim mismatch: cmvn "
72  << stats.NumRows() << 'x' << stats.NumCols()
73  << ", feats " << feats->NumRows() << 'x' << feats->NumCols();
74  }
75  if (stats.NumRows() == 1 && var_norm)
76  KALDI_ERR << "You requested variance normalization but no variance stats "
77  << "are supplied.";
78 
79  double count = stats(0, dim);
80  // Do not change the threshold of 1.0 here: in the balanced-cmvn code, when
81  // computing an offset and representing it as stats, we use a count of one.
82  if (count < 1.0)
83  KALDI_ERR << "Insufficient stats for cepstral mean and variance normalization: "
84  << "count = " << count;
85 
86  Matrix<BaseFloat> norm(2, dim); // norm(0, d) = mean offset
87  // norm(1, d) = scale, e.g. x(d) <-- x(d)*norm(1, d) + norm(0, d).
88  for (int32 d = 0; d < dim; d++) {
89  double mean, offset, scale;
90  mean = stats(0, d)/count;
91  if (!var_norm) {
92  scale = 1.0;
93  offset = -mean;
94  } else {
95  double var = (stats(1, d)/count) - mean*mean,
96  floor = 1.0e-20;
97  if (var < floor) {
98  KALDI_WARN << "Flooring cepstral variance from " << var << " to "
99  << floor;
100  var = floor;
101  }
102  scale = 1.0 / sqrt(var);
103  if (scale != scale || 1/scale == 0.0)
104  KALDI_ERR << "NaN or infinity in cepstral mean/variance computation";
105  offset = -(mean*scale);
106  }
107  if (mean_norm) {
108  norm(0, d) = offset;
109  } else {
110  norm(0, d) = 0;
111  }
112  norm(1, d) = scale;
113  }
114  // Apply the normalization.
115  if (var_norm)
116  feats->MulColsVec(norm.Row(1));
117  feats->AddVecToRows(1.0, norm.Row(0));
118 }

140  << "count = " << count;
141 
142  Matrix<BaseFloat> norm(2, dim); // norm(0, d) = mean offset
143  // norm(1, d) = scale, e.g. x(d) <-- x(d)*norm(1, d) + norm(0, d).
144  for (int32 d = 0; d < dim; d++) {
145  double mean, offset, scale;
146  mean = stats(0, d) / count;
147  if (!var_norm) {
148  scale = 1.0;
149  offset = mean;
150  } else {

156  var = floor;
157  }

164  offset = 0;
165  }

168  norm(1, d) = scale;
169  }
