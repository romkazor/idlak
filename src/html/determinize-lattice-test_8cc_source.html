27 void TestLatticeStringRepository() {
28  typedef int32 IntType;
29 
30  LatticeStringRepository<IntType> sr;
31  typedef LatticeStringRepository<IntType>::Entry Entry;
32 
33  for(int i = 0; i < 100; i++) {
34  int len = kaldi::Rand() % 5;
35  vector<IntType> str(len), str2(kaldi::Rand() % 4);
36  const Entry *e = NULL;
37  for(int i = 0; i < len; i++) {
38  str[i] = kaldi::Rand() % 5;
39  e = sr.Successor(e, str[i]);
40  }
41  sr.ConvertToVector(e, &str2);
42  assert(str == str2);
43 
44  int len2 = kaldi::Rand() % 5;
45  str2.resize(len2);
46  const Entry *f = sr.EmptyString(); // NULL
47  for(int i = 0; i < len2; i++) {
48  str2[i] = kaldi::Rand() % 5;
49  f = sr.Successor(f, str2[i]);
50  }
51  vector<IntType> prefix, prefix2(kaldi::Rand() % 10),
52  prefix3;
53  for(int i = 0; i < len && i < len2; i++) {
54  if (str[i] == str2[i]) prefix.push_back(str[i]);
55  else break;
56  }
57  const Entry *g = sr.CommonPrefix(e, f);
58  sr.ConvertToVector(g, &prefix2);
59  sr.ConvertToVector(e, &prefix3);
60  sr.ReduceToCommonPrefix(f, &prefix3);
61  assert(prefix == prefix2);
62  assert(prefix == prefix3);
63  assert(sr.IsPrefixOf(g, e));
64  assert(sr.IsPrefixOf(g, f));
65  if (str.size() > prefix.size())
66  assert(!sr.IsPrefixOf(e, g));
67  }
68 }

71 // test that determinization proceeds correctly on general
72 // FSTs (not guaranteed determinzable, but we use the
73 // max-states option to stop it getting out of control).
74 template<class Arc> void TestDeterminizeLattice() {
75  typedef typename Arc::Weight Weight;
76  typedef int32 Int;
77  typedef ArcTpl<CompactLatticeWeightTpl<Weight, Int> > CompactArc;
78 
79  for(int i = 0; i < 100; i++) {
80  RandFstOptions opts;
81  opts.n_states = 4;
82  opts.n_arcs = 10;
83  opts.n_final = 2;
84  opts.allow_empty = false;
85  opts.weight_multiplier = 0.5; // impt for the randomly generated weights
86  // to be exactly representable in float,
87  // or this test fails because numerical differences can cause symmetry in
88  // weights to be broken, which causes the wrong path to be chosen as far
89  // as the string part is concerned.
90 
91  VectorFst<Arc> *fst = RandFst<Arc>();
92  std::cout << "FST before lattice-determinizing is:\n";
93  {
94  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true, "\t");
95  fstprinter.Print(&std::cout, "standard output");
96  }
97  VectorFst<Arc> det_fst;
98  try {
99  DeterminizeLatticeOptions lat_opts;
100  lat_opts.max_mem = 100;
101 
102  if (!DeterminizeLattice<TropicalWeight, int32>(*fst, &det_fst, lat_opts, NULL))
103  throw std::runtime_error("could not determinize");
104  std::cout << "FST after lattice-determinizing is:\n";
105  {
106  FstPrinter<Arc> fstprinter(det_fst, NULL, NULL, NULL, false, true, "\t");
107  fstprinter.Print(&std::cout, "standard output");
108  }
109  assert(det_fst.Properties(kIDeterministic, true) & kIDeterministic);
110  // OK, now determinize it a different way and check equivalence.
111  // [note: it's not normal determinization, it's taking the best path
112  // for any input-symbol sequence....
113  VectorFst<CompactArc> compact_fst, compact_det_fst;
114  ConvertLattice<Weight, Int>(*fst, &compact_fst, false);
115  std::cout << "Compact FST is:\n";
116  {
117  FstPrinter<CompactArc> fstprinter(compact_fst, NULL, NULL, NULL, false, true, "\t");
118  fstprinter.Print(&std::cout, "standard output");
119  }
120  if (kaldi::Rand() % 2 == 1)
121  ConvertLattice<Weight, Int>(det_fst, &compact_det_fst, false);
122  else
123  if (!DeterminizeLattice<TropicalWeight, int32>(*fst, &compact_det_fst, lat_opts, NULL))
124  throw std::runtime_error("could not determinize");
125 
126  std::cout << "Compact version of determinized FST is:\n";
127  {
128  FstPrinter<CompactArc> fstprinter(compact_det_fst, NULL, NULL, NULL, false, true, "\t");
129  fstprinter.Print(&std::cout, "standard output");
130  }
131 
132  assert(RandEquivalent(compact_det_fst, compact_fst, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length, max*/));
133  } catch (...) {
134  std::cout << "Failed to lattice-determinize this FST (probably not determinizable)\n";
135  }
136  delete fst;
137  }
138 }
