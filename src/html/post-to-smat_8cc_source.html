41  ParseOptions po(usage);

49  po.Read(argc, argv);

54

56  po.PrintUsage();

59 
60  std::string posteriors_rspecifier = po.GetArg(1),

62 
63 
64  SequentialPosteriorReader posterior_reader(posteriors_rspecifier);
65 
66  TableWriter<KaldiObjectHolder<SparseMatrix<BaseFloat> > > sparse_matrix_writer(

68 
69  int32 num_done = 0;
70  for (; !posterior_reader.Done(); posterior_reader.Next()) {
71  const kaldi::Posterior &posterior = posterior_reader.Value();
72  // The following constructor will throw an error if there is some kind of

74  SparseMatrix<BaseFloat> smat(dim, posterior);
75  sparse_matrix_writer.Write(posterior_reader.Key(), smat);
76  num_done++;
77  }
