29 int main(int argc, char *argv[]) {
30  try {
31  using namespace kaldi;
32  using namespace kaldi::nnet3;
33  typedef kaldi::int32 int32;
34  typedef kaldi::int64 int64;
35 
36  const char *usage =
37  "Propagate the features through raw neural network model "
38  "and write the output.\n"
39  "If --apply-exp=true, apply the Exp() function to the output "
40  "before writing it out.\n"
41  "\n"
42  "Usage: nnet3-compute [options] <nnet-in> <features-rspecifier> <matrix-wspecifier>\n"
43  " e.g.: nnet3-compute final.raw scp:feats.scp ark:nnet_prediction.ark\n"
44  "See also: nnet3-compute-from-egs, nnet3-chain-compute-post\n"
45  "Note: this program does not currently make very efficient use of the GPU.\n";
46 
47  ParseOptions po(usage);
48  Timer timer;
49 
50  NnetSimpleComputationOptions opts;
51  opts.acoustic_scale = 1.0; // by default do no scaling in this recipe.
52 
53  bool apply_exp = false, use_priors = false;
54  std::string use_gpu = "yes";
55 
56  std::string ivector_rspecifier,
57  online_ivector_rspecifier,
58  utt2spk_rspecifier;
59  int32 online_ivector_period = 0;
60  opts.Register(&po);
61 
62  po.Register("ivectors", &ivector_rspecifier, "Rspecifier for "
63  "iVectors as vectors (i.e. not estimated online); per utterance "
64  "by default, or per speaker if you provide the --utt2spk option.");
65  po.Register("utt2spk", &utt2spk_rspecifier, "Rspecifier for "
66  "utt2spk option used to get ivectors per speaker");
67  po.Register("online-ivectors", &online_ivector_rspecifier, "Rspecifier for "
68  "iVectors estimated online, as matrices. If you supply this,"
69  " you must set the --online-ivector-period option.");
70  po.Register("online-ivector-period", &online_ivector_period, "Number of frames "
71  "between iVectors in matrices supplied to the --online-ivectors "
72  "option");
73  po.Register("apply-exp", &apply_exp, "If true, apply exp function to "
74  "output");
75  po.Register("use-gpu", &use_gpu,
76  "yes|no|optional|wait, only has effect if compiled with CUDA");
77  po.Register("use-priors", &use_priors, "If true, subtract the logs of the "
78  "priors stored with the model (in this case, "
79  "a .mdl file is expected as input).");
80 
81  po.Read(argc, argv);
82 
83  if (po.NumArgs() != 3) {
84  po.PrintUsage();
85  exit(1);
86  }
87 
88 #if HAVE_CUDA==1
89  CuDevice::Instantiate().SelectGpuId(use_gpu);
90 #endif
91 
92  std::string nnet_rxfilename = po.GetArg(1),
93  feature_rspecifier = po.GetArg(2),
94  matrix_wspecifier = po.GetArg(3);
95 
96  Nnet raw_nnet;
97  AmNnetSimple am_nnet;
98  if (use_priors) {
99  bool binary;
100  TransitionModel trans_model;
101  Input ki(nnet_rxfilename, &binary);
102  trans_model.Read(ki.Stream(), binary);
103  am_nnet.Read(ki.Stream(), binary);
104  } else {
105  ReadKaldiObject(nnet_rxfilename, &raw_nnet);
106  }
107  Nnet &nnet = (use_priors ? am_nnet.GetNnet() : raw_nnet);
108  SetBatchnormTestMode(true, &nnet);
109  SetDropoutTestMode(true, &nnet);
110  CollapseModel(CollapseModelConfig(), &nnet);
111 
112  Vector<BaseFloat> priors;
113  if (use_priors)
114  priors = am_nnet.Priors();
115 
116  RandomAccessBaseFloatMatrixReader online_ivector_reader(
117  online_ivector_rspecifier);
118  RandomAccessBaseFloatVectorReaderMapped ivector_reader(
119  ivector_rspecifier, utt2spk_rspecifier);
120 
121  CachingOptimizingCompiler compiler(nnet, opts.optimize_config);
122 
123  BaseFloatMatrixWriter matrix_writer(matrix_wspecifier);
124 
125  int32 num_success = 0, num_fail = 0;
126  int64 frame_count = 0;
127 
128  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
129 
130  for (; !feature_reader.Done(); feature_reader.Next()) {
131  std::string utt = feature_reader.Key();
132  const Matrix<BaseFloat> &features (feature_reader.Value());
133  if (features.NumRows() == 0) {
134  KALDI_WARN << "Zero-length utterance: " << utt;
135  num_fail++;
136  continue;
137  }
138  const Matrix<BaseFloat> *online_ivectors = NULL;
139  const Vector<BaseFloat> *ivector = NULL;
140  if (!ivector_rspecifier.empty()) {
141  if (!ivector_reader.HasKey(utt)) {
142  KALDI_WARN << "No iVector available for utterance " << utt;
143  num_fail++;
144  continue;
145  } else {
146  ivector = &ivector_reader.Value(utt);
147  }
148  }
149  if (!online_ivector_rspecifier.empty()) {
150  if (!online_ivector_reader.HasKey(utt)) {
151  KALDI_WARN << "No online iVector available for utterance " << utt;
152  num_fail++;
153  continue;
154  } else {
155  online_ivectors = &online_ivector_reader.Value(utt);
156  }
157  }
158 
159  DecodableNnetSimple nnet_computer(
160  opts, nnet, priors,
161  features, &compiler,
162  ivector, online_ivectors,
163  online_ivector_period);
164 
165  Matrix<BaseFloat> matrix(nnet_computer.NumFrames(),
166  nnet_computer.OutputDim());
167  for (int32 t = 0; t < nnet_computer.NumFrames(); t++) {
168  SubVector<BaseFloat> row(matrix, t);
169  nnet_computer.GetOutputForFrame(t, &row);
170  }
171 
172  if (apply_exp)
173  matrix.ApplyExp();
174 
175  matrix_writer.Write(utt, matrix);
176 
177  frame_count += features.NumRows();
178  num_success++;
179  }
180 
181 #if HAVE_CUDA==1
182  CuDevice::Instantiate().PrintProfile();
183 #endif
184  double elapsed = timer.Elapsed();
185  KALDI_LOG << "Time taken "<< elapsed
186  << "s: real-time factor assuming 100 frames/sec is "
187  << (elapsed*100.0/frame_count);
188  KALDI_LOG << "Done " << num_success << " utterances, failed for "
189  << num_fail;
190 
191  if (num_success != 0) return 0;
192  else return 1;
193  } catch(const std::exception &e) {
194  std::cerr << e.what();
195  return -1;
196  }
197 }
