55  int sz = Rand() % 73;

66  if (omit_empty_strings) KALDI_ASSERT(str_vec[i] != "");
67  new_full.append(str_vec[i]);
68  if (i < str_vec.size() -1) new_full.append(delim);
69  }
70  std::string new_full2;
71  JoinVectorToString(str_vec, delim.c_str(), omit_empty_strings, &new_full2);
72  if (omit_empty_strings) { // sequences of delimiters cannot be matched
73  size_t start = full.find_first_not_of(delim),
74  end = full.find_last_not_of(delim);
75  if (start == std::string::npos) { // only delimiters
76  KALDI_ASSERT(end == std::string::npos);
77  } else {
78  std::string full_test;
