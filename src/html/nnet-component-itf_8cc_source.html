204 bool Component::IsComputable(const MiscComputationInfo &misc_info,
205  const Index &output_index,
206  const IndexSet &input_index_set,
207  std::vector<Index> *used_inputs) const {
208  // the default Component dependency is for an output index to map directly to
209  // the same input index, which is required to compute the output.
210  if (!input_index_set(output_index))
211  return false;

213  used_inputs->clear();
214  used_inputs->push_back(output_index);
215  }
216  return true;

254 std::string UpdatableComponent::ReadUpdatableCommon(std::istream &is,
255  bool binary) {
256  std::ostringstream opening_tag;
257  opening_tag << '<' << this->Type() << '>';
258  std::string token;
259  ReadToken(is, binary, &token);
260  if (token == opening_tag.str()) {
261  // if the first token is the opening tag, then
262  // ignore it and get the next tag.
263  ReadToken(is, binary, &token);
264  }
265  if (token == "<LearningRateFactor>") {
266  ReadBasicType(is, binary, &learning_rate_factor_);
267  ReadToken(is, binary, &token);
268  } else {
269  learning_rate_factor_ = 1.0;
270  }
271  if (token == "<IsGradient>") {
272  ReadBasicType(is, binary, &is_gradient_);
273  ReadToken(is, binary, &token);
274  } else {
275  is_gradient_ = false;
276  }
277  if (token == "<MaxChange>") {
278  ReadBasicType(is, binary, &max_change_);
279  ReadToken(is, binary, &token);
280  } else {
281  max_change_ = 0.0;
282  }
283  if (token == "<L2Regularize>") {
284  ReadBasicType(is, binary, &l2_regularize_);
285  ReadToken(is, binary, &token);
286  } else {
287  l2_regularize_ = 0.0;
288  }
289  if (token == "<LearningRate>") {
290  ReadBasicType(is, binary, &learning_rate_);
291  return "";
292  } else {
293  return token;
294  }
295 }

297 void UpdatableComponent::WriteUpdatableCommon(std::ostream &os,
298  bool binary) const {
299  std::ostringstream opening_tag;
300  opening_tag << '<' << this->Type() << '>';
301  std::string token;
302  WriteToken(os, binary, opening_tag.str());
303  if (learning_rate_factor_ != 1.0) {
304  WriteToken(os, binary, "<LearningRateFactor>");
305  WriteBasicType(os, binary, learning_rate_factor_);
306  }
307  if (is_gradient_) {
308  WriteToken(os, binary, "<IsGradient>");
309  WriteBasicType(os, binary, is_gradient_);
310  }
311  if (max_change_ > 0.0) {
312  WriteToken(os, binary, "<MaxChange>");
313  WriteBasicType(os, binary, max_change_);
314  }
315  if (l2_regularize_ > 0.0) {
316  WriteToken(os, binary, "<L2Regularize>");
317  WriteBasicType(os, binary, l2_regularize_);
318  }
319  WriteToken(os, binary, "<LearningRate>");
320  WriteBasicType(os, binary, learning_rate_);
321 }

340 void NonlinearComponent::StoreStatsInternal(
341  const CuMatrixBase<BaseFloat> &out_value,
342  const CuMatrixBase<BaseFloat> *deriv) {
343  KALDI_ASSERT(out_value.NumCols() == dim_);
344 
345  // Check we have the correct dimensions.
346  if (value_sum_.Dim() != dim_ ||
347  (deriv != NULL && deriv_sum_.Dim() != dim_)) {
348  if (value_sum_.Dim() != dim_) {
349  value_sum_.Resize(dim_);
350  count_ = 0.0;
351  }
352  if (deriv != NULL && deriv_sum_.Dim() != dim_) {
353  deriv_sum_.Resize(dim_);
354  count_ = 0.0;
355  value_sum_.SetZero();
356  }
357  }
358  count_ += out_value.NumRows();
359  CuVector<BaseFloat> temp(dim_);
360  temp.AddRowSumMat(1.0, out_value, 0.0);
361  value_sum_.AddVec(1.0, temp);
362  if (deriv != NULL) {
363  temp.AddRowSumMat(1.0, *deriv, 0.0);
364  deriv_sum_.AddVec(1.0, temp);
365  }
366 }

368 void NonlinearComponent::StoreBackpropStats(
369  const CuMatrixBase<BaseFloat> &out_deriv) {
370  // Only store these stats about every 4 minibatches. Make sure to always
371  // store the stats on the very first minibatch, or it would interact badly
372  // with the ConsolidateMemory() code.
373  if (RandInt(0, 3) == 0 && oderiv_count_ != 0)
374  return;
375 
376  KALDI_ASSERT(out_deriv.NumCols() == dim_);
377 
378  // Check we have the correct dimensions.
379  if (oderiv_sumsq_.Dim() != dim_) {
380  oderiv_sumsq_.Resize(dim_);
381  oderiv_count_ = 0.0;
382  }
383  CuVector<BaseFloat> temp(dim_);
384  temp.AddDiagMat2(1.0, out_deriv, kTrans, 0.0);
385  oderiv_sumsq_.AddVec(1.0, temp);
386  oderiv_count_ += out_deriv.NumRows();
387 }

400 std::string NonlinearComponent::Info() const {
401  std::stringstream stream;
402  stream << Type() << ", dim=" << dim_;
403  if (block_dim_ != dim_)
404  stream << ", block-dim=" << block_dim_;
405  if (self_repair_lower_threshold_ != BaseFloat(kUnsetThreshold))
406  stream << ", self-repair-lower-threshold=" << self_repair_lower_threshold_;
407  if (self_repair_upper_threshold_ != BaseFloat(kUnsetThreshold))
408  stream << ", self-repair-upper-threshold=" << self_repair_upper_threshold_;
409  if (self_repair_scale_ != 0.0)
410  stream << ", self-repair-scale=" << self_repair_scale_;
411  if (count_ > 0 && value_sum_.Dim() == dim_) {
412  stream << ", count=" << std::setprecision(3) << count_
413  << std::setprecision(6);
414  stream << ", self-repaired-proportion="
415  << (num_dims_processed_ > 0 ?
416  num_dims_self_repaired_ / num_dims_processed_ : 0);
417  Vector<double> value_avg_dbl(value_sum_);
418  Vector<BaseFloat> value_avg(value_avg_dbl);
419  value_avg.Scale(1.0 / count_);
420  stream << ", value-avg=" << SummarizeVector(value_avg);
421  if (deriv_sum_.Dim() == dim_) {
422  Vector<double> deriv_avg(deriv_sum_);
423  deriv_avg.Scale(1.0 / count_);
424  stream << ", deriv-avg=" << SummarizeVector(deriv_avg);
425  }
426  }
427  if (oderiv_count_ > 0 && oderiv_sumsq_.Dim() == dim_) {
428  Vector<double> oderiv_rms(oderiv_sumsq_);
429  oderiv_rms.Scale(1.0 / oderiv_count_);
430  // The ApplyMin() is so that the statement after it does not fail even if we
431  // had subtracted models (e.g. in full_progress.*.log).
432  oderiv_rms.ApplyFloor(0.0);
433  oderiv_rms.ApplyPow(0.5);
434  stream << ", oderiv-rms=" << SummarizeVector(oderiv_rms)
435  << ", oderiv-count=" << oderiv_count_;
436  }
437  return stream.str();
438 }

450 void NonlinearComponent::Add(BaseFloat alpha, const Component &other_in) {
451  const NonlinearComponent *other =
452  dynamic_cast<const NonlinearComponent*>(&other_in);
453  KALDI_ASSERT(other != NULL);
454  if (value_sum_.Dim() == 0 && other->value_sum_.Dim() != 0)
455  value_sum_.Resize(other->value_sum_.Dim());
456  if (deriv_sum_.Dim() == 0 && other->deriv_sum_.Dim() != 0)
457  deriv_sum_.Resize(other->deriv_sum_.Dim());
458  if (oderiv_sumsq_.Dim() == 0 && other->oderiv_sumsq_.Dim() != 0)
459  oderiv_sumsq_.Resize(other->oderiv_sumsq_.Dim());
460  if (other->value_sum_.Dim() != 0)
461  value_sum_.AddVec(alpha, other->value_sum_);
462  if (other->deriv_sum_.Dim() != 0)
463  deriv_sum_.AddVec(alpha, other->deriv_sum_);
464  if (other->oderiv_sumsq_.Dim() != 0)
465  oderiv_sumsq_.AddVec(alpha, other->oderiv_sumsq_);
466  count_ += alpha * other->count_;
467  oderiv_count_ += alpha * other->oderiv_count_;
468  num_dims_self_repaired_ += alpha * other->num_dims_self_repaired_;
469  num_dims_processed_ += alpha * other->num_dims_processed_;
470 }

472 void NonlinearComponent::Read(std::istream &is, bool binary) {
473  std::ostringstream ostr_beg, ostr_end;
474  ostr_beg << "<" << Type() << ">"; // e.g. "<SigmoidComponent>"
475  ostr_end << "</" << Type() << ">"; // e.g. "</SigmoidComponent>"
476  ExpectOneOrTwoTokens(is, binary, ostr_beg.str(), "<Dim>");
477  ReadBasicType(is, binary, &dim_); // Read dimension.
478  if (PeekToken(is, binary) == 'B') {
479  ExpectToken(is, binary, "<BlockDim>");
480  ReadBasicType(is, binary, &block_dim_);
481  } else {
482  block_dim_ = dim_;
483  }
484  ExpectToken(is, binary, "<ValueAvg>");
485  value_sum_.Read(is, binary);
486  ExpectToken(is, binary, "<DerivAvg>");
487  deriv_sum_.Read(is, binary);
488  ExpectToken(is, binary, "<Count>");
489  ReadBasicType(is, binary, &count_);
490  if (PeekToken(is, binary) == 'O') {
491  ExpectToken(is, binary, "<OderivRms>");
492  oderiv_sumsq_.Read(is, binary);
493  oderiv_sumsq_.ApplyPow(2.0);
494  ExpectToken(is, binary, "<OderivCount>");
495  ReadBasicType(is, binary, &oderiv_count_);
496  } else {
497  oderiv_count_ = 0.0;
498  oderiv_sumsq_.Resize(0);
499  }
500  value_sum_.Scale(count_);
501  deriv_sum_.Scale(count_);
502  oderiv_sumsq_.Scale(oderiv_count_);
503 
504  std::string token;
505  ReadToken(is, binary, &token);
506  if (token[0] != '<') {
507  // this should happen only rarely, in case we couldn't push back the
508  // '<' to the stream in PeekToken().
509  token = '<' + token;
510  }
511  if (token == "<NumDimsSelfRepaired>") {
512  ReadBasicType(is, binary, &num_dims_self_repaired_);
513  ReadToken(is, binary, &token);
514  }
515  if (token == "<NumDimsProcessed>") {
516  ReadBasicType(is, binary, &num_dims_processed_);
517  ReadToken(is, binary, &token);
518  }
519  if (token == "<SelfRepairLowerThreshold>") {
520  ReadBasicType(is, binary, &self_repair_lower_threshold_);
521  ReadToken(is, binary, &token);
522  }
523  if (token == "<SelfRepairUpperThreshold>") {
524  ReadBasicType(is, binary, &self_repair_upper_threshold_);
525  ReadToken(is, binary, &token);
526  }
527  if (token == "<SelfRepairScale>") {
528  ReadBasicType(is, binary, &self_repair_scale_);
529  ReadToken(is, binary, &token);
530  }
531  if (token != ostr_end.str()) {
532  KALDI_ERR << "Expected token " << ostr_end.str()
533  << ", got " << token;
534  }
535 }

537 void NonlinearComponent::Write(std::ostream &os, bool binary) const {
538  std::ostringstream ostr_beg, ostr_end;
539  ostr_beg << "<" << Type() << ">"; // e.g. "<SigmoidComponent>"
540  ostr_end << "</" << Type() << ">"; // e.g. "</SigmoidComponent>"
541  WriteToken(os, binary, ostr_beg.str());
542  WriteToken(os, binary, "<Dim>");
543  WriteBasicType(os, binary, dim_);
544  if (block_dim_ != dim_) {
545  WriteToken(os, binary, "<BlockDim>");
546  WriteBasicType(os, binary, block_dim_);
547  }
548  // Write the values and derivatives in a count-normalized way, for
549  // greater readability in text form.
550  WriteToken(os, binary, "<ValueAvg>");
551  Vector<BaseFloat> temp(value_sum_);
552  if (count_ != 0.0) temp.Scale(1.0 / count_);
553  temp.Write(os, binary);
554 
555  WriteToken(os, binary, "<DerivAvg>");
556  temp.Resize(deriv_sum_.Dim());
557  temp.CopyFromVec(deriv_sum_);
558  if (count_ != 0.0) temp.Scale(1.0 / count_);
559  temp.Write(os, binary);
560 
561  WriteToken(os, binary, "<Count>");
562  WriteBasicType(os, binary, count_);
563 
564  WriteToken(os, binary, "<OderivRms>");
565  temp.Resize(oderiv_sumsq_.Dim());
566  temp.CopyFromVec(oderiv_sumsq_);
567  if (oderiv_count_ != 0.0) temp.Scale(1.0 / oderiv_count_);
568  // The ApplyMin() is so that the statement after it does not fail even if we
569  // had subtracted models (e.g. in full_progress.*.log).
570  temp.ApplyFloor(0.0);
