30 class CovarianceStats {
31  public:
32  CovarianceStats(int32 dim): tot_covar_(dim),
33  between_covar_(dim),
34  num_spk_(0),
35  num_utt_(0) { }
36 
38  void GetTotalCovar(SpMatrix<double> *tot_covar) const {
39  KALDI_ASSERT(num_utt_ > 0);
40  *tot_covar = tot_covar_;
41  tot_covar->Scale(1.0 / num_utt_);
42  }
43  void GetWithinCovar(SpMatrix<double> *within_covar) {
44  KALDI_ASSERT(num_utt_ - num_spk_ > 0);
45  *within_covar = tot_covar_;
46  within_covar->AddSp(-1.0, between_covar_);
47  within_covar->Scale(1.0 / num_utt_);
48  }
49  void AccStats(const Matrix<double> &utts_of_this_spk) {
50  int32 num_utts = utts_of_this_spk.NumRows();
51  tot_covar_.AddMat2(1.0, utts_of_this_spk, kTrans, 1.0);
52  Vector<double> spk_average(Dim());
53  spk_average.AddRowSumMat(1.0 / num_utts, utts_of_this_spk);
54  between_covar_.AddVec2(num_utts, spk_average);
55  num_utt_ += num_utts;
56  num_spk_ += 1;
57  }
59  bool SingularTotCovar() { return (num_utt_ < Dim()); }
60  bool Empty() { return (num_utt_ - num_spk_ == 0); }
61  std::string Info() {
62  std::ostringstream ostr;
63  ostr << num_spk_ << " speakers, " << num_utt_ << " utterances. ";
64  return ostr.str();
65  }
66  int32 Dim() { return tot_covar_.NumRows(); }
67  // Use default constructor and assignment operator.
68  void AddStats(const CovarianceStats &other) {
69  tot_covar_.AddSp(1.0, other.tot_covar_);
70  between_covar_.AddSp(1.0, other.between_covar_);
71  num_spk_ += other.num_spk_;
72  num_utt_ += other.num_utt_;
73  }
74  private:
75  KALDI_DISALLOW_COPY_AND_ASSIGN(CovarianceStats);
76  SpMatrix<double> tot_covar_;
77  SpMatrix<double> between_covar_;
78  int32 num_spk_;
79  int32 num_utt_;
80 };

84 void ComputeNormalizingTransform(const SpMatrix<Real> &covar,
85  MatrixBase<Real> *proj) {
86  int32 dim = covar.NumRows();
87  TpMatrix<Real> C(dim); // Cholesky of covar, covar = C C^T
88  C.Cholesky(covar);
89  C.Invert(); // The matrix that makes covar unit is C^{-1}, because
90  // C^{-1} covar C^{-T} = C^{-1} C C^T C^{-T} = I.
91  proj->CopyFromTp(C, kNoTrans); // set "proj" to C^{-1}.
92 }

94 void ComputeLdaTransform(
95  const std::map<std::string, Vector<BaseFloat> *> &utt2ivector,
96  const std::map<std::string, std::vector<std::string> > &spk2utt,
97  BaseFloat total_covariance_factor,
98  MatrixBase<BaseFloat> *lda_out) {
99  KALDI_ASSERT(!utt2ivector.empty());
100  int32 lda_dim = lda_out->NumRows(), dim = lda_out->NumCols();
101  KALDI_ASSERT(dim == utt2ivector.begin()->second->Dim());
102  KALDI_ASSERT(lda_dim > 0 && lda_dim <= dim);
103 
104  CovarianceStats stats(dim);
105 
106  std::map<std::string, std::vector<std::string> >::const_iterator iter;
107  for (iter = spk2utt.begin(); iter != spk2utt.end(); ++iter) {
108  const std::vector<std::string> &uttlist = iter->second;
109  KALDI_ASSERT(!uttlist.empty());
110 
111  int32 N = uttlist.size(); // number of utterances.
112  Matrix<double> utts_of_this_spk(N, dim);
113  for (int32 n = 0; n < N; n++) {
114  std::string utt = uttlist[n];
115  KALDI_ASSERT(utt2ivector.count(utt) != 0);
116  utts_of_this_spk.Row(n).CopyFromVec(
117  *(utt2ivector.find(utt)->second));
118  }
119  stats.AccStats(utts_of_this_spk);
120  }
121 
122  KALDI_LOG << "Stats have " << stats.Info();
123  KALDI_ASSERT(!stats.Empty());
124  KALDI_ASSERT(!stats.SingularTotCovar() &&
125  "Too little data for iVector dimension.");
126 
127 
128  SpMatrix<double> total_covar;
129  stats.GetTotalCovar(&total_covar);
130  SpMatrix<double> within_covar;
131  stats.GetWithinCovar(&within_covar);
132 
133 
134  SpMatrix<double> mat_to_normalize(dim);
135  mat_to_normalize.AddSp(total_covariance_factor, total_covar);
136  mat_to_normalize.AddSp(1.0 - total_covariance_factor, within_covar);
137 
138  Matrix<double> T(dim, dim);
139  ComputeNormalizingTransform(mat_to_normalize, &T);
140 
141  SpMatrix<double> between_covar(total_covar);
142  between_covar.AddSp(-1.0, within_covar);
143 
144  SpMatrix<double> between_covar_proj(dim);
145  between_covar_proj.AddMat2Sp(1.0, T, kNoTrans, between_covar, 0.0);
146 
147  Matrix<double> U(dim, dim);
148  Vector<double> s(dim);
149  between_covar_proj.Eig(&s, &U);
150  bool sort_on_absolute_value = false; // any negative ones will go last (they
151  // shouldn't exist anyway so doesn't
152  // really matter)
153  SortSvd(&s, &U, static_cast<Matrix<double>*>(NULL),
154  sort_on_absolute_value);
155 
156  KALDI_LOG << "Singular values of between-class covariance after projecting "
157  << "with interpolated [total/within] covariance with a weight of "
158  << total_covariance_factor << " on the total covariance, are: " << s;
159 
160  // U^T is the transform that will diagonalize the between-class covariance.
161  // U_part is just the part of U that corresponds to the kept dimensions.
162  SubMatrix<double> U_part(U, 0, dim, 0, lda_dim);
163 
164  // We first transform by T and then by U_part^T. This means T
165  // goes on the right.
166  Matrix<double> temp(lda_dim, dim);
167  temp.AddMatMat(1.0, U_part, kTrans, T, kNoTrans, 0.0);
168  lda_out->CopyFromMat(temp);
169 }

171 void ComputeAndSubtractMean(
172  std::map<std::string, Vector<BaseFloat> *> utt2ivector,
173  Vector<BaseFloat> *mean_out) {
174  int32 dim = utt2ivector.begin()->second->Dim();
175  size_t num_ivectors = utt2ivector.size();
176  Vector<double> mean(dim);
177  std::map<std::string, Vector<BaseFloat> *>::iterator iter;
178  for (iter = utt2ivector.begin(); iter != utt2ivector.end(); ++iter)
179  mean.AddVec(1.0 / num_ivectors, *(iter->second));
180  mean_out->Resize(dim);
181  mean_out->CopyFromVec(mean);
182  for (iter = utt2ivector.begin(); iter != utt2ivector.end(); ++iter)
183  iter->second->AddVec(-1.0, *mean_out);
184 }

190 int main(int argc, char *argv[]) {
191  using namespace kaldi;
192  typedef kaldi::int32 int32;
193  try {
194  const char *usage =
195  "Compute an LDA matrix for iVector system. Reads in iVectors per utterance,\n"
196  "and an utt2spk file which it uses to help work out the within-speaker and\n"
197  "between-speaker covariance matrices. Outputs an LDA projection to a\n"
198  "specified dimension. By default it will normalize so that the projected\n"
199  "within-class covariance is unit, but if you set --normalize-total-covariance\n"
200  "to true, it will normalize the total covariance.\n"
201  "Note: the transform we produce is actually an affine transform which will\n"
202  "also set the global mean to zero.\n"
203  "\n"
204  "Usage: ivector-compute-lda [options] <ivector-rspecifier> <utt2spk-rspecifier> "
205  "<lda-matrix-out>\n"
206  "e.g.: \n"
207  " ivector-compute-lda ark:ivectors.ark ark:utt2spk lda.mat\n";
208 
209  ParseOptions po(usage);
210 
211  int32 lda_dim = 100; // Dimension we reduce to
212  BaseFloat total_covariance_factor = 0.0;
213  bool binary = true;
214 
215  po.Register("dim", &lda_dim, "Dimension we keep with the LDA transform");
216  po.Register("total-covariance-factor", &total_covariance_factor,
217  "If this is 0.0 we normalize to make the within-class covariance "
218  "unit; if 1.0, the total covariance; if between, we normalize "
219  "an interpolated matrix.");
220  po.Register("binary", &binary, "Write output in binary mode");
221 
222  po.Read(argc, argv);
223 
224  if (po.NumArgs() != 3) {
225  po.PrintUsage();
226  exit(1);
227  }
228 
229  std::string ivector_rspecifier = po.GetArg(1),
230  utt2spk_rspecifier = po.GetArg(2),
231  lda_wxfilename = po.GetArg(3);
232 
233  int32 num_done = 0, num_err = 0, dim = 0;
234 
235  SequentialBaseFloatVectorReader ivector_reader(ivector_rspecifier);
236  RandomAccessTokenReader utt2spk_reader(utt2spk_rspecifier);
237 
238  std::map<std::string, Vector<BaseFloat> *> utt2ivector;
239  std::map<std::string, std::vector<std::string> > spk2utt;
240 
241  for (; !ivector_reader.Done(); ivector_reader.Next()) {
242  std::string utt = ivector_reader.Key();
243  const Vector<BaseFloat> &ivector = ivector_reader.Value();
244  if (utt2ivector.count(utt) != 0) {
245  KALDI_WARN << "Duplicate iVector found for utterance " << utt
246  << ", ignoring it.";
247  num_err++;
248  continue;
249  }
250  if (!utt2spk_reader.HasKey(utt)) {
251  KALDI_WARN << "utt2spk has no entry for utterance " << utt
252  << ", skipping it.";
253  num_err++;
254  continue;
255  }
256  std::string spk = utt2spk_reader.Value(utt);
257  utt2ivector[utt] = new Vector<BaseFloat>(ivector);
258  if (dim == 0) {
259  dim = ivector.Dim();
260  } else {
261  KALDI_ASSERT(dim == ivector.Dim() && "iVector dimension mismatch");
262  }
263  spk2utt[spk].push_back(utt);
264  num_done++;
265  }
266 
267  KALDI_LOG << "Read " << num_done << " utterances, "
268  << num_err << " with errors.";
269 
270  if (num_done == 0) {
271  KALDI_ERR << "Did not read any utterances.";
272  } else {
273  KALDI_LOG << "Computing within-class covariance.";
274  }
275 
276  Vector<BaseFloat> mean;
277  ComputeAndSubtractMean(utt2ivector, &mean);
278  KALDI_LOG << "2-norm of iVector mean is " << mean.Norm(2.0);
279 
280 
281  Matrix<BaseFloat> lda_mat(lda_dim, dim + 1); // LDA matrix without the offset term.
282  SubMatrix<BaseFloat> linear_part(lda_mat, 0, lda_dim, 0, dim);
283  ComputeLdaTransform(utt2ivector,
284  spk2utt,
285  total_covariance_factor,
286  &linear_part);
287  Vector<BaseFloat> offset(lda_dim);
288  offset.AddMatVec(-1.0, linear_part, kNoTrans, mean, 0.0);
289  lda_mat.CopyColFromVec(offset, dim); // add mean-offset to transform
290 
291  KALDI_VLOG(2) << "2-norm of transformed iVector mean is "
292  << offset.Norm(2.0);
293 
294  WriteKaldiObject(lda_mat, lda_wxfilename, binary);
295 
296  KALDI_LOG << "Wrote LDA transform to "
297  << PrintableWxfilename(lda_wxfilename);
298 
299  std::map<std::string, Vector<BaseFloat> *>::iterator iter;
300  for (iter = utt2ivector.begin(); iter != utt2ivector.end(); ++iter)
301  delete iter->second;
302  utt2ivector.clear();
303 
304  return 0;
305  } catch(const std::exception &e) {
306  std::cerr << e.what();
307  return -1;
308  }
309 }
