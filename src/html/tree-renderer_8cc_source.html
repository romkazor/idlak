29 TreeRenderer::RenderNonLeaf(int32 id, const EventKeyType &key, bool in_query) {
30  std::string color = in_query? kEdgeColorQuery: kEdgeColor;
31  int32 width = in_query? kEdgeWidthQuery: kEdgeWidth;
32  std::string label;
33  if (key == kPdfClass) {
34  label = "\"PdfClass = ?\"";
35  } else if (key == 0) {
36  if (N_ == 1 && P_ == 0) // monophone tree?
37  label = "\"Phone = ?\"";
38  else if (N_ == 3 && P_ == 1) // triphone tree?
39  label = "\"LContext = ?\"";
40  } else if (key == 2 && N_ == 3 && P_ == 1) {
41  label = "\"RContext = ?\"";
42  } else if (key >= 0 && key <= N_-1) {
43  if (P_ == key)
44  label = "\"Center = ?\"";
45  else {
46  std::ostringstream oss;
47  oss << "\"Ctx Position " << key << " = ?\"";
48  label = oss.str();
49  }
50  } else {
51  KALDI_ERR << "Invalid decision tree key: " << key;
52  }
53 
54  out_ << id << "[label=" << label << ", color=" << color
55  << ", penwidth=" << width << "];" << std::endl;
56 }

80 void TreeRenderer::RenderSplit(const EventType *query, int32 id) {
81  ExpectToken(is_, binary_, "SE");
82  EventKeyType key;
83  ReadBasicType(is_, binary_, &key);
84  ConstIntegerSet<EventValueType> yes_set;
85  yes_set.Read(is_, binary_);
86  ExpectToken(is_, binary_, "{");
87 
88  EventValueType value = -30000000; // just a value I guess is invalid
89  if (query != NULL)
90  EventMap::Lookup(*query, key, &value);
91  const EventType *query_yes = yes_set.count(value)? query: NULL;
92  const EventType *query_no = (query_yes == NULL)? query: NULL;
93  std::string color_yes = (query_yes)? kEdgeColorQuery: kEdgeColor;
94  std::string color_no = (query && !query_yes)? kEdgeColorQuery: kEdgeColor;
95  int32 width_yes = (query_yes)? kEdgeWidthQuery: kEdgeWidth;
96  int32 width_no = (query && !query_yes)? kEdgeWidthQuery: kEdgeWidth;
97  RenderNonLeaf(id, key, (query != NULL)); // Draw the node itself
98  std::string yes_label = MakeEdgeLabel(key, yes_set);
99  out_ << "\t" << id << " -> " << next_id_++ << " ["; // YES edge
100  if (use_tooltips_) {
101  out_ << "tooltip=\"" << yes_label << "\", label=YES"
102  << ", penwidth=" << width_yes << ", color=" << color_yes << "];\n";
103  } else {
104  out_ << "label=\"" << yes_label << "\", penwidth=" << width_yes
105  << ", penwidth=" << width_yes << ", color=" << color_yes << "];\n";
106  }
107  RenderSubTree(query_yes, next_id_-1); // Render YES subtree
108  out_ << "\t" << id << " -> " << next_id_++ << "[label=NO" // NO edge
109  << ", color=" << color_no << ", penwidth=" << width_no << "];\n";
110  RenderSubTree(query_no, next_id_-1); // Render NO subtree
111 
112  ExpectToken(is_, binary_, "}");
113 }

115 void TreeRenderer::RenderTable(const EventType *query, int32 id) {
116  ExpectToken(is_, binary_, "TE");
117  EventKeyType key;
118  ReadBasicType(is_, binary_, &key);
119  uint32 size;
120  ReadBasicType(is_, binary_, &size);
121  ExpectToken(is_, binary_, "(");
122 
123  EventValueType value = -3000000; // just a value I hope is invalid
124  if (query != NULL)
125  EventMap::Lookup(*query, key, &value);
126  RenderNonLeaf(id, key, (query != NULL));
127  for (size_t t = 0; t < size; t++) {
128  std::string color = (t == value)? kEdgeColorQuery: kEdgeColor;
129  int32 width = (t==value)? kEdgeWidthQuery: kEdgeWidth;
130  std::ostringstream label;
131  if (key == kPdfClass) {
132  label << t;
133  } else if (key >= 0 && key < N_) {
134  if (t == 0) {
135  ExpectToken(is_, binary_, "NULL"); // consume the invalid/NULL entry
136  continue;
137  }
138  std::string phone = phone_syms_.Find(static_cast<kaldi::int64>(t));
139  if (phone.empty())
140  KALDI_ERR << "Phone ID found in a TableEventMap, but not in the "
141  << "phone symbol table! ID: " << t;
142  label << phone;
143  } else {
144  KALDI_ERR << "TableEventMap: Invalid event key: " << key;
145  }
146  // draw the edge to the child subtree
147  out_ << "\t" << id << " -> " << next_id_++ << " [label=" << label.str()
148  << ", color=" << color << ", penwidth=" << width << "];\n";
149  const EventType *query_child = (t == value)? query: NULL;
150  RenderSubTree(query_child, next_id_-1); // render the child subtree
151  }
152 
153  ExpectToken(is_, binary_, ")");
154 }

167 void TreeRenderer::RenderSubTree(const EventType *query, int32 id) {
168  char c = Peek(is_, binary_);
169  if (c == 'N') {
170  ExpectToken(is_, binary_, "NULL"); // consume NULL entries
171  return;
172  } else if (c == 'C') {
173  RenderConstant(query, id);
174  } else if (c == 'T') {
175  RenderTable(query, id);
176  } else if (c == 'S') {
177  RenderSplit(query, id);
178  } else {
179  KALDI_ERR << "EventMap::read, was not expecting character " << CharToString(c)
180  << ", at file position " << is_.tellg();
181  }
182 }

184 void TreeRenderer::Render(const EventType *query = 0) {
185  ExpectToken(is_, binary_, "ContextDependency");
186  ReadBasicType(is_, binary_, &N_);
187  ReadBasicType(is_, binary_, &P_);
188  ExpectToken(is_, binary_, "ToPdf");
189  if (query && query->size() != N_+1)
190  KALDI_ERR << "Invalid query size \"" << query->size() << "\"! Expected \""
191  << N_+1 << '"';
192  out_ << "digraph EventMap {\n";
193  RenderSubTree(query, next_id_++);
194  out_ << "}\n";
195  ExpectToken(is_, binary_, "EndContextDependency");
196 }
