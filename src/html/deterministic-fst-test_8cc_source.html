28 bool FileExists(string strFilename) {
29  struct stat stFileInfo;
30  bool blnReturn;
31  int intStat;
32 
33  // Attempt to get the file attributes
34  intStat = stat(strFilename.c_str(), &stFileInfo);
35  if (intStat == 0) {
36  // We were able to get the file attributes
37  // so the file obviously exists.
38  blnReturn = true;
39  } else {
40  // We were not able to get the file attributes.
41  // This may mean that we don't have permission to
42  // access the folder which contains this file. If you
43  // need to do that level of checking, lookup the
44  // return values of stat which will give you
45  // more details on why stat failed.
46  blnReturn = false;
47  }
48 
49  return blnReturn;
50 }

61 StdVectorFst* CreateBackoffFst() {
62  StdVectorFst *fst = new StdVectorFst();
63  fst->AddState(); // state 0
64  fst->SetStart(0);
65  fst->AddArc(0, StdArc(10, 10, 0.0, 1));
66 
67  fst->AddState(); // state 1
68  fst->AddArc(1, StdArc(12, 12, 0.0, 4));
69  fst->AddArc(1, StdArc(0,0, 0.1,2)); // backoff from 1 to 2
70 
71  fst->AddState(); // state 2
72  fst->AddArc(2, StdArc(13, 13, 0.2, 4));
73  fst->AddArc(2, StdArc(0,0, 0.3,3)); // backoff from 2 to 3
74 
75  fst->AddState(); // state 3
76  fst->AddArc(3, StdArc(14, 14, 0.4, 4));
77 
78  fst->AddState(); // state 4
79  fst->AddArc(4, StdArc(15, 15, 0.5, 5));
80 
81  fst->AddState(); // state 5
82  fst->SetFinal(5, 0.6);
83 
84  return fst;

115 // Follow paths from an input fst representing a string
116 // (poor man's composition)
117 Weight WalkSinglePath(StdVectorFst *ifst, DeterministicOnDemandFst<StdArc> *dfst) {
118  StdArc oarc; // = new StdArc();
119  StateId isrc=ifst->Start();
120  StateId dsrc=dfst->Start();
121  Weight totalCost = Weight::One();
122 
123  while (ifst->Final(isrc) == Weight::Zero()) { // while not final
124  fst::ArcIterator<StdVectorFst> aiter(*ifst, isrc);
125  const StdArc &iarc = aiter.Value();
126  if (dfst->GetArc(dsrc, iarc.olabel, &oarc)) {
127  Weight cost = Times(iarc.weight, oarc.weight);
128  // cout << " Matched label "<<iarc.olabel<<" at summed cost "<<cost<<endl;
129  totalCost = Times(totalCost, cost);
130  } else {
131  cout << " Can't match arc ["<<iarc.ilabel<<","<<iarc.olabel<<","<<iarc.weight<<"] from "<<isrc<<endl;
132  exit(1);
133  }
134  isrc = iarc.nextstate;
135  KALDI_LOG << "Setting dsrc = " << oarc.nextstate;
136  dsrc = oarc.nextstate;
137  }
138  totalCost = Times(totalCost, dfst->Final(dsrc));
139 
140  cout << " Total cost: " << totalCost << endl;
141  return totalCost;
142 }

145 void TestBackoffAndCache() {
146  // Build from existing fst
147  cout << "Test with single generated backoff FST" << endl;
148  StdVectorFst *nfst = CreateBackoffFst();
149  StdVectorFst *rfst = CreateResultFst();
150 
151  // before using, make sure that it is input sorted
152  ArcSort(nfst, StdILabelCompare());
153  BackoffDeterministicOnDemandFst<StdArc> dfst1a(*nfst);
154  CacheDeterministicOnDemandFst<StdArc> dfst1(&dfst1a);
155 
156  // Compare all arcs in dfst1 with expected result
157  for (StateIterator<StdVectorFst> riter(*rfst); !riter.Done(); riter.Next()) {
158  StateId rsrc = riter.Value();
159  // verify that states have same weight (or final status)
160  assert(ApproxEqual(rfst->Final(rsrc), dfst1.Final(rsrc)));
161  for (ArcIterator<StdVectorFst> aiter(*rfst, rsrc); !aiter.Done(); aiter.Next()) {
162  StdArc rarc = aiter.Value();
163  StdArc darc;
164  if (dfst1.GetArc(rsrc, rarc.ilabel, &darc)) {
165  assert(ApproxEqual(rarc.weight, darc.weight, 0.001));
166  assert(rarc.ilabel==darc.ilabel);
167  assert(rarc.olabel==darc.olabel);
168  assert(rarc.nextstate == darc.nextstate);
169  cerr << " Got same arc at state "<<rsrc<<": "<<rarc.ilabel<<" "<<darc.ilabel<<endl;
170  } else {
171  cerr << "Couldn't find arc "<<rarc.ilabel<<" for state "<<rsrc<<endl;
172  exit(1);
173  }
174  }
175  }
176  delete nfst;
177  delete rfst;
178 }
