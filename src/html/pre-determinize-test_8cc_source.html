30 template<class Arc> void TestPreDeterminize() {
31  typedef typename Arc::Label Label;
32  typedef typename Arc::StateId StateId;
33  typedef typename Arc::Weight Weight;
34 
35  VectorFst<Arc> *fst = new VectorFst<Arc>();
36  int n_syms = 2 + kaldi::Rand() % 5, n_states = 3 + kaldi::Rand() % 10, n_arcs = 5 + kaldi::Rand() % 30, n_final = 1 + kaldi::Rand()%3; // Up to 2 unique symbols.
37  cout << "Testing pre-determinize with "<<n_syms<<" symbols, "<<n_states<<" states and "<<n_arcs<<" arcs and "<<n_final<<" final states.\n";
38  SymbolTable *sptr = NULL;
39 
40  vector<Label> all_syms; // including epsilon.
41  // Put symbols in the symbol table from 1..n_syms-1.
42  for (size_t i = 0;i < (size_t)n_syms;i++)
43  all_syms.push_back(i);
44 
45  // Create states.
46  vector<StateId> all_states;
47  for (size_t i = 0;i < (size_t)n_states;i++) {
48  StateId this_state = fst->AddState();
49  if (i == 0) fst->SetStart(i);
50  all_states.push_back(this_state);
51  }
52  // Set final states.
53  for (size_t j = 0;j < (size_t)n_final;j++) {
54  StateId id = all_states[kaldi::Rand() % n_states];
55  Weight weight = (Weight)(0.33*(kaldi::Rand() % 5) );
56  printf("calling SetFinal with %d and %f\n", id, weight.Value());
57  fst->SetFinal(id, weight);
58  }
59  // Create arcs.
60  for (size_t i = 0;i < (size_t)n_arcs;i++) {
61  Arc a;
62  a.nextstate = all_states[kaldi::Rand() % n_states];
63  a.ilabel = all_syms[kaldi::Rand() % n_syms];
64  a.olabel = all_syms[kaldi::Rand() % n_syms]; // same input+output vocab.
65  a.weight = (Weight) (0.33*(kaldi::Rand() % 2));
66  StateId start_state = all_states[kaldi::Rand() % n_states];
67  fst->AddArc(start_state, a);
68  }
69 
70  std::cout <<" printing before trimming\n";
71  {
72  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
73  fstprinter.Print(&std::cout, "standard output");
74  }
75  // Trim resulting FST.
76  Connect(fst);
77 
78  std::cout <<" printing after trimming\n";
79  {
80  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
81  fstprinter.Print(&std::cout, "standard output");
82  }
83 
84  VectorFst<Arc> *fst_copy_orig = new VectorFst<Arc>(*fst);
85 
86  vector<Label> extra_syms;
87  if (fst->Start() != kNoStateId) { // "Connect" did not make it empty....
88  typename Arc::Label highest_sym = HighestNumberedInputSymbol(*fst);
89  PreDeterminize(fst, highest_sym+1, &extra_syms);
90  }
91 
92  std::cout <<" printing after predeterminization\n";
93  {
94  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
95  fstprinter.Print(&std::cout, "standard output");
96  }
97 
98 
99  { // Remove epsilon. All default args.
100  bool connect = true;
101  Weight weight_threshold = Weight::Zero();
102  int64 nstate = -1; // Relates to pruning.
103  double delta = kDelta; // I think a small weight value. Relates to some kind of pruning,
104  // I guess. But with no epsilon cycles, probably doensn't matter.
105  RmEpsilon(fst, connect, weight_threshold, nstate, delta);
106  }
107 
108  std::cout <<" printing after epsilon removal\n";
109  {
110  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
111  fstprinter.Print(&std::cout, "standard output");
112  }
113 
114 
115  VectorFst<Arc> ofst;
116  DeterminizeOptions<Arc> opts; // Default options.
117  Determinize(*fst, &ofst, opts);
118  std::cout <<" printing after determinization\n";
119  {
120  FstPrinter<Arc> fstprinter(ofst, sptr, sptr, NULL, false, true, "\t");
121  fstprinter.Print(&std::cout, "standard output");
122  }
123 
124  int64 num_removed = DeleteISymbols(&ofst, extra_syms);
125  std::cout <<" printing after removing "<<num_removed<<" instances of extra symbols\n";
126  {
127  FstPrinter<Arc> fstprinter(ofst, sptr, sptr, NULL, false, true, "\t");
128  fstprinter.Print(&std::cout, "standard output");
129  }
130 
131  std::cout <<" Checking equivalent to original FST.\n";
132  // giving Rand() as a seed stops the random number generator from always being reset to
133  // the same point each time, while maintaining determinism of the test.
134  assert(RandEquivalent(ofst, *fst_copy_orig, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length-- max?*/));
135 
136  delete fst;
137  delete fst_copy_orig;
138 }

140 template<class Arc> void TestAddSelfLoops() {
141  typedef typename Arc::Label Label;
142  typedef typename Arc::StateId StateId;
143  typedef typename Arc::Weight Weight;
144 
145  VectorFst<Arc> *fst = new VectorFst<Arc>();
146  SymbolTable *ilabels = new SymbolTable("my-symbol-table");
147  SymbolTable *olabels = new SymbolTable("my-symbol-table-2");
148  Label i0 = ilabels->AddSymbol("<eps>");
149  Label i1 = ilabels->AddSymbol("1");
150  Label i2 = ilabels->AddSymbol("2");
151 
152  Label o0 = olabels->AddSymbol("<eps>");
153  Label o1 = olabels->AddSymbol("1");
154 
155  assert(i0 == 0 && o0 == 0);
156  StateId s0 = fst->AddState(), s1 = fst->AddState(), s2 = fst->AddState();
157  fst->SetStart(s0);
158  assert(s0 == 0);
159 
160  fst->SetFinal(s2, (Weight)2); // state 2 is final.
161  {
162  Arc arc;
163  arc.ilabel = i1;
164  arc.olabel = o0;
165  arc.nextstate = 1;
166  arc.weight = (Weight)1;
167  fst->AddArc(s0, arc); // arc from 0 to 1 with epsilon out.
168  }
169  {
170  Arc arc;
171  arc.ilabel = i2;
172  arc.olabel = o1;
173  arc.nextstate = 2;
174  arc.weight = (Weight)2;
175  fst->AddArc(s1, arc); // arc from 1 to 2 with "1" out.
176  }
177  std::cout <<" printing before adding self-loops\n";
178  {
179  FstPrinter<Arc> fstprinter(*fst, ilabels, olabels, NULL, false, true, "\t");
180  fstprinter.Print(&std::cout, "standard output");
181  }
182 
183 
184  // So states 1 and 2 should have self-loops on.
185  size_t num_extra = kaldi::Rand() % 5;
186  vector<Label> extra_ilabels, extra_olabels;
187  CreateNewSymbols(ilabels, num_extra, "in#", &extra_ilabels);
188  CreateNewSymbols(olabels, num_extra, "out#", &extra_olabels);
189 
190  AddSelfLoops(fst, extra_ilabels, extra_olabels);
191 
192  assert(fst->NumArcs(0) == 1);
193  assert(fst->NumArcs(1) == 1 + num_extra);
194  assert(fst->NumArcs(2) == num_extra);
195 
196  std::cout <<" printing after adding self-loops\n";
197  {
198  FstPrinter<Arc> fstprinter(*fst, ilabels, olabels, NULL, false, true, "\t");
199  fstprinter.Print(&std::cout, "standard output");
200  }
201 
202  delete fst;
203  delete ilabels;
204  delete olabels;
205 }
