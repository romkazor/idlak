28 // this comparator will be used to sort pairs using first_element
29 // we declare it as a struct as it will also be used by std::lower_bound
30 // method which will supply elements of different types to the function
31 struct FirstElementComparator {
32  int first_element(int32 t) const {
33  return t;
34  }
35 
36  int first_element(std::pair<int32, int32> t) const {
37  return t.first;
38  }
39 
40  template< typename T1, typename T2>
41  bool operator()( T1 const & t1, T2 const & t2) const {
42  return first_element(t1) < first_element(t2);
43  }
44 };

46 // This comparator is used with std::find_if function to search for pairs
47 // whose first element is equal to the given pair
48 struct FirstElementIsEqualComparator :
49  public std::unary_function<std::pair<int32, int32>, bool>
50 {
51  explicit FirstElementIsEqualComparator(const int32 element):
52  element_(element) {}
53  bool operator() (std::pair<int32, int32> const &arg)
54  { return (arg.first == element_); }
55  int32 element_;
56 };

58 // This comparator is used with std::find_if function to search for pairs
59 // whose .first and .second elements are equal to the given pair
60 struct PairIsEqualComparator :
61  public std::unary_function<std::pair<int32, int32>, bool>
62 {
63  explicit PairIsEqualComparator(const std::pair<int32, int32> pair):
64  pair_(pair) {}
65  bool operator() (std::pair<int32, int32> const &arg)
66  {
67  if (pair_.first == arg.first)
68  return pair_.second == arg.second;
69  return false;
70  }
71  std::pair<int32, int32> pair_;
72 };

74 // this comparator will be used to sort pairs initially by second element in
75 // descending order and then by first element in descending order.
76 // note, std::sort accepts an actual function as an alternative to a
77 // function object.
78 bool SecondElementComparator(const std::pair<int32, int32>& first_pair,
79  const std::pair<int32, int32>& second_pair) {
80  if (first_pair.second == second_pair.second)
81  return first_pair.first > second_pair.first;
82  return first_pair.second > second_pair.second;
83 }

85 // Function to sort the lists in a vector of lists of pairs, by the first
86 // element of the pair
87 void SortSubmatLists(
88  // vector of list of location pairs
89  const std::vector<std::vector<std::pair<int32, int32> > > submat_lists,
90  // a copy of the input submat_lists where the lists are sorted
91  // (this will be used in the caller function for sort and find functions)
92  std::vector<std::vector<std::pair<int32, int32> > > * sorted_submat_lists,
93  // maximum size of the submat_lists
94  int32* max_submat_list_size
95  )
96 {
97  *max_submat_list_size = 0;
98  sorted_submat_lists->reserve(submat_lists.size());
99  KALDI_ASSERT(submat_lists.size() > 0);
100  for (int32 i = 0; i < submat_lists.size(); i++) {
101  if (submat_lists[i].size() > *max_submat_list_size)
102  *max_submat_list_size = submat_lists[i].size();
103  sorted_submat_lists->push_back(submat_lists[i]);
104  std::sort((*sorted_submat_lists)[i].begin(),
105  (*sorted_submat_lists)[i].end(),
106  FirstElementComparator());
107  }
108 }

110 // Function to compute a histogram of the submat_index,
111 // which is the first_element in the location pair, given vector of list of
112 // location pairs
113 void ComputeSubmatIndexHistogram(
114  // vector of list of pairs of location pairs where the lists are sorted
115  // by submat_indexes (.first element)
116  const std::vector<std::vector<std::pair<int32, int32> > >
117  sorted_submat_lists,
118  // a histogram of submat_indexes where
119  // the keys are submat_indexes and values are a vector of frequencies
120  // of first occurrence, second occurrence, etc. of a submat_index
121  // in a submat_list
122  unordered_map<int32, std::vector<int32> >* submat_histogram
123  ) {
124  KALDI_ASSERT(sorted_submat_lists.size() > 0);
125  // computing the submat_histogram
126  // counting the occurrences of each element in the current submat_list;
127  // each new occurrence of the same element, in this list, is counted
128  // as a seperate symbol for frequency counts
129  for (int32 i = 0; i < sorted_submat_lists.size(); i++) {
130  int j = 0;
131  unordered_map<int32, std::vector<int32> >::iterator histogram_iterator
132  = submat_histogram->end();
133  int32 repetition_count = 0;
134  while (j < sorted_submat_lists[i].size()) {
135  if ((histogram_iterator == submat_histogram->end()) ||
136  (histogram_iterator->first != sorted_submat_lists[i][j].first)) {
137  histogram_iterator =
138  submat_histogram->find(sorted_submat_lists[i][j].first);
139  repetition_count = 0;
140  // if a histogram entry was not found for this submat_index, add one
141  if (histogram_iterator == submat_histogram->end()) {
142  (*submat_histogram)[sorted_submat_lists[i][j].first];
143  histogram_iterator = submat_histogram->find(
144  sorted_submat_lists[i][j].first);
145  }
146  }
147 
148  if (repetition_count >= (histogram_iterator->second).size()) {
149  // this is the first time the submat_index repeated this many times
150  // so add an entry for this in the count vector
151  (histogram_iterator->second).push_back(1);
152  } else {
153  (histogram_iterator->second)[repetition_count]++;
154  }
155  repetition_count++;
156  j++;
157  }
158  }
159 }

162 // Function to find the first occurrence of a submat_index in list of location
163 // pairs from a vector of list of locations pairs.
164 // The occurrences are returned as a list of vector iterators,
165 // pointing to the position of the pair in the list or to the
166 // end of the list (when the pair is not present)
167 void FindSubmatIndexInSubmatLists(
168  // submat_index to search in the submat_lists
169  int32 submat_index,
170  // sorted_submat_lists is a pointer as we want non-const iterators in the
171  // output
172  std::vector<std::vector<std::pair<int32, int32> > > *sorted_submat_lists,
173  // a vector of iterators to store the location of the pairs
174  std::vector<std::vector<std::pair<int32, int32> >::iterator>
175  *output_iterator_list,
176  // the max size of the submat_lists if the found pairs have been removed
177  int32 *max_remaining_submat_list_size) {
178 
179  output_iterator_list->reserve(sorted_submat_lists->size());
180  *max_remaining_submat_list_size = 0;
181  for (int32 i = 0; i < sorted_submat_lists->size(); i++) {
182  std::vector< std::pair<int32, int32> > & submat_list =
183  (*sorted_submat_lists)[i];
184  output_iterator_list->push_back(
185  std::find_if(submat_list.begin(), submat_list.end(),
186  FirstElementIsEqualComparator(submat_index)));
187  int32 remaining_submat_list_size = submat_list.size();
188  if (output_iterator_list->back() != submat_list.end()) {
189  // since the submat_index is present in this submat_list
190  // if submat_index was deleted from the list
191  // the remaining submat_list's size is reduced by 1
192  remaining_submat_list_size--;
193  }
194  *max_remaining_submat_list_size = remaining_submat_list_size
195  > *max_remaining_submat_list_size ? remaining_submat_list_size :
196  *max_remaining_submat_list_size;
197  }
198 }

200 // Function to extract the identified pairs (identified with an iterator)
201 // from a vector of list of pairs, "to extract" means to copy into
202 // a list and erase the original pair from the submat_lists
203 void ExtractGivenPairsFromSubmatLists(
204  std::vector<std::vector<std::pair<int32, int32> >::iterator>
205  &input_iterator_list,
206  std::vector<std::vector<std::pair<int32, int32> > > *sorted_submat_lists,
207  std::vector<std::pair<int32, int32> > *list_of_pairs) {
208  list_of_pairs->reserve(sorted_submat_lists->size());
209  for (int32 i = 0; i < input_iterator_list.size(); i++) {
210  if (input_iterator_list[i] != (*sorted_submat_lists)[i].end()) {
211  // there was an element with the submat_index in the current list
212  list_of_pairs->push_back(*input_iterator_list[i]);
213  (*sorted_submat_lists)[i].erase(input_iterator_list[i]);
214  } else {
215  // insert a dummy element. Callers of this function expect the dummy
216  // element to be (-1, -1)
217  list_of_pairs->push_back(std::make_pair(-1, -1));
218  }
219  }
220 }

222 // Function to extract the last pairs from a vector of list of pairs
223 // a dummy is added when the list is empty
224 static void ExtractLastPairFromSubmatLists(
225  std::vector<std::vector<std::pair<int32, int32> > > *sorted_submat_lists,
226  std::vector<std::pair<int32, int32> > *list_of_pairs) {
227  list_of_pairs->reserve(sorted_submat_lists->size());
228  for (int32 i = 0; i < sorted_submat_lists->size(); i++) {
229  if ((*sorted_submat_lists)[i].size() == 0) {
230  // the value of the dummy has to be (-1, -1) as down stream code has
231  // expects -1 values for dummies
232  list_of_pairs->push_back(std::make_pair(-1, -1));
233  continue;
234  }
235  list_of_pairs->push_back((*sorted_submat_lists)[i].back());
236  (*sorted_submat_lists)[i].pop_back();
237  }
238 }

240 // Function which does the actual splitting of submat_lists. But it operates on
241 // sorted submat_lists and uses submat_histogram_vector.
242 // See SplitLocations, below for the algorithm
243 static void SplitLocationsUsingSubmatHistogram(
244  // maximum size of the lists in the sorted_submat_lists
245  int32 max_submat_list_size,
246  // a vector of list of pairs where each list is expected to be sorted
247  // this is a pointer as the lists will be modified
248  std::vector<std::vector<std::pair<int32, int32> > > *sorted_submat_lists,
249  // a vector of pairs to represent a histogram
250  // this is a pointer as the vector will be sorted
251  std::vector<std::pair<int32, int32> > *submat_histogram_vector,
252  // a vector of lists of pairs with rearranged pairs
253  std::vector<std::vector<std::pair<int32, int32> > > *split_lists) {
254 
255  // sort the submat_histogram_vector based on second element of pair
256  // in descending order then first element of pair in descending order
257  std::sort(submat_histogram_vector->begin(),
258  submat_histogram_vector->end(), SecondElementComparator);
259 
260  int32 prev_max_remaining_submat_list_size = max_submat_list_size;
261  std::vector<std::pair<int32, int32> >::iterator iter;
262  for (iter = submat_histogram_vector->begin();
263  iter != submat_histogram_vector->end();
264  ++iter) {
265  std::pair<int32, int32> submat_index_and_count = *iter;
266  std::vector<std::vector<std::pair<int32, int32> >::iterator>
267  output_iterator_list;
268  int32 max_remaining_submat_list_size = 0;
269  FindSubmatIndexInSubmatLists(submat_index_and_count.first,
270  sorted_submat_lists,
271  &output_iterator_list,
272  &max_remaining_submat_list_size);
273  if (max_remaining_submat_list_size
274  < prev_max_remaining_submat_list_size) {
275  // since we will have a smaller max_remaining_submat_list_size by
276  // splitting this submat_index into a seperate list,
277  // we will split it;
278  std::vector<std::pair<int32, int32> > list_of_pairs;
279  ExtractGivenPairsFromSubmatLists(output_iterator_list,
280  sorted_submat_lists,
281  &list_of_pairs);
282  split_lists->push_back(list_of_pairs);
283  prev_max_remaining_submat_list_size = max_remaining_submat_list_size;
284  }
285  }
286 
287  // rearrange the remaining pairs into lists where
288  // pairs with multiple first elements are allowed
289  // Note : we don't yet know if there is any advantage of having multiple
290  // calls to the same submat in kAddRowsMulti. If this is actually helpful
291  // then use the sorted_histogram_vector to first copy submat_indexes which
292  // did not make it to kAddRows calls
293  for (int32 i = 0; i < prev_max_remaining_submat_list_size; i++) {
294  std::vector<std::pair<int32, int32> > list_of_pairs;
295  ExtractLastPairFromSubmatLists(sorted_submat_lists,
296  &list_of_pairs);
297  split_lists->push_back(list_of_pairs);
298  }
299 }

301 // Function rearranges the submat_lists (see nnet-compute-utils.h for
302 // description of submat_lists), into lists that can be used as inputs
303 // for kAddRows and kAddRowsMulti calls.
304 // kAddRows requires a list of pairs where all the first elements correspond to
305 // the same submat_index.
306 // kAddRowsMulti uses a list of pairs where the first elements can correspond to
307 // multiple submat_index locations.
308 // ------------------------
309 // The maximum size of a list in submat_lists is the minimum number of
310 // kAddRowsMulti calls necessary.
311 // In the current implementation we replace kAddRowsMulti calls with
312 // kAddRows calls wherever possible, while not increasing the number of calls.
313 //
314 // Algorithm :
315 // The function computes a histogram of submat_indexes and spans through the
316 // submat_indexes in descending order of frequency. For each submat_index a
317 // decision is made to copy it using a kAddRows call or not.
318 // A kAddRow call is made for a submat_index if splitting it into a seperate
319 // list reduces the max_submat_list_size by one, i.e., reduces the number of
320 // remaining kAddRowsMulti calls.
321 // submat_indexes which cannot be assigned to kAddRow calls are rearranged into
322 // lists for kAddRowsMulti calls.
323 //
324 // Note : To decide splits we could have solved a combinatorial
325 // optimization problem where we find the best set of
326 // kAddRows + kAddRowsMulti calls;
327 // but given that both these calls have similar costs,
328 // and that the average number of elements in a submat_list is around 4,
329 // it does not make sense to
330 // choose a kAddRows call unless it is able to immediately reduce a
331 // kAddRowsMulti call. So we simplify the process and stay away from any
332 // complex search algorithms. We might implement a solution where a more
333 // elaborate search is done,if the length of the lists increases
334 // for newer NN architectures, as even minor savings in speed due to increased
335 // number of kAddRows calls can accumulate compensating for the additional calls
336 
337 void SplitLocations(
338  const std::vector<std::vector<std::pair<int32, int32> > > &submat_lists,
339  std::vector<std::vector<std::pair<int32, int32> > > *split_lists) {
340 
341  // a histogram of the submat_indexes in the submat_lists
342  // each occurence in a given submat_list is considered unique so we maintain
343  // a vector to count each occurrence separately.
344  // The i'th element in the vector corresponds to the count of
345  // the (i+1)'th occurrence of a submat_index in a submat_list
346  unordered_map<int32, std::vector<int32> > submat_histogram;
347 
348  int32 max_submat_list_size = 0;
349 
350  // initializing a vector of list of pairs to store the sorted submat_lists
351  std::vector<std::vector< std::pair<int32, int32> > >
352  sorted_submat_lists;
353  SortSubmatLists(submat_lists, &sorted_submat_lists, &max_submat_list_size);
354  ComputeSubmatIndexHistogram(sorted_submat_lists, &submat_histogram);
355  // the vector has same information as the submat_histogram, but it is
356  // suitable for sorting according to frequency. The first elements of pairs
357  // can be repeated, these correspond to different occurrences in the same list
358  std::vector<std::pair<int32, int32> > submat_histogram_vector;
359  // copy the key, occurence_counts from submat_histogram to a vector
360  unordered_map<int32, std::vector<int32> >::iterator hist_iter;
361  for (hist_iter = submat_histogram.begin();
362  hist_iter != submat_histogram.end();
363  ++hist_iter) {
364  for (int32 i = 0; i < (hist_iter->second).size(); i++) {
365  submat_histogram_vector.push_back(
366  std::make_pair(hist_iter->first, (hist_iter->second)[i]));
367  }
368  }
369  SplitLocationsUsingSubmatHistogram(max_submat_list_size, &sorted_submat_lists,
370  &submat_histogram_vector, split_lists);
371 }

373 /* If it is the case for some i >= 0 that all the .first elements of
374  "location_vector" are either i or -1, then output i to first_value and the
375  .second elements into "second_values", and return true. Otherwise return
376  false and the outputs are don't-cares. */
377 bool ConvertToIndexes(
378  const std::vector<std::pair<int32, int32> > &location_vector,
379  int32 *first_value,
380  std::vector<int32> *second_values) {
381  *first_value = -1;
382  second_values->clear();
383  second_values->reserve(location_vector.size());
384  std::vector<std::pair<int32, int32> >::const_iterator iter;
385  for (iter = location_vector.begin(); iter < location_vector.end(); ++iter) {
386  if (iter->first != -1) {
387  if (*first_value == -1)
388  *first_value = iter->first;
389  if (iter->first != *first_value)
390  return false;
391  second_values->push_back(iter->second);
392  } else {
393  second_values->push_back(-1);
394  }
395  }
396  return true;
397 }

400 // see declaration in header for documentation
401 void EnsureContiguousProperty(
402  const std::vector<int32> &indexes,
403  std::vector<std::vector<int32> > *indexes_out) {
404  indexes_out->clear();
405  indexes_out->reserve(3);
406  if (indexes.empty()) return;
407  int32 max_value = *std::max_element(indexes.begin(), indexes.end());
408  if (max_value == -1) return;
409  std::vector<int32> num_segments_seen(max_value + 1, 0);
410  int32 dim = indexes.size(), num_output_vectors = 0;
411  for (int32 i = 0; i < dim;) {
412  // note, we increment i within the loop.
413  if (indexes[i] == -1) {
414  i++;
415  continue;
416  }
417  int32 value = indexes[i], start_index = i;
418  for (; i < dim && indexes[i] == value; i++);
419  int32 end_index = i; // one past the end.
420  // the input 'indexes' contains a sequence of possibly-repeated instances of
421  // the value 'value', starting at index 'start_index', with 'end_index' as
422  // one past the end.
423  int32 this_num_segments_seen = num_segments_seen[value]++;
424  if (this_num_segments_seen >= num_output_vectors) { // we have nowhere to
425  // put it.
426  indexes_out->resize(++num_output_vectors);
427  indexes_out->back().resize(dim, -1); // fill newly added vector with -1's.
428  }
429  std::vector<int32> &this_out_vec((*indexes_out)[this_num_segments_seen]);
430  std::vector<int32>::iterator iter = this_out_vec.begin() + start_index,
431  end = this_out_vec.begin() + end_index;
432  // Fill the appropriate range of the output vector with 'value'
433  for (; iter != end; ++iter) *iter = value;
434  }
435 }

467 void SplitPairList(std::vector<std::pair<int32, int32> >& list,
468  std::vector<std::vector<std::pair<int32, int32> > >* split_lists) {
469  split_lists->clear();
470  typedef unordered_map<std::pair<int32, int32>,
471  int32, PairHasher<int32> > MapType;
472  // this maps a pair not equal to -1,-1, to the number of times we've already seen it.
473  MapType pair_to_count;
474  int32 cur_num_lists = 0;
475 
476  for (int32 i = 0; i < list.size(); i++) {
477  if (list[i].first == -1)
478  continue;
479  MapType::iterator iter = pair_to_count.find(list[i]);
480  int32 this_count;
481  if (iter == pair_to_count.end())
482  pair_to_count[list[i]] = this_count = 1;
483  else
484  this_count = (++iter->second);
485  if (this_count > cur_num_lists) {
486  KALDI_ASSERT(this_count == cur_num_lists + 1);
487  split_lists->resize(this_count);
488  split_lists->back().resize(list.size(),
489  std::pair<int32, int32>(-1, -1));
490  cur_num_lists++;
491  }
492  (*split_lists)[this_count-1][i] = list[i];
493  }
494  if (split_lists->size() == 0)
495  KALDI_ERR << "Input list has just dummy pairs";
496 }

498 void SplitLocationsBackward(
499  const std::vector<std::vector<std::pair<int32, int32> > > &submat_lists,
500  std::vector<std::vector<std::pair<int32, int32> > > *split_lists) {
501  std::vector<std::vector<std::pair<int32, int32> > > split_lists_intermediate;
502  // Split the submat_lists
503  SplitLocations(submat_lists, &split_lists_intermediate);
504  for (size_t i = 0; i < split_lists_intermediate.size(); i++) {
505  int32 first_value;
506  std::vector<int32> second_values;
507  if (ConvertToIndexes(split_lists_intermediate[i],
508  &first_value, &second_values)) {
509  // the .first values in split_lists_intermediate[i] are all the same (or
510  // equal to -1).
511  if (first_value == -1) {
512  // all the .first values were equal to -1. this is like a NULL marker.
513  continue;
514  }
515  std::vector<std::vector<int32> > second_values_split;
516  EnsureContiguousProperty(second_values, &second_values_split);
517  if (second_values_split.size() == 1) {
518  // this branch is an optimization for speed.
519  split_lists->push_back(split_lists_intermediate[i]);
520  } else {
521  for (size_t j = 0; j < second_values_split.size(); j++) {
522  split_lists->resize(split_lists->size() + 1);
523  const std::vector<int32> &input_list = second_values_split[j];
524  std::vector<std::pair<int32, int32> > &output_list =
525  split_lists->back();
526  output_list.resize(input_list.size());
527  int32 size = input_list.size();
528  for (int32 k = 0; k < size; k++) {
529  int32 row = input_list[k];
530  if (row == -1) output_list[k].first = -1;
531  else output_list[k].first = first_value;
532  output_list[k].second = row;
533  }
534  }
535  }
536  } else {
537  // the .first values are not the same
538  // splitting the list of pairs to ensure unique pairs, unless it is
539  // (-1,-1)
540  std::vector<std::vector<std::pair<int32, int32> > > new_split_lists;
541  SplitPairList(split_lists_intermediate[i],
542  &new_split_lists);
543  for (int32 j = 0; j < new_split_lists.size(); j++) {
544  split_lists->push_back(new_split_lists[j]);
545  }
546  }
547  }
548 }

550 // This function returns true if for each integer i != -1, all the indexes j at
551 // which indexes[j] == i are consecutive with no gaps (more formally: if j1 < j2
552 // < j3 and indexes[j1] == indexes[j3], then indexes[j1] == indexes[j2]). If
553 // so, it also outputs to "reverse_indexes" the begin and end of these ranges,
554 // so that indexes[j] == i for all j such that (*reverse_indexes)[i].first <= j
555 // && j < (*reverse_indexes)[i].second.
556 bool HasContiguousProperty(
557  const std::vector<int32> &indexes,
558  std::vector<std::pair<int32, int32> > *reverse_indexes) {
559  reverse_indexes->clear();
560  int32 num_indexes = indexes.size();
561  if (num_indexes == 0)
562  return true;
563  int32 num_input_indexes =
564  *std::max_element(indexes.begin(), indexes.end()) + 1;
565  KALDI_ASSERT(num_input_indexes >= 0);
566  if (num_input_indexes == 0) {
567  // we don't really expect this input, filled with -1's.
568  KALDI_WARN << "HasContiguousProperty called on vector of -1's.";
569  return true;
570  }
571  reverse_indexes->resize(num_input_indexes,
572  std::pair<int32,int32>(-1, -1));
573  // set each pair's "first" to the min index of all elements
574  // of "indexes" with that value, and the "second" to the
575  // max plus one.
576  for (int32 i = 0; i < num_indexes; i++) {
577  int32 j = indexes[i];
578  if (j == -1) continue;
579  KALDI_ASSERT(j >= 0);
580  std::pair<int32, int32> &pair = (*reverse_indexes)[j];
581  if (pair.first == -1) {
582  pair.first = i;
583  pair.second = i + 1;
584  } else {
585  pair.first = std::min(pair.first, i);
586  pair.second = std::max(pair.second, i + 1);
587  }
588  }
589  // check that the contiguous property holds.
590  for (int32 i = 0; i < num_input_indexes; i++) {
591  std::pair<int32, int32> pair = (*reverse_indexes)[i];
592  if (pair.first != -1) {
593  for (int32 j = pair.first; j < pair.second; j++)
594  if (indexes[j] != i)
595  return false;
596  }
597  }
598  return true;
599 }
