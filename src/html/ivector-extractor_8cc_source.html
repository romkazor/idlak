43 // This function basically inverts the input and puts it in the output, but it's
44 // smart numerically. It uses the prior knowledge that the "inverse_floor" can
45 // have no eigenvalues less than one, so it applies that floor (in double
46 // precision) before inverting. This avoids certain numerical problems that can
47 // otherwise occur.
48 // static
49 void IvectorExtractor::InvertWithFlooring(const SpMatrix<double> &inverse_var,
50  SpMatrix<double> *var) {
51  SpMatrix<double> dbl_var(inverse_var);
52  int32 dim = inverse_var.NumRows();
53  Vector<double> s(dim);
54  Matrix<double> P(dim, dim);
55  // Solve the symmetric eigenvalue problem, inverse_var = P diag(s) P^T.
56  inverse_var.Eig(&s, &P);
57  s.ApplyFloor(1.0);
58  s.InvertElements();
59  var->AddMat2Vec(1.0, P, kNoTrans, s, 0.0);
60 }

63 void IvectorExtractor::GetIvectorDistribution(
64  const IvectorExtractorUtteranceStats &utt_stats,
65  VectorBase<double> *mean,
66  SpMatrix<double> *var) const {
67  if (!IvectorDependentWeights()) {
68  Vector<double> linear(IvectorDim());
69  SpMatrix<double> quadratic(IvectorDim());
70  GetIvectorDistMean(utt_stats, &linear, &quadratic);
71  GetIvectorDistPrior(utt_stats, &linear, &quadratic);
72  if (var != NULL) {
73  var->CopyFromSp(quadratic);
74  var->Invert(); // now it's a variance.
75 
76  // mean of distribution = quadratic^{-1} * linear...
77  mean->AddSpVec(1.0, *var, linear, 0.0);
78  } else {
79  quadratic.Invert();
80  mean->AddSpVec(1.0, quadratic, linear, 0.0);
81  }
82  } else {
83  Vector<double> linear(IvectorDim());
84  SpMatrix<double> quadratic(IvectorDim());
85  GetIvectorDistMean(utt_stats, &linear, &quadratic);
86  GetIvectorDistPrior(utt_stats, &linear, &quadratic);
87  // At this point, "linear" and "quadratic" contain
88  // the mean and prior-related terms, and we avoid
89  // recomputing those.
90 
91  Vector<double> cur_mean(IvectorDim());
92 
93  SpMatrix<double> quadratic_inv(IvectorDim());
94  InvertWithFlooring(quadratic, &quadratic_inv);
95  cur_mean.AddSpVec(1.0, quadratic_inv, linear, 0.0);
96 
97  KALDI_VLOG(3) << "Trace of quadratic is " << quadratic.Trace()
98  << ", condition is " << quadratic.Cond();
99  KALDI_VLOG(3) << "Trace of quadratic_inv is " << quadratic_inv.Trace()
100  << ", condition is " << quadratic_inv.Cond();
101 
102  // The loop is finding successively better approximation points
103  // for the quadratic expansion of the weights.
104  int32 num_iters = 4;
105  double change_threshold = 0.1; // If the iVector changes by less than
106  // this (in 2-norm), we abort early.
107  for (int32 iter = 0; iter < num_iters; iter++) {
108  if (GetVerboseLevel() >= 3) {
109  KALDI_VLOG(3) << "Auxf on iter " << iter << " is "
110  << GetAuxf(utt_stats, cur_mean, &quadratic_inv);
111  int32 show_dim = 5;
112  if (show_dim > cur_mean.Dim()) show_dim = cur_mean.Dim();
113  KALDI_VLOG(3) << "Current distribution mean is "
114  << cur_mean.Range(0, show_dim) << "... "
115  << ", var trace is " << quadratic_inv.Trace();
116  }
117  Vector<double> this_linear(linear);
118  SpMatrix<double> this_quadratic(quadratic);
119  GetIvectorDistWeight(utt_stats, cur_mean,
120  &this_linear, &this_quadratic);
121  InvertWithFlooring(this_quadratic, &quadratic_inv);
122  Vector<double> mean_diff(cur_mean);
123  cur_mean.AddSpVec(1.0, quadratic_inv, this_linear, 0.0);
124  mean_diff.AddVec(-1.0, cur_mean);
125  double change = mean_diff.Norm(2.0);
126  KALDI_VLOG(2) << "On iter " << iter << ", iVector changed by " << change;
127  if (change < change_threshold)
128  break;
129  }
130  mean->CopyFromVec(cur_mean);
131  if (var != NULL)
132  var->CopyFromSp(quadratic_inv);
133  }
134 }

137 IvectorExtractor::IvectorExtractor(
138  const IvectorExtractorOptions &opts,
139  const FullGmm &fgmm) {
140  KALDI_ASSERT(opts.ivector_dim > 0);
141  Sigma_inv_.resize(fgmm.NumGauss());
142  for (int32 i = 0; i < fgmm.NumGauss(); i++) {
143  const SpMatrix<BaseFloat> &inv_var = fgmm.inv_covars()[i];
144  Sigma_inv_[i].Resize(inv_var.NumRows());
145  Sigma_inv_[i].CopyFromSp(inv_var);
146  }
147  Matrix<double> gmm_means;
148  fgmm.GetMeans(&gmm_means);
149  KALDI_ASSERT(!Sigma_inv_.empty());
150  int32 feature_dim = Sigma_inv_[0].NumRows(),
151  num_gauss = Sigma_inv_.size();
152 
153  prior_offset_ = 100.0; // hardwired for now. Must be nonzero.
154  gmm_means.Scale(1.0 / prior_offset_);
155 
156  M_.resize(num_gauss);
157  for (int32 i = 0; i < num_gauss; i++) {
158  M_[i].Resize(feature_dim, opts.ivector_dim);
159  M_[i].SetRandn();
160  M_[i].CopyColFromVec(gmm_means.Row(i), 0);
161  }
162  if (opts.use_weights) { // will regress the log-weights on the iVector.
163  w_.Resize(num_gauss, opts.ivector_dim);
164  } else {
165  w_vec_.Resize(fgmm.NumGauss());
166  w_vec_.CopyFromVec(fgmm.weights());
167  }
168  ComputeDerivedVars();
169 }

182 void IvectorExtractor::ComputeDerivedVars() {
183  KALDI_LOG << "Computing derived variables for iVector extractor";
184  gconsts_.Resize(NumGauss());
185  for (int32 i = 0; i < NumGauss(); i++) {
186  double var_logdet = -Sigma_inv_[i].LogPosDefDet();
187  gconsts_(i) = -0.5 * (var_logdet + FeatDim() * M_LOG_2PI);
188  // the gconsts don't contain any weight-related terms.
189  }
190  U_.Resize(NumGauss(), IvectorDim() * (IvectorDim() + 1) / 2);
191  Sigma_inv_M_.resize(NumGauss());
192 
193  // Note, we could have used RunMultiThreaded for this and similar tasks we
194  // have here, but we found that we don't get as complete CPU utilization as we
195  // could because some tasks finish before others.
196  {
197  TaskSequencerConfig sequencer_opts;
198  sequencer_opts.num_threads = g_num_threads;
199  TaskSequencer<IvectorExtractorComputeDerivedVarsClass> sequencer(
200  sequencer_opts);
201  for (int32 i = 0; i < NumGauss(); i++)
202  sequencer.Run(new IvectorExtractorComputeDerivedVarsClass(this, i));
203  }
204  KALDI_LOG << "Done.";
205 }

221 void IvectorExtractor::GetIvectorDistWeight(
222  const IvectorExtractorUtteranceStats &utt_stats,
223  const VectorBase<double> &mean,
224  VectorBase<double> *linear,
225  SpMatrix<double> *quadratic) const {
226  // If there is no w_, then weights do not depend on the iVector
227  // and the weights contribute nothing to the distribution.
228  if (!IvectorDependentWeights())
229  return;
230 
231  Vector<double> logw_unnorm(NumGauss());
232  logw_unnorm.AddMatVec(1.0, w_, kNoTrans, mean, 0.0);
233 
234  Vector<double> w(logw_unnorm);
235  w.ApplySoftMax(); // now w is the weights.
236 
237  // See eq.58 in SGMM paper
238  // http://www.sciencedirect.com/science/article/pii/S088523081000063X
239  // linear_coeff(i) = \gamma_{jmi} - \gamma_{jm} \hat{w}_{jmi} + \max(\gamma_{jmi}, \gamma_{jm} \hat{w}_{jmi} \hat{\w}_i \v_{jm}
240  // here \v_{jm} corresponds to the iVector. Ignore the j,m indices.
241  Vector<double> linear_coeff(NumGauss());
242  Vector<double> quadratic_coeff(NumGauss());
243  double gamma = utt_stats.gamma_.Sum();
244  for (int32 i = 0; i < NumGauss(); i++) {
245  double gamma_i = utt_stats.gamma_(i);
246  double max_term = std::max(gamma_i, gamma * w(i));
247  linear_coeff(i) = gamma_i - gamma * w(i) + max_term * logw_unnorm(i);
248  quadratic_coeff(i) = max_term;
249  }
250  linear->AddMatVec(1.0, w_, kTrans, linear_coeff, 1.0);
251 
252  // *quadratic += \sum_i quadratic_coeff(i) w_i w_i^T, where w_i is
253  // i'th row of w_.
254  quadratic->AddMat2Vec(1.0, w_, kTrans, quadratic_coeff, 1.0);
255 }

257 void IvectorExtractor::GetIvectorDistMean(
258  const IvectorExtractorUtteranceStats &utt_stats,
259  VectorBase<double> *linear,
260  SpMatrix<double> *quadratic) const {
261  int32 I = NumGauss();
262  for (int32 i = 0; i < I; i++) {
263  double gamma = utt_stats.gamma_(i);
264  if (gamma != 0.0) {
265  SubVector<double> x(utt_stats.X_, i); // == \gamma(i) \m_i
266  // next line: a += \gamma_i \M_i^T \Sigma_i^{-1} \m_i
267  linear->AddMatVec(1.0, Sigma_inv_M_[i], kTrans, x, 1.0);
268  }
269  }
270  SubVector<double> q_vec(quadratic->Data(), IvectorDim()*(IvectorDim()+1)/2);
271  q_vec.AddMatVec(1.0, U_, kTrans, utt_stats.gamma_, 1.0);
272 }

285 double IvectorExtractor::GetAcousticAuxfWeight(
286  const IvectorExtractorUtteranceStats &utt_stats,
287  const VectorBase<double> &mean,
288  const SpMatrix<double> *var) const {
289  if (!IvectorDependentWeights()) { // Not using the weight-projection matrices.
290  Vector<double> log_w_vec(w_vec_);
291  log_w_vec.ApplyLog();
292  return VecVec(log_w_vec, utt_stats.gamma_);
293  } else {
294  Vector<double> w(NumGauss());
295  w.AddMatVec(1.0, w_, kNoTrans, mean, 0.0); // now w is unnormalized
296  // log-weights.
297 
298  double lse = w.LogSumExp();
299  w.Add(-lse); // Normalize so log-weights sum to one.
300 
301  // "ans" below is the point-value of the weight auxf, without
302  // considering the variance. At the moment, "w" contains
303  // the normalized log weights.
304  double ans = VecVec(w, utt_stats.gamma_);
305 
306  w.ApplyExp(); // now w is the weights.
307 
308  if (var == NULL) {
309  return ans;
310  } else {
311  // Below, "Jacobian" will be the derivative d(log_w) / d(ivector)
312  // = (I - w w^T) W, where W (w_ in the code) is the projection matrix
313  // from iVector space to unnormalized log-weights, and w is the normalized
314  // weight values at the current point.
315  Matrix<double> Jacobian(w_);
316  Vector<double> WTw(IvectorDim()); // W^T w
317  WTw.AddMatVec(1.0, w_, kTrans, w, 0.0);
318  Jacobian.AddVecVec(1.0, w, WTw); // Jacobian += (w (W^T w)^T = w^T w W)
319 
320  // the matrix S is the negated 2nd derivative of the objf w.r.t. the iVector \x.
321  SpMatrix<double> S(IvectorDim());
322  S.AddMat2Vec(1.0, Jacobian, kTrans, Vector<double>(utt_stats.gamma_), 0.0);
323  ans += -0.5 * TraceSpSp(S, *var);
324  return ans;
325  }
326  }
327 }

343 // gets logdet of a matrix while suppressing exceptions; always returns finite
344 // value even if there was a problem.
345 static double GetLogDetNoFailure(const SpMatrix<double> &var) {
346  try {
347  return var.LogPosDefDet();
348  } catch (...) {
349  Vector<double> eigs(var.NumRows());
350  var.Eig(&eigs);
351  int32 floored;
352  eigs.ApplyFloor(1.0e-20, &floored);
353  if (floored > 0)
354  KALDI_WARN << "Floored " << floored << " eigenvalues of variance.";
355  eigs.ApplyLog();
356  return eigs.Sum();
357  }

360 /*
361  Get the prior-related part of the auxiliary function. Suppose
362  the ivector is x, the prior distribution is p(x), the likelihood
363  of the data given x (itself an auxiliary function) is q(x) and
364  the prior is r(x).
365 
366  In the case where we just have a point ivector x and no p(x),
367  the prior-related term we return will just be q(x).
368 
369  If we have a distribution over x, we define an auxiliary function
370  t(x) = \int p(x) log(q(x)r(x) / p(x)) dx.
371  We separate this into data-dependent and prior parts, where the prior
372  part is
373  \int p(x) log(q(x) / p(x)) dx
374  (which this function returns), and the acoustic part is
375  \int p(x) log(r(x)) dx.
376  Note that the fact that we divide by p(x) in the prior part and not
377  the acoustic part is a bit arbitrary, at least the way we have written
378  it down here; it doesn't matter where we do it, but this way seems
379  more natural.
380 */
381 double IvectorExtractor::GetPriorAuxf(
382  const VectorBase<double> &mean,
383  const SpMatrix<double> *var) const {
384  KALDI_ASSERT(mean.Dim() == IvectorDim());
385 
386  Vector<double> offset(mean);
387  offset(0) -= prior_offset_; // The mean of the prior distribution
388  // may only be nonzero in the first dimension. Now, "offset" is the
389  // offset of ivector from the prior's mean.
390 
391 
392  if (var == NULL) {
393  // The log-determinant of the variance of the prior distribution is one,
394  // since it's the unit matrix.
395  return -0.5 * (VecVec(offset, offset) + IvectorDim()*M_LOG_2PI);
396  } else {
397  // The mean-related part of the answer will be
398  // -0.5 * (VecVec(offset, offset), just like above.
399  // The variance-related part will be
400  // \int p(x) . -0.5 (x^T I x - x^T var^{-1} x + logdet(I) - logdet(var)) dx
401  // and using the fact that x is distributed with variance "var", this is:
402  //= \int p(x) . -0.5 (x^T I x - x^T var^{-1} x + logdet(I) - logdet(var)) dx
403  // = -0.5 ( trace(var I) - trace(var^{-1} var) + 0.0 - logdet(var))
404  // = -0.5 ( trace(var) - dim(var) - logdet(var))
405 
406  KALDI_ASSERT(var->NumRows() == IvectorDim());
407  return -0.5 * (VecVec(offset, offset) + var->Trace() -
408  IvectorDim() - GetLogDetNoFailure(*var));
409  }
410 }

413  If the distribution over the ivector given by "mean" and
414  "var" is p(x), and the acoustic auxiliary-function given
415  x is r(x), this function returns
416  \int p(x) log r(x) dx

418 */
419 double IvectorExtractor::GetAcousticAuxf(
420  const IvectorExtractorUtteranceStats &utt_stats,
421  const VectorBase<double> &mean,

423  double weight_auxf = GetAcousticAuxfWeight(utt_stats, mean, var),
424  gconst_auxf = GetAcousticAuxfGconst(utt_stats),
425  mean_auxf = GetAcousticAuxfMean(utt_stats, mean, var),

429  << (gconst_auxf/T) << ", mean " << (mean_auxf/T) << ", var "

431  return weight_auxf + gconst_auxf + mean_auxf + var_auxf;

434 /* This is the part of the auxf that involves the data
435  means, and we also involve the gconsts here.
436  Let \m_i be the observed data mean vector for the
437  i'th Gaussian (which we get from the utterance-specific
438  stats). Let \mu_i(\x) be the mean of the i'th Gaussian,
439  written as a function of the iVector \x.
440 
441  \int_\x p(\x) (\sum_i \gamma_i -0.5 (\mu(\x) - \m_i)^T \Sigma_i^{-1} (\mu(\x) - \m_i)) d\x
442 
443  To compute this integral we'll first write out the summation as a function of \x.
444 
445  \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
446  + \gamma_i \mu(\x)^T \Sigma_i^{-1} \m_i
447  -0.5 \gamma_i \mu(\x)^T \Sigma_i^{-1} \m_i
448  = \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
449  + \gamma_i \x^T \M_i^T \Sigma_i^{-1} \m_i
450  -0.5 \gamma_i \x^T \M_i^T \Sigma_i^{-1} \x
451  = K + \x^T \a - 0.5 \x^T \B \x,
452  where K = \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
453  \a = \sum_i \gamma_i \M_i^T \Sigma_i^{-1} \m_i
454  \B = \sum_i \gamma_i \U_i (where \U_i = \M_i^T \Sigma_i^{-1} \M_i
455  has been stored previously).
456  Note: the matrix M in the stats actually contains \gamma_i \m_i,
457  i.e. the mean times the count, so we need to modify the definitions
458  above accordingly.
459 */
460 double IvectorExtractor::GetAcousticAuxfMean(
461  const IvectorExtractorUtteranceStats &utt_stats,
462  const VectorBase<double> &mean,
463  const SpMatrix<double> *var) const {
464  double K = 0.0;
465  Vector<double> a(IvectorDim()), temp(FeatDim());
466 
467  int32 I = NumGauss();
468  for (int32 i = 0; i < I; i++) {
469  double gamma = utt_stats.gamma_(i);
470  if (gamma != 0.0) {
471  Vector<double> x(utt_stats.X_.Row(i)); // == \gamma(i) \m_i
472  temp.AddSpVec(1.0 / gamma, Sigma_inv_[i], x, 0.0);
473  // now temp = Sigma_i^{-1} \m_i.
474  // next line: K += -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
475  K += -0.5 * VecVec(x, temp);
476  // next line: a += \gamma_i \M_i^T \Sigma_i^{-1} \m_i
477  a.AddMatVec(gamma, M_[i], kTrans, temp, 1.0);
478  }
479  }
480  SpMatrix<double> B(IvectorDim());
481  SubVector<double> B_vec(B.Data(), IvectorDim()*(IvectorDim()+1)/2);
482  B_vec.AddMatVec(1.0, U_, kTrans, Vector<double>(utt_stats.gamma_), 0.0);
483 
484  double ans = K + VecVec(mean, a) - 0.5 * VecSpVec(mean, B, mean);
485  if (var != NULL)
486  ans -= 0.5 * TraceSpSp(*var, B);
487  return ans;
488 }

497 double IvectorExtractor::GetAcousticAuxfVariance(
498  const IvectorExtractorUtteranceStats &utt_stats) const {
499  if (utt_stats.S_.empty()) {
500  // we did not store the variance, so assume it's as predicted
501  // by the model itself.
502  // for each Gaussian i, we have a term -0.5 * gamma(i) * trace(Sigma[i] * Sigma[i]^{-1})
503  // = -0.5 * gamma(i) * FeatDim().
504  return -0.5 * utt_stats.gamma_.Sum() * FeatDim();
505  } else {
506  int32 I = NumGauss();
507  double ans = 0.0;
508  for (int32 i = 0; i < I; i++) {
509  double gamma = utt_stats.gamma_(i);
510  if (gamma != 0.0) {
511  SpMatrix<double> var(utt_stats.S_[i]);
512  var.Scale(1.0 / gamma);
513  Vector<double> mean(utt_stats.X_.Row(i));
514  mean.Scale(1.0 / gamma);
515  var.AddVec2(-1.0, mean); // get centered covariance..
516  ans += -0.5 * gamma * TraceSpSp(var, Sigma_inv_[i]);
517  }
518  }
519  return ans;
520  }
521 }

523 void IvectorExtractor::TransformIvectors(const MatrixBase<double> &T,
524  double new_prior_offset) {
525  Matrix<double> Tinv(T);
526  Tinv.Invert();
527  // w <-- w Tinv. (construct temporary copy with Matrix<double>(w))
528  if (IvectorDependentWeights())
529  w_.AddMatMat(1.0, Matrix<double>(w_), kNoTrans, Tinv, kNoTrans, 0.0);
530  // next: M_i <-- M_i Tinv. (construct temporary copy with Matrix<double>(M_[i]))
531  for (int32 i = 0; i < NumGauss(); i++)
532  M_[i].AddMatMat(1.0, Matrix<double>(M_[i]), kNoTrans, Tinv, kNoTrans, 0.0);
533  KALDI_LOG << "Setting iVector prior offset to " << new_prior_offset;
534  prior_offset_ = new_prior_offset;

537 void OnlineIvectorEstimationStats::AccStats(
538  const IvectorExtractor &extractor,
539  const VectorBase<BaseFloat> &feature,
540  const std::vector<std::pair<int32, BaseFloat> > &gauss_post) {
541  KALDI_ASSERT(extractor.IvectorDim() == this->IvectorDim());
542  KALDI_ASSERT(!extractor.IvectorDependentWeights());
543 
544  Vector<double> feature_dbl(feature);
545  double tot_weight = 0.0;
546  int32 ivector_dim = this->IvectorDim(),
547  quadratic_term_dim = (ivector_dim * (ivector_dim + 1)) / 2;
548  SubVector<double> quadratic_term_vec(quadratic_term_.Data(),
549  quadratic_term_dim);
550 
551  for (size_t idx = 0; idx < gauss_post.size(); idx++) {
552  int32 g = gauss_post[idx].first;
553  double weight = gauss_post[idx].second;
554  // allow negative weights; it's needed in the online iVector extraction
555  // with speech-silence detection based on decoder traceback (we subtract
556  // stuff we previously added if the traceback changes).
557  if (weight == 0.0)
558  continue;
559  linear_term_.AddMatVec(weight, extractor.Sigma_inv_M_[g], kTrans,
560  feature_dbl, 1.0);
561  SubVector<double> U_g(extractor.U_, g);
562  quadratic_term_vec.AddVec(weight, U_g);
563  tot_weight += weight;
564  }
565  if (max_count_ > 0.0) {
566  // see comments in header RE max_count for explanation. It relates to
567  // prior scaling when the count exceeds max_count_
568  double old_num_frames = num_frames_,
569  new_num_frames = num_frames_ + tot_weight;
570  double old_prior_scale = std::max(old_num_frames, max_count_) / max_count_,
571  new_prior_scale = std::max(new_num_frames, max_count_) / max_count_;
572  // The prior_scales are the inverses of the scales we would put on the stats
573  // if we were implementing this by scaling the stats. Instead we
574  // scale the prior term.
575  double prior_scale_change = new_prior_scale - old_prior_scale;
576  if (prior_scale_change != 0.0) {
577  linear_term_(0) += prior_offset_ * prior_scale_change;
578  quadratic_term_.AddToDiag(prior_scale_change);
579  }
580  }
581 
582  num_frames_ += tot_weight;
583 }

585 void OnlineIvectorEstimationStats::Scale(double scale) {
586  KALDI_ASSERT(scale >= 0.0 && scale <= 1.0);
587  double old_num_frames = num_frames_;
588  num_frames_ *= scale;
589  quadratic_term_.Scale(scale);
590  linear_term_.Scale(scale);
591 
592  // Scale back up the prior term, by adding in whatever we scaled down.
593  if (max_count_ == 0.0) {
594  linear_term_(0) += prior_offset_ * (1.0 - scale);
595  quadratic_term_.AddToDiag(1.0 - scale);
596  } else {
597  double new_num_frames = num_frames_;
598  double old_prior_scale =
599  scale * std::max(old_num_frames, max_count_) / max_count_,
600  new_prior_scale = std::max(new_num_frames, max_count_) / max_count_;
601  // old_prior_scale is the scale the prior term currently has in the stats,
602  // i.e. the previous scale times "scale" as we just scaled the stats.
603  // new_prior_scale is the scale we want the prior term to have.
604  linear_term_(0) += prior_offset_ * (new_prior_scale - old_prior_scale);
605  quadratic_term_.AddToDiag(new_prior_scale - old_prior_scale);
606  }
607 }

878  R_cache_lock_.lock();

880  KALDI_VLOG(1) << "Flushing cache for IvectorExtractorStats";
881  // Store these quantities as copies in memory so other threads can use the
882  // cache while we update R_ from the cache.

884  R_gamma_cache_.Range(0, R_num_cached_,

888  0, R_ivec_scatter_cache_.NumCols()));
889  R_num_cached_ = 0; // As far as other threads are concerned, the cache is
890  // cleared and they may write to it.
891  R_cache_lock_.unlock();
892  R_lock_.lock();

896  } else {

915 // This function commits stats for a single sample of the ivector,
916 // to update the weight projection w_.
917 void IvectorExtractorStats::CommitStatsForWPoint(
918  const IvectorExtractor &extractor,
919  const IvectorExtractorUtteranceStats &utt_stats,
920  const VectorBase<double> &ivector,
921  double weight) {
922  int32 num_gauss = extractor.NumGauss();
923  // Compare this function with GetIvectorDistWeight(), from which it
924  // was derived.
925  Vector<double> logw_unnorm(num_gauss);
926  logw_unnorm.AddMatVec(1.0, extractor.w_, kNoTrans, ivector, 0.0);
927 
928  Vector<double> w(logw_unnorm);
929  w.ApplySoftMax(); // now w is the weights.
930 
931  Vector<double> linear_coeff(num_gauss);
932  Vector<double> quadratic_coeff(num_gauss);
933  double gamma = utt_stats.gamma_.Sum();
934  for (int32 i = 0; i < num_gauss; i++) {
935  double gamma_i = utt_stats.gamma_(i);
936  double max_term = std::max(gamma_i, gamma * w(i));
937  linear_coeff(i) = gamma_i - gamma * w(i) + max_term * logw_unnorm(i);
938  quadratic_coeff(i) = max_term;
939  }
940  weight_stats_lock_.lock();
941  G_.AddVecVec(weight, linear_coeff, Vector<double>(ivector));
942 
943  int32 ivector_dim = extractor.IvectorDim();
944  SpMatrix<double> outer_prod(ivector_dim);
945  outer_prod.AddVec2(1.0, ivector);
946  SubVector<double> outer_prod_vec(outer_prod.Data(),
947  ivector_dim * (ivector_dim + 1) / 2);
948  Q_.AddVecVec(weight, quadratic_coeff, outer_prod_vec);
949  weight_stats_lock_.unlock();
950 }

952 void IvectorExtractorStats::CommitStatsForW(
953  const IvectorExtractor &extractor,
954  const IvectorExtractorUtteranceStats &utt_stats,
955  const VectorBase<double> &ivec_mean,
956  const SpMatrix<double> &ivec_var) {
957  KALDI_ASSERT(config_.num_samples_for_weights > 1);
958 
959  Matrix<double> rand(config_.num_samples_for_weights, extractor.IvectorDim());
960  rand.SetRandn();
961  TpMatrix<double> ivec_stddev(extractor.IvectorDim());
962  ivec_stddev.Cholesky(ivec_var);
963  Matrix<double> ivecs(config_.num_samples_for_weights, extractor.IvectorDim());
964  ivecs.AddMatTp(1.0, rand, kNoTrans, ivec_stddev, kTrans, 0.0);
965  // Now make the ivecs zero-mean
966  Vector<double> avg_ivec(extractor.IvectorDim());
967  avg_ivec.AddRowSumMat(1.0 / config_.num_samples_for_weights, ivecs);
968  ivecs.AddVecToRows(-1.0, avg_ivec);
969  // Correct the variance for what we just did, so the expected
970  // variance still has the correct value.
971  ivecs.Scale(sqrt(config_.num_samples_for_weights / (config_.num_samples_for_weights - 1.0)));
972  // Add the mean of the distribution to "ivecs".
973  ivecs.AddVecToRows(1.0, ivec_mean);
974  // "ivecs" is now a sample from the iVector distribution.
975  for (int32 samp = 0; samp < config_.num_samples_for_weights; samp++)
976  CommitStatsForWPoint(extractor, utt_stats,
977  ivecs.Row(samp),
978  1.0 / config_.num_samples_for_weights);
979 }

1233 double IvectorExtractorStats::UpdateProjection(
1234  const IvectorExtractorEstimationOptions &opts,
1235  int32 i,
1236  IvectorExtractor *extractor) const {
1237  int32 I = extractor->NumGauss(), S = extractor->IvectorDim();
1238  KALDI_ASSERT(i >= 0 && i < I);
1239  /*
1240  For Gaussian index i, maximize the auxiliary function
1241  Q_i(x) = tr(M_i^T Sigma_i^{-1} Y_i) - 0.5 tr(Sigma_i^{-1} M_i R_i M_i^T)
1242  */
1243  if (gamma_(i) < opts.gaussian_min_count) {
1244  KALDI_WARN << "Skipping Gaussian index " << i << " because count "
1245  << gamma_(i) << " is below min-count.";
1246  return 0.0;
1247  }
1248  SpMatrix<double> R(S, kUndefined), SigmaInv(extractor->Sigma_inv_[i]);
1249  SubVector<double> R_vec(R_, i); // i'th row of R; vectorized form of SpMatrix.
1250  SubVector<double> R_sp(R.Data(), S * (S+1) / 2);
1251  R_sp.CopyFromVec(R_vec); // copy to SpMatrix's memory.
1252 
1253  Matrix<double> M(extractor->M_[i]);
1254  SolverOptions solver_opts;
1255  solver_opts.name = "M";
1256  solver_opts.diagonal_precondition = true;
1257  double impr = SolveQuadraticMatrixProblem(R, Y_[i], SigmaInv, solver_opts, &M),
1258  gamma = gamma_(i);
1259  if (i < 4) {
1260  KALDI_VLOG(1) << "Objf impr for M for Gaussian index " << i << " is "
1261  << (impr / gamma) << " per frame over " << gamma << " frames.";
1262  }
1263  extractor->M_[i].CopyFromMat(M);
1264  return impr;
1265 }

1268  const SubMatrix<double> &T,
1269  IvectorExtractor *extractor,
1270  Matrix<double> *A) const {

1277  // avg of these rows in a SubVector that updates the data of the SpMatrix

1284  uavg_vec.AddMatVec(1.0/num_gauss, extractor->U_, kTrans, w_uniform, 0.0);
1285  } else {
1286  uavg_vec.AddMatVec(1.0, extractor->U_, kTrans, extractor->w_vec_, 0.0);
1287  }
1288

1290  Tinv.Invert();

1292 
1293  Vavg_temp.AddMatMatMat(1.0, Tinv, kTrans, SubMatrix<double>(Uavg_temp,
1294  1, ivector_dim-1, 1, ivector_dim-1),
1295  kNoTrans, Tinv, kNoTrans, 0.0);
1296  Vavg.CopyFromMat(Vavg_temp);
1297 
1298  Vector<double> s(ivector_dim-1);
1299  Matrix<double> P(ivector_dim-1, ivector_dim-1);
1300  Vavg.Eig(&s, &P);
1301  SortSvd(&s, &P);
1302  A->Resize(P.NumCols(), P.NumRows());
1303  A->SetZero();
1304  A->AddMat(1.0, P, kTrans);
1305  KALDI_LOG << "Eigenvalues of Vavg: " << s;
1306 }

1351 double IvectorExtractorStats::UpdateVariances(
1352  const IvectorExtractorEstimationOptions &opts,
1353  IvectorExtractor *extractor) const {
1354  int32 num_gauss = extractor->NumGauss(),
1355  feat_dim = extractor->FeatDim(),
1356  ivector_dim = extractor->IvectorDim();
1357  KALDI_ASSERT(!S_.empty());
1358  double tot_objf_impr = 0.0;
1359 
1360  // "raw_variances" will be the variances directly from
1361  // the stats, without any flooring.
1362  std::vector<SpMatrix<double> > raw_variances(num_gauss);
1363  SpMatrix<double> var_floor(feat_dim);
1364  double var_floor_count = 0.0;
1365 
1366  for (int32 i = 0; i < num_gauss; i++) {
1367  if (gamma_(i) < opts.gaussian_min_count) continue; // warned in UpdateProjections
1368  SpMatrix<double> &S(raw_variances[i]);
1369  S = S_[i]; // Set it to the raw scatter statistics.
1370 
1371  // The equations for estimating the variance are similar to
1372  // those used in SGMMs. We need to convert it to a centered
1373  // covariance, and for this we can use a combination of other
1374  // stats and the model parameters.
1375 
1376  Matrix<double> M(extractor->M_[i]);
1377  // Y * M^T.
1378  Matrix<double> YM(feat_dim, feat_dim);
1379  YM.AddMatMat(1.0, Y_[i], kNoTrans, M, kTrans, 0.0);
1380  Matrix<double> YMMY(YM, kTrans);
1381  YMMY.AddMat(1.0, YM);
1382  // Now, YMMY = Y * M^T + M * Y^T. This is a kind of cross-term
1383  // between the mean and the data, which we subtract.
1384  SpMatrix<double> YMMY_sp(YMMY, kTakeMeanAndCheck);
1385  S.AddSp(-1.0, YMMY_sp);
1386 
1387  // Add in a mean-squared term.
1388  SpMatrix<double> R(ivector_dim); // will be scatter of iVectors, weighted
1389  // by count for this Gaussian.
1390  SubVector<double> R_vec(R.Data(),
1391  ivector_dim * (ivector_dim + 1) / 2);
1392  R_vec.CopyFromVec(R_.Row(i)); //
1393 
1394  S.AddMat2Sp(1.0, M, kNoTrans, R, 1.0);
1395 
1396  var_floor.AddSp(1.0, S);
1397  var_floor_count += gamma_(i);
1398  S.Scale(1.0 / gamma_(i));
1399  }
1400  KALDI_ASSERT(var_floor_count > 0.0);
1401  KALDI_ASSERT(opts.variance_floor_factor > 0.0 &&
1402  opts.variance_floor_factor <= 1.0);
1403 
1404  var_floor.Scale(opts.variance_floor_factor / var_floor_count);
1405 
1406  // var_floor should not be singular in any normal case, but previously
1407  // we've had situations where cholesky on it failed (perhaps due to
1408  // people using linearly dependent features). So we floor its
1409  // singular values.
1410  int eig_floored = var_floor.ApplyFloor(var_floor.MaxAbsEig() * 1.0e-04);
1411  if (eig_floored > 0) {
1412  KALDI_WARN << "Floored " << eig_floored << " eigenvalues of the "
1413  << "variance floor matrix. This is not expected. Maybe your "
1414  << "feature data is linearly dependent.";
1415  }
1416 
1417  int32 tot_num_floored = 0;
1418  for (int32 i = 0; i < num_gauss; i++) {
1419  SpMatrix<double> &S(raw_variances[i]); // un-floored variance.
1420  if (S.NumRows() == 0) continue; // due to low count.
1421  SpMatrix<double> floored_var(S);
1422  SpMatrix<double> old_inv_var(extractor->Sigma_inv_[i]);
1423 
1424  int32 num_floored = floored_var.ApplyFloor(var_floor);
1425  tot_num_floored += num_floored;
1426  if (num_floored > 0)
1427  KALDI_LOG << "For Gaussian index " << i << ", floored "
1428  << num_floored << " eigenvalues of variance.";
1429  // this objf is per frame;
1430  double old_objf = -0.5 * (TraceSpSp(S, old_inv_var) -
1431  old_inv_var.LogPosDefDet());
1432 
1433  SpMatrix<double> new_inv_var(floored_var);
1434  new_inv_var.Invert();
1435 
1436  double new_objf = -0.5 * (TraceSpSp(S, new_inv_var) -
1437  new_inv_var.LogPosDefDet());
1438  if (i < 4) {
1439  KALDI_VLOG(1) << "Objf impr/frame for variance for Gaussian index "
1440  << i << " was " << (new_objf - old_objf);
1441  }
1442  tot_objf_impr += gamma_(i) * (new_objf - old_objf);
1443  extractor->Sigma_inv_[i].CopyFromSp(new_inv_var);
1444  }
1445  double floored_percent = tot_num_floored * 100.0 / (num_gauss * feat_dim);
1446  KALDI_LOG << "Floored " << floored_percent << "% of all Gaussian eigenvalues";
1447 
1448  KALDI_LOG << "Overall objf impr/frame for variances was "
1449  << (tot_objf_impr / gamma_.Sum()) << " over "
1450  << gamma_.Sum() << " frames.";
1451  return tot_objf_impr / gamma_.Sum();
1452 }

1527 double IvectorExtractorStats::PriorDiagnostics(double old_prior_offset) const {
1528  // The iVectors had a centered covariance "covar"; we want to figure out
1529  // the objective-function change from rescaling. It's as if we were
1530  // formerly modeling "covar" with the unit matrix, and we're now modeling
1531  // it with "covar" itself. This is ignoring flooring issues. Of course,
1532  // we implement it through rescaling the space, but it has the same effect.
1533  // We also need to take into account that before the rescaling etc., the
1534  // old mean might have been wrong.
1535 
1536  int32 ivector_dim = ivector_sum_.Dim();
1537  Vector<double> sum(ivector_sum_);
1538  sum.Scale(1.0 / num_ivectors_);
1539  SpMatrix<double> covar(ivector_scatter_);
1540  covar.Scale(1.0 / num_ivectors_);
1541  covar.AddVec2(-1.0, sum); // Get the centered covariance.
1542 
1543  // Now work out the offset from the old prior's mean.
1544  Vector<double> mean_offset(sum);
1545  mean_offset(0) -= old_prior_offset;
1546 
1547  SpMatrix<double> old_covar(covar); // the covariance around the old mean.
1548  old_covar.AddVec2(1.0, mean_offset);
1549  // old likelihood = -0.5 * (Trace(I old_covar) + logdet(I) + [ignored])
1550  double old_like = -0.5 * old_covar.Trace();
1551  // new likelihood is if we updated the variance to equal "covar"... this isn't
1552  // how we did it (we use rescaling of the ivectors) but it has the same
1553  // effect. -0.5 * (Trace(covar^{-1} covar) + logdet(covar))
1554  double new_like = -0.5 * (ivector_dim + covar.LogPosDefDet()),
1555  like_change = new_like - old_like,
1556  like_change_per_frame = like_change * num_ivectors_ / gamma_.Sum();
1557 
1558  KALDI_LOG << "Overall auxf improvement from prior is " << like_change_per_frame
1559  << " per frame, or " << like_change << " per iVector.";

1566  const IvectorExtractorEstimationOptions &opts,
1567  IvectorExtractor *extractor) const {

1570  Vector<double> sum(ivector_sum_);
1571  sum.Scale(1.0 / num_ivectors_);
1572  SpMatrix<double> covar(ivector_scatter_);
1573  covar.Scale(1.0 / num_ivectors_);
1574  covar.AddVec2(-1.0, sum); // Get the centered covariance.
1575 
1576  int32 ivector_dim = extractor->IvectorDim();
1577  Vector<double> s(ivector_dim);
1578  Matrix<double> P(ivector_dim, ivector_dim);
1579  // decompose covar = P diag(s) P^T:
1580  covar.Eig(&s, &P);
1581  KALDI_LOG << "Eigenvalues of iVector covariance range from "
1582  << s.Min() << " to " << s.Max();
1583  int32 num_floored;
1584  s.ApplyFloor(1.0e-07, &num_floored);
1585  if (num_floored > 0)
1586  KALDI_WARN << "Floored " << num_floored << " eigenvalues of covar "
1587  << "of iVectors.";
1588 
1589  Matrix<double> T(P, kTrans);
1590  { // set T to a transformation that makes covar unit
1591  // (modulo floored eigenvalues).
1592  Vector<double> scales(s);
1593  scales.ApplyPow(-0.5);
1594  T.MulRowsVec(scales);
1595  if (num_floored == 0) { // a check..
1596  SpMatrix<double> Tproj(ivector_dim);
1597  Tproj.AddMat2Sp(1.0, T, kNoTrans, covar, 0.0);
1598  KALDI_ASSERT(Tproj.IsUnit(1.0e-06));
1599  }
1600  }
1601 
1602  Vector<double> sum_proj(ivector_dim);
1603  sum_proj.AddMatVec(1.0, T, kNoTrans, sum, 0.0);
1604 
1605  KALDI_ASSERT(sum_proj.Norm(2.0) != 0.0);
1606 
1607  // We need a projection that (like T) makes "covar" unit,
1608  // but also that sends "sum" to a multiple of the vector e0 = [ 1 0 0 0 .. ].
1609  // We'll do this by a transform that follows T, of the form
1610  // (I - 2 a a^T), where a is unit. [i.e. a Householder reflection].
1611  // Firstly, let x equal sum_proj normalized to unit length.
1612  // We'll let a = alpha x + beta e0, for suitable coefficients alpha and beta,
1613  // To project sum_proj (or equivalenty, x) to a multiple of e0, we'll need that
1614  // the x term in
1615  // (I - 2(alpha x + beta e0)(alpha x + beta e0) x
1616  // equals zero., i.e. 1 - 2 alpha (alpha x^T x + beta e0^T x) == 0,
1617  // (1 - 2 alpha^2 - 2 alpha beta x0) = 0
1618  // To ensure that a is unit, we require that
1619  // (alpha x + beta e0).(alpha x + beta e0) = 1, i.e.
1620  // alpha^2 + beta^2 + 2 alpha beta x0 = 1
1621  // at wolframalpha.com,
1622  // Solve[ {a^2 + b^2 + 2 a b x = 1}, {1 - 2 a^2 - 2 a b x = 0}, {a, b} ]
1623  // gives different solutions, but the one that keeps the offset positive
1624  // after projection seems to be:
1625  // alpha = 1/(sqrt(2)sqrt(1 - x0)), beta = -alpha
1626 
1627  Matrix<double> U(ivector_dim, ivector_dim);
1628  U.SetUnit();
1629  Vector<double> x(sum_proj);
1630  x.Scale(1.0 / x.Norm(2.0));
1631  double x0 = x(0), alpha, beta;
1632  alpha = 1.0 / (M_SQRT2 * sqrt(1.0 - x0));
1633  beta = -alpha;
1634  Vector<double> a(x);
1635  a.Scale(alpha);
1636  a(0) += beta;
1637  U.AddVecVec(-2.0, a, a);
1638 
1639  Matrix<double> V(ivector_dim, ivector_dim);
1640  V.AddMatMat(1.0, U, kNoTrans, T, kNoTrans, 0.0);
1641 
1642  // Optionally replace transform V with V' such that V' makes the
1643  // covariance unit and additionally diagonalizes the quadratic
1644  // term.
1645  if (opts.diagonalize) {
1646 
1647  SubMatrix<double> Vsub(V, 1, V.NumRows()-1, 0, V.NumCols());
1648  Matrix<double> Vtemp(SubMatrix<double>(V, 1, V.NumRows()-1,
1649  0, V.NumCols())),
1650  A;
1651  GetOrthogonalIvectorTransform(SubMatrix<double>(Vtemp, 0,
1652  Vtemp.NumRows(), 1, Vtemp.NumCols()-1),
1653  extractor, &A);
1654 
1655  // It is necessary to exclude the first row of V in this transformation
1656  // so that the sum_vproj has the form [ x 0 0 0 .. ], where x > 0.
1657  Vsub.AddMatMat(1.0, A, kNoTrans, Vtemp, kNoTrans, 0.0);
1658  }
1659 
1660  if (num_floored == 0) { // a check..
1661  SpMatrix<double> Vproj(ivector_dim);
1662  Vproj.AddMat2Sp(1.0, V, kNoTrans, covar, 0.0);
1663  KALDI_ASSERT(Vproj.IsUnit(1.0e-04));
1664  }
1665 
1666 
1667  Vector<double> sum_vproj(ivector_dim);
1668  sum_vproj.AddMatVec(1.0, V, kNoTrans, sum, 0.0);
1669  // Make sure sum_vproj is of the form [ x 0 0 0 .. ] with x > 0.
1670  // (the x > 0 part isn't really necessary, it's just nice to know.)
1671  KALDI_ASSERT(ApproxEqual(sum_vproj(0), sum_vproj.Norm(2.0)));
1672 
1673  double ans = PriorDiagnostics(extractor->prior_offset_);
1674 
1675  extractor->TransformIvectors(V, sum_vproj(0));
1676 
1677  return ans;
1678 }
