33 void ReadSymbolList(const std::string &rxfilename,
34  fst::SymbolTable *word_syms,
35  LabelPairVector *lpairs) {
36  Input ki(rxfilename);
37  std::string line;
38  KALDI_ASSERT(lpairs != NULL);
39  lpairs->clear();
40  while (getline(ki.Stream(), line)) {
41  std::string sym;
42  std::istringstream ss(line);
43  ss >> sym >> std::ws;
44  if (ss.fail() || !ss.eof()) {
45  KALDI_ERR << "Bad line in symbol list: "<< line
46  << ", file is: " << PrintableRxfilename(rxfilename);
47  }
48  fst::StdArc::Label lab = word_syms->Find(sym.c_str());
49  if (lab == -1) { // fst::kNoSymbol
50  KALDI_ERR << "Can't find symbol in symbol table: "
51  << line << ", file is: "
52  << PrintableRxfilename(rxfilename);
53  }
54  lpairs->emplace_back(lab, 0);
55  }
56 }

58 // convert from Lattice to standard FST
59 // also maps wildcard symbols to epsilons
60 // then removes epsilons
61 void ConvertLatticeToUnweightedAcceptor(const kaldi::Lattice &ilat,
62  const LabelPairVector &wildcards,
63  fst::StdVectorFst *ofst) {
64  // first convert from lattice to normal FST
65  fst::ConvertLattice(ilat, ofst);
66  // remove weights, project to output, sort according to input arg
67  fst::Map(ofst, fst::RmWeightMapper<fst::StdArc>());
68  fst::Project(ofst, fst::PROJECT_OUTPUT); // The words are on the output side
69  fst::Relabel(ofst, wildcards, wildcards);
70  fst::RmEpsilon(ofst); // Don't tolerate epsilons as they make it hard to
71  // tally errors
72  fst::ArcSort(ofst, fst::StdILabelCompare());
73 }

75 void CreateEditDistance(const fst::StdVectorFst &fst1,
76  const fst::StdVectorFst &fst2,
77  fst::StdVectorFst *pfst) {
78  typedef fst::StdArc StdArc;
79  typedef fst::StdArc::Weight Weight;
80  typedef fst::StdArc::Label Label;
81  Weight correct_cost(0.0);
82  Weight substitution_cost(1.0);
83  Weight insertion_cost(1.0);
84  Weight deletion_cost(1.0);
85 
86  // create set of output symbols in fst1
87  std::vector<Label> fst1syms, fst2syms;
88  GetOutputSymbols(fst1, false /*no epsilons*/, &fst1syms);
89  GetInputSymbols(fst2, false /*no epsilons*/, &fst2syms);
90 
91  pfst->AddState();
92  pfst->SetStart(0);
93  for (size_t i = 0; i < fst1syms.size(); i++)
94  pfst->AddArc(0, StdArc(fst1syms[i], 0, deletion_cost, 0)); // deletions
95 
96  for (size_t i = 0; i < fst2syms.size(); i++)
97  pfst->AddArc(0, StdArc(0, fst2syms[i], insertion_cost, 0)); // insertions
98 
99  // stupid implementation O(N^2)
100  for (size_t i = 0; i < fst1syms.size(); i++) {
101  Label label1 = fst1syms[i];
102  for (size_t j = 0; j < fst2syms.size(); j++) {
103  Label label2 = fst2syms[j];
104  Weight cost(label1 == label2 ? correct_cost : substitution_cost);
105  pfst->AddArc(0, StdArc(label1, label2, cost, 0)); // substitutions
106  }
107  }
108  pfst->SetFinal(0, Weight::One());
109  ArcSort(pfst, fst::StdOLabelCompare());
110 }

112 void CountErrors(const fst::StdVectorFst &fst,
113  int32 *correct,
114  int32 *substitutions,
115  int32 *insertions,
116  int32 *deletions,
117  int32 *num_words) {
118  typedef fst::StdArc::StateId StateId;
119  typedef fst::StdArc::Weight Weight;
120  *correct = *substitutions = *insertions = *deletions = *num_words = 0;
121 
122  // go through the first complete path in fst (there should be only one)
123  StateId src = fst.Start();
124  while (fst.Final(src)== Weight::Zero()) { // while not final
125  for (fst::ArcIterator<fst::StdVectorFst> aiter(fst, src);
126  !aiter.Done(); aiter.Next()) {
127  fst::StdArc arc = aiter.Value();
128  if (arc.ilabel == arc.olabel && arc.ilabel != 0) {
129  (*correct)++;
130  (*num_words)++;
131  } else if (arc.ilabel == 0 && arc.olabel != 0) {
132  (*deletions)++;
133  (*num_words)++;
134  } else if (arc.ilabel != 0 && arc.olabel == 0) {
135  (*insertions)++;
136  } else if (arc.ilabel != 0 && arc.olabel != 0) {
137  (*substitutions)++;
138  (*num_words)++;
139  } else {
140  KALDI_ASSERT(arc.ilabel == 0 && arc.olabel == 0);
141  }
142  src = arc.nextstate;
143  continue; // jump to next state
144  }
145  }
146 }

174  const char *usage =
175  "Finds the path having the smallest edit-distance between a lattice\n"
176  "and a reference string.\n"
177  "\n"
178  "Usage: lattice-oracle [options] <test-lattice-rspecifier> \\\n"

195  std::string wildcard_symbols;
196  std::string lats_wspecifier;

198  po.Register("word-symbol-table", &word_syms_filename,
199  "Symbol table for words [for debug output]");
200  po.Register("wildcard-symbols-list", &wild_syms_rxfilename, "Filename "
201  "(generally rxfilename) for file containing text-form list of "
202  "symbols that don't count as errors; this option requires "
203  "--word-symbol-table. Deprecated; use --wildcard-symbols "
204  "option.");
205  po.Register("wildcard-symbols", &wildcard_symbols,
206  "Colon-separated list of integer ids of symbols that "
207  "don't count as errors. Preferred alternative to deprecated "
208  "option --wildcard-symbols-list.");
209  po.Register("write-lattices", &lats_wspecifier, "If supplied, write the "
210  "lattice that contains only the oracle path to the given "
211  "wspecifier.");
212 
213  po.Read(argc, argv);
214 
215  if (po.NumArgs() != 3 && po.NumArgs() != 4) {
216  po.PrintUsage();
217  exit(1);
218  }
219 
220  std::string lats_rspecifier = po.GetArg(1),
221  reference_rspecifier = po.GetArg(2),
222  transcriptions_wspecifier = po.GetArg(3),
223  edit_distance_wspecifier = po.GetOptArg(4);
224 
225  // will read input as lattices
226  SequentialLatticeReader lattice_reader(lats_rspecifier);
227  RandomAccessInt32VectorReader reference_reader(reference_rspecifier);
228  Int32VectorWriter transcriptions_writer(transcriptions_wspecifier);
229  Int32Writer edit_distance_writer(edit_distance_wspecifier);
230  CompactLatticeWriter lats_writer(lats_wspecifier);
231 
232  fst::SymbolTable *word_syms = NULL;
233  if (word_syms_filename != "")
234  if (!(word_syms = fst::SymbolTable::ReadText(word_syms_filename)))
235  KALDI_ERR << "Could not read symbol table from file "
236  << word_syms_filename;
237 
238  LabelPairVector wildcards;
239  if (wild_syms_rxfilename != "") {
240  KALDI_WARN << "--wildcard-symbols-list option deprecated.";
241  KALDI_ASSERT(wildcard_symbols.empty() && "Do not use both "
242  "--wildcard-symbols and --wildcard-symbols-list options.");
243  KALDI_ASSERT(word_syms != NULL && "--wildcard-symbols-list option "
244  "requires --word-symbol-table option");
245  ReadSymbolList(wild_syms_rxfilename, word_syms, &wildcards);
246  } else {
247  std::vector<fst::StdArc::Label> wildcard_symbols_vec;
248  if (!SplitStringToIntegers(wildcard_symbols, ":", false,
249  &wildcard_symbols_vec)) {
250  KALDI_ERR << "Expected colon-separated list of integers for "
251  << "--wildcard-symbols option, got: " << wildcard_symbols;
252  }
253  for (size_t i = 0; i < wildcard_symbols_vec.size(); i++)
254  wildcards.emplace_back(wildcard_symbols_vec[i], 0);
255  }
256 
257  int32 n_done = 0, n_fail = 0;
258  int32 tot_correct = 0, tot_substitutions = 0,
259  tot_insertions = 0, tot_deletions = 0, tot_words = 0;
260 
261  for (; !lattice_reader.Done(); lattice_reader.Next()) {
262  std::string key = lattice_reader.Key();
263  const Lattice &lat = lattice_reader.Value();
264  std::cerr << "Lattice " << key << " read." << std::endl;
265 
266  // remove all weights while creating a standard FST
267  VectorFst<StdArc> lattice_fst;
268  ConvertLatticeToUnweightedAcceptor(lat, wildcards, &lattice_fst);
269  CheckFst(lattice_fst, "lattice_fst_", key);
270 
271  // TODO: map certain symbols (using an FST created with CreateMapFst())
272  if (!reference_reader.HasKey(key)) {
273  KALDI_WARN << "No reference present for utterance " << key;
274  n_fail++;
275  continue;
276  }
277  const std::vector<int32> &reference = reference_reader.Value(key);
278  VectorFst<StdArc> reference_fst;
279  MakeLinearAcceptor(reference, &reference_fst);
280 
281  // Remove any wildcards in reference.
282  fst::Relabel(&reference_fst, wildcards, wildcards);
283  CheckFst(reference_fst, "reference_fst_", key);
284 
285  // recreate edit distance fst if necessary
286  fst::StdVectorFst edit_distance_fst;
287  CreateEditDistance(lattice_fst, reference_fst, &edit_distance_fst);
288 
289  // compose with edit distance transducer
290  VectorFst<StdArc> edit_ref_fst;
291  fst::Compose(edit_distance_fst, reference_fst, &edit_ref_fst);
292  CheckFst(edit_ref_fst, "composed_", key);
293 
294  // make sure composed FST is input sorted
295  fst::ArcSort(&edit_ref_fst, fst::StdILabelCompare());
296 
297  // compose with previous result
298  VectorFst<StdArc> result_fst;
299  fst::Compose(lattice_fst, edit_ref_fst, &result_fst);
300  CheckFst(result_fst, "result_", key);
301 
302  // find out best path
303  VectorFst<StdArc> best_path;
304  fst::ShortestPath(result_fst, &best_path);
305  CheckFst(best_path, "best_path_", key);
306 
307  if (best_path.Start() == fst::kNoStateId) {
308  KALDI_WARN << "Best-path failed for key " << key;
309  n_fail++;
310  } else {
311  // count errors
312  int32 correct, substitutions, insertions, deletions, num_words;
313  CountErrors(best_path, &correct, &substitutions,
314  &insertions, &deletions, &num_words);
315  int32 tot_errs = substitutions + insertions + deletions;
316  if (edit_distance_wspecifier != "")
317  edit_distance_writer.Write(key, tot_errs);
318  KALDI_LOG << "%WER " << (100.*tot_errs) / num_words << " [ " << tot_errs
319  << " / " << num_words << ", " << insertions << " insertions, "
320  << deletions << " deletions, " << substitutions << " sub ]";
321  tot_correct += correct;
322  tot_substitutions += substitutions;
323  tot_insertions += insertions;
324  tot_deletions += deletions;
325  tot_words += num_words;
326 
327  std::vector<int32> oracle_words;
328  std::vector<int32> reference_words;
329  Weight weight;
330  GetLinearSymbolSequence(best_path, &oracle_words,
331  &reference_words, &weight);
332  KALDI_LOG << "For utterance " << key << ", best cost " << weight;
333  if (transcriptions_wspecifier != "")
334  transcriptions_writer.Write(key, oracle_words);
335  if (word_syms != NULL) {
336  std::cerr << key << " (oracle) ";
337  for (size_t i = 0; i < oracle_words.size(); i++) {
338  std::string s = word_syms->Find(oracle_words[i]);
339  if (s == "")
340  KALDI_ERR << "Word-id " << oracle_words[i]
341  << " not in symbol table.";
342  std::cerr << s << ' ';
343  }
344  std::cerr << '\n' << key << " (reference) ";
345  for (size_t i = 0; i < reference_words.size(); i++) {
346  std::string s = word_syms->Find(reference_words[i]);
347  if (s == "")
348  KALDI_ERR << "Word-id " << reference_words[i]
349  << " not in symbol table.";
350  std::cerr << s << ' ';
351  }
352  std::cerr << '\n';
353  }
354 
355  // If requested, write the lattice that only contains the oracle path.
356  if (lats_wspecifier != "") {
357  CompactLattice oracle_clat_mask;
358  MakeLinearAcceptor(oracle_words, &oracle_clat_mask);
359 
360  CompactLattice clat;
361  CompactLattice oracle_clat;
362  ConvertLattice(lat, &clat);
363  fst::Relabel(&clat, wildcards, LabelPairVector());
364  fst::ArcSort(&clat, fst::ILabelCompare<CompactLatticeArc>());
365  fst::Compose(oracle_clat_mask, clat, &oracle_clat_mask);
366  fst::ShortestPath(oracle_clat_mask, &oracle_clat);
367  fst::Project(&oracle_clat, fst::PROJECT_OUTPUT);
368  TopSortCompactLatticeIfNeeded(&oracle_clat);
369 
370  if (oracle_clat.Start() == fst::kNoStateId) {
371  KALDI_WARN << "Failed to find the oracle path in the original "
372  << "lattice: " << key;
373  } else {
374  lats_writer.Write(key, oracle_clat);
375  }
376  }
377  }
378  n_done++;
379  }
380  delete word_syms;
381  int32 tot_errs = tot_substitutions + tot_deletions + tot_insertions;
382  // Warning: the script egs/s5/*/steps/oracle_wer.sh parses the next line.
383  KALDI_LOG << "Overall %WER " << (100.*tot_errs)/tot_words << " [ "
384  << tot_errs << " / " << tot_words << ", " << tot_insertions
385  << " insertions, " << tot_deletions << " deletions, "
386  << tot_substitutions << " substitutions ]";
387  KALDI_LOG << "Scored " << n_done << " lattices, " << n_fail
388  << " not present in ref.";
389  } catch(const std::exception &e) {
390  std::cerr << e.what();
391  return -1;
392  }
