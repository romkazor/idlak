53  KALDI_ASSERT(linear_params_.NumRows() > 0 &&
54  !time_offsets_.empty() &&

63 std::string TdnnComponent::Info() const {
64  std::ostringstream stream;
65  stream << UpdatableComponent::Info();
66  if (orthonormal_constraint_ != 0.0)
67  stream << ", orthonormal-constraint=" << orthonormal_constraint_;
68  stream << ", time-offsets=";
69  for (size_t i = 0; i < time_offsets_.size(); i++) {
70  if (i != 0) stream << ',';
71  stream << time_offsets_[i];
72  }
73  PrintParameterStats(stream, "linear-params", linear_params_,
74  false, // include_mean
75  true, // include_row_norms
76  true, // include_column_norms
77  GetVerboseLevel() >= 2); // include_singular_values
78  if (bias_params_.Dim() == 0) {
79  stream << ", has-bias=false";
80  } else {
81  PrintParameterStats(stream, "bias", bias_params_, true);
82  }
83  if (!use_natural_gradient_) {
84  stream << ", use-natural-gradient=false";
85  } else {
86  stream << ", rank-in=" << preconditioner_in_.GetRank()
87  << ", rank-out=" << preconditioner_out_.GetRank()
88  << ", num-samples-history=" << preconditioner_in_.GetNumSamplesHistory()
89  << ", update-period=" << preconditioner_in_.GetUpdatePeriod()
90  << ", alpha-in=" << preconditioner_in_.GetAlpha()
91  << ", alpha-out=" << preconditioner_out_.GetAlpha();

97 void TdnnComponent::InitFromConfig(ConfigLine *cfl) {
98  // 1. Config values inherited from UpdatableComponent.
99  InitLearningRatesFromConfig(cfl);
100 
101  // 2. Structural config values
102  std::string time_offsets;
103 
104  int32 input_dim = -1, output_dim = -1;
105 
106  bool ok = cfl->GetValue("time-offsets", &time_offsets) &&
107  cfl->GetValue("input-dim", &input_dim) &&
108  cfl->GetValue("output-dim", &output_dim);
109  if (!ok || input_dim <= 0 || output_dim <= 0 ||
110  !SplitStringToIntegers(time_offsets, ",", false, &time_offsets_) ||
111  time_offsets_.empty()) {
112  KALDI_ERR << "Bad initializer: there is a problem with "
113  "time-offsets, input-dim or output-dim (not defined?): "
114  << cfl->WholeLine();
115  }
116 
117  if (std::set<int32>(time_offsets_.begin(),
118  time_offsets_.end()).size() != time_offsets_.size()) {
119  KALDI_ERR << "Bad initializer: repeated time-offsets: "
120  << cfl->WholeLine();
121  }
122 
123  // 3. Parameter-initialization configs, "has-bias", and
124  // orthonormal-constraint.
125  orthonormal_constraint_ = 0.0;
126  BaseFloat param_stddev = -1, bias_mean = 0.0, bias_stddev = 1.0;
127  bool use_bias = true;
128  cfl->GetValue("param-stddev", &param_stddev);
129  cfl->GetValue("bias-stddev", &bias_stddev);
130  cfl->GetValue("bias-mean", &bias_mean);
131  cfl->GetValue("use-bias", &use_bias);
132  cfl->GetValue("orthonormal-constraint", &orthonormal_constraint_);
133  if (param_stddev < 0.0) {
134  param_stddev = 1.0 / sqrt(input_dim * time_offsets_.size());
135  }
136  // initialize the parameters.
137  linear_params_.Resize(output_dim,
138  input_dim * time_offsets_.size());
139  linear_params_.SetRandn();
140  linear_params_.Scale(param_stddev);
141 
142  if (use_bias) {
143  bias_params_.Resize(output_dim);
144  bias_params_.SetRandn();
145  bias_params_.Scale(bias_stddev);
146  bias_params_.Add(bias_mean);
147  } else {
148  bias_params_.Resize(0);
149  }
150 
151  // 4. Natural-gradient related configs.
152  use_natural_gradient_ = true;
153  int32 rank_out = -1, rank_in = -1;
154  BaseFloat alpha_out = 4.0, alpha_in = 4.0,
155  num_samples_history = 2000.0;
156  cfl->GetValue("use-natural-gradient", &use_natural_gradient_);
157  cfl->GetValue("rank-in", &rank_in);
158  cfl->GetValue("rank-out", &rank_out);
159  cfl->GetValue("alpha-in", &alpha_in);
160  cfl->GetValue("alpha-out", &alpha_out);
161  cfl->GetValue("num-samples-history", &num_samples_history);
162 
163  int32 spliced_input_dim =
164  input_dim * static_cast<int32>(time_offsets_.size());
165  if (rank_in < 0)
166  rank_in = std::min<int32>(20, (spliced_input_dim + 1) / 2);
167  preconditioner_in_.SetRank(rank_in);
168  if (rank_out < 0)
169  rank_out = std::min<int32>(80, (output_dim + 1) / 2);
170  preconditioner_out_.SetRank(rank_out);
171  preconditioner_in_.SetNumSamplesHistory(num_samples_history);
172  preconditioner_out_.SetNumSamplesHistory(num_samples_history);
173 
174  preconditioner_in_.SetAlpha(alpha_in);
175  preconditioner_out_.SetAlpha(alpha_out);
176 
177  preconditioner_in_.SetUpdatePeriod(4);
178  preconditioner_out_.SetUpdatePeriod(4);
179 }

181 void* TdnnComponent::Propagate(
182  const ComponentPrecomputedIndexes *indexes_in,
183  const CuMatrixBase<BaseFloat> &in,
184  CuMatrixBase<BaseFloat> *out) const {
185  const PrecomputedIndexes *indexes =
186  dynamic_cast<const PrecomputedIndexes*>(indexes_in);
187  KALDI_ASSERT(indexes != NULL);
188 
189  if (bias_params_.Dim() != 0)
190  out->CopyRowsFromVec(bias_params_);
191  // if bias_params_.Dim() == 0 we don't need to zero 'out' at
192  // this point because in that case we set the flag kPropagateAdds,
193  // so the calling code knows that the Propagate function *adds to*
194  // the 'out' matrix, so it should (typicaly) be zeroed before calling
195  // Propagate().
196 
197  KALDI_ASSERT(indexes->row_offsets.size() == time_offsets_.size());
198 
199  int32 num_offsets = time_offsets_.size(),
200  input_dim = InputDim();
201  for (int32 i = 0; i < num_offsets; i++) {
202  CuSubMatrix<BaseFloat> in_part = GetInputPart(in, out->NumRows(),
203  indexes->row_stride,
204  indexes->row_offsets[i]);
205  CuSubMatrix<BaseFloat> linear_params_part(linear_params_,
206  0, linear_params_.NumRows(),
207  i * input_dim, input_dim);
208  out->AddMatMat(1.0, in_part, kNoTrans, linear_params_part, kTrans, 1.0);
209  }
210  return NULL;
211 }

213 void TdnnComponent::Backprop(
214  const std::string &debug_info,
215  const ComponentPrecomputedIndexes *indexes_in,
216  const CuMatrixBase<BaseFloat> &in_value,
217  const CuMatrixBase<BaseFloat> &, // out_value
218  const CuMatrixBase<BaseFloat> &out_deriv,
219  void*, // memo
220  Component *to_update_in,
221  CuMatrixBase<BaseFloat> *in_deriv) const {
222  const PrecomputedIndexes *indexes =
223  dynamic_cast<const PrecomputedIndexes*>(indexes_in);
224  KALDI_ASSERT(indexes != NULL &&
225  indexes->row_offsets.size() == time_offsets_.size());
226  int32 num_offsets = time_offsets_.size(),
227  input_dim = InputDim();
228 
229  if (in_deriv != NULL) {
230  // Propagate the derivatives back to the input data.
231  for (int32 i = 0; i < num_offsets; i++) {
232  CuSubMatrix<BaseFloat> in_deriv_part =
233  GetInputPart(*in_deriv, out_deriv.NumRows(),
234  indexes->row_stride, indexes->row_offsets[i]);
235  CuSubMatrix<BaseFloat> linear_params_part(linear_params_,
236  0, linear_params_.NumRows(),
237  i * input_dim, input_dim);
238  // note: this component has the property kBackpropAdds, which is why the
239  // final 1.0 is there in the following call (otherwise we'd have to zero
240  // *in_deriv first).
241  in_deriv_part.AddMatMat(1.0, out_deriv, kNoTrans,
242  linear_params_part, kNoTrans, 1.0);
243  }
244  }
245 
246  if (to_update_in != NULL) {
247  TdnnComponent *to_update =
248  dynamic_cast<TdnnComponent*>(to_update_in);
249  KALDI_ASSERT(to_update != NULL);
250 
251  if (to_update->learning_rate_ == 0.0)
252  return;
253 
254  if (to_update->is_gradient_ || !to_update->use_natural_gradient_)
255  to_update->UpdateSimple(*indexes, in_value, out_deriv);
256  else
257  to_update->UpdateNaturalGradient(*indexes, in_value, out_deriv);
258  }
259 }

261 void TdnnComponent::UpdateSimple(
262  const PrecomputedIndexes &indexes,
263  const CuMatrixBase<BaseFloat> &in_value,
264  const CuMatrixBase<BaseFloat> &out_deriv) {
265 
266  if (bias_params_.Dim() != 0)
267  bias_params_.AddRowSumMat(learning_rate_, out_deriv);
268 
269  int32 input_dim = in_value.NumCols(),
270  num_offsets = time_offsets_.size();
271  for (int32 i = 0; i < num_offsets; i++) {
272  CuSubMatrix<BaseFloat> in_value_part =
273  GetInputPart(in_value, out_deriv.NumRows(),
274  indexes.row_stride,
275  indexes.row_offsets[i]);
276  CuSubMatrix<BaseFloat> linear_params_part(linear_params_,
277  0, linear_params_.NumRows(),
278  i * input_dim, input_dim);
279  linear_params_part.AddMatMat(learning_rate_, out_deriv, kTrans,
280  in_value_part, kNoTrans, 1.0);
281  }
282 }

284 void TdnnComponent::UpdateNaturalGradient(
285  const PrecomputedIndexes &indexes,
286  const CuMatrixBase<BaseFloat> &in_value,
287  const CuMatrixBase<BaseFloat> &out_deriv) {
288 
289  int32 num_offsets = time_offsets_.size(),
290  num_rows = out_deriv.NumRows(),
291  input_dim = in_value.NumCols(),
292  spliced_input_dim = num_offsets * input_dim,
293  augmented_input_dim =
294  spliced_input_dim + (bias_params_.Dim() != 0 ? 1 : 0);
295 
296  // in_value_temp is the fully spliced input with a column of ones appended to
297  // it.
298  CuMatrix<BaseFloat> in_value_temp(num_rows,
299  augmented_input_dim);
300  if (bias_params_.Dim() != 0) {
301  // set the last column of in_value_temp to 1.0
302  in_value_temp.Range(0, num_rows, spliced_input_dim, 1).Set(1.0);
303  }
304 
305  for (int32 i = 0; i < num_offsets; i++) {
306  CuSubMatrix<BaseFloat> in_value_temp_part(in_value_temp,
307  0, num_rows,
308  i * input_dim, input_dim),
309  in_value_part = GetInputPart(in_value,
310  num_rows,
311  indexes.row_stride,
312  indexes.row_offsets[i]);
313  in_value_temp_part.CopyFromMat(in_value_part);
314  }
315 
316  CuMatrix<BaseFloat> out_deriv_temp(out_deriv);
317 
318  // These "scale" values get will get multiplied into the learning rate (faster
319  // than having the matrices scaled inside the preconditioning code).
320  BaseFloat in_scale, out_scale;
321 
322  preconditioner_in_.PreconditionDirections(&in_value_temp, &in_scale);
323  preconditioner_out_.PreconditionDirections(&out_deriv_temp, &out_scale);
324 
325  // "scale" is a scaling factor coming from the PreconditionDirections calls
326  // (it's faster to have them output a scaling factor than to have them scale
327  // their outputs).
328  BaseFloat scale = in_scale * out_scale,
329  local_lrate = scale * learning_rate_;
330 
331  if (bias_params_.Dim() != 0) {
332  // this "precon_ones" is what happens to the vector of 1's representing
333  // offsets, after multiplication by the preconditioner.
334  CuVector<BaseFloat> precon_ones(num_rows);
335  precon_ones.CopyColFromMat(in_value_temp, spliced_input_dim);
336  bias_params_.AddMatVec(local_lrate, out_deriv_temp, kTrans,
337  precon_ones, 1.0);
338  }
339 
340  CuSubMatrix<BaseFloat> in_value_precon_part(in_value_temp,
341  0, num_rows,
342  0, spliced_input_dim);
343 
344  linear_params_.AddMatMat(local_lrate, out_deriv_temp, kTrans,
345  in_value_precon_part, kNoTrans, 1.0);
346 }

348 void TdnnComponent::ReorderIndexes(
349  std::vector<Index> *input_indexes,
350  std::vector<Index> *output_indexes) const {
351  using namespace time_height_convolution;
352 
353  // The following figures out a regular structure for the input and
354  // output indexes, in case there were gaps (which is unlikely in typical
355  // situations).
356  ConvolutionComputationIo io;
357  GetComputationIo(*input_indexes, *output_indexes, &io);
358  ModifyComputationIo(&io);
359 
360  std::vector<Index> modified_input_indexes,
361  modified_output_indexes;
362  // The following call ensures that 'modified_input_indexes' and
363  // 'modified_output_indexes' have the required ordering (where t has the
364  // largest stride and each (n,x) pair is repeated for each 't' value), as well
365  // as doing padding (setting t values to kNoTime where it had to insert
366  // elements to ensure regular structure).
367  GetIndexesForComputation(io, *input_indexes, *output_indexes,
368  &modified_input_indexes,
369  &modified_output_indexes);
370 
371  // It will be quite rare that this function actually changes
372  // 'input_indexes' or 'output_indexes', because in most cases,
373  // the indexes will already have the required structure and
374  // ordering.
375  input_indexes->swap(modified_input_indexes);
376  output_indexes->swap(modified_output_indexes);
377 }

407 void TdnnComponent::Read(std::istream &is, bool binary) {
408  std::string token = ReadUpdatableCommon(is, binary);
409  ExpectToken(is, binary, "<TimeOffsets>");
410  ReadIntegerVector(is, binary, &time_offsets_);
411  ExpectToken(is, binary, "<LinearParams>");
412  linear_params_.Read(is, binary);
413  ExpectToken(is, binary, "<BiasParams>");
414  bias_params_.Read(is, binary);
415  ExpectToken(is, binary, "<OrthonormalConstraint>");
416  ReadBasicType(is, binary, &orthonormal_constraint_);
417  ExpectToken(is, binary, "<UseNaturalGradient>");
418  ReadBasicType(is, binary, &use_natural_gradient_);
419  int32 rank_in, rank_out;
420  BaseFloat alpha_in, alpha_out,
421  num_samples_history;
422  ExpectToken(is, binary, "<NumSamplesHistory>");
423  ReadBasicType(is, binary, &num_samples_history);
424  { // This can be simplified after a while. It's to read a format of the model
425  // that was never checked into master, but with which I (Dan) did many of
426  // the experiments while tuning the resnet TDNN-F.
427  std::string token;
428  ReadToken(is, binary, &token);
429  if (token == "<AlphaInOut>") {
430  ReadBasicType(is, binary, &alpha_in);
431  ReadBasicType(is, binary, &alpha_out);
432  } else {
433  KALDI_ASSERT(token == "<Alpha>");
434  ReadBasicType(is, binary, &alpha_in);
435  alpha_out = alpha_in;
436  }
437  }
438  preconditioner_in_.SetAlpha(alpha_in);
439  preconditioner_out_.SetAlpha(alpha_out);
440  ExpectToken(is, binary, "<RankInOut>");
441  ReadBasicType(is, binary, &rank_in);
442  ReadBasicType(is, binary, &rank_out);
443  preconditioner_in_.SetRank(rank_in);
444  preconditioner_out_.SetRank(rank_out);
445  preconditioner_in_.SetNumSamplesHistory(num_samples_history);
446  preconditioner_out_.SetNumSamplesHistory(num_samples_history);
447  // the update periods are not configurable.
448  preconditioner_in_.SetUpdatePeriod(4);
449  preconditioner_out_.SetUpdatePeriod(4);
450  ExpectToken(is, binary, "</TdnnComponent>");
451  Check();
452 }

470  const MiscComputationInfo &misc_info,

479  used_inputs->clear();

481  }
482  for (size_t i = 0; i < size; i++) {

485  if (used_inputs != NULL) {
486  // This input index is available.
487  used_inputs->push_back(index);
488  }
489  } else {
490  return false;
491  }
492  }
493  return true;
494 }

496 // static
497 CuSubMatrix<BaseFloat> TdnnComponent::GetInputPart(
498  const CuMatrixBase<BaseFloat> &input_matrix,
499  int32 num_output_rows,
500  int32 row_stride,
501  int32 row_offset) {
502  KALDI_ASSERT(row_offset >= 0 && row_stride >= 1 &&
503  input_matrix.NumRows() >=
504  row_offset + (row_stride * num_output_rows) - (row_stride - 1));
505  // constructor takes args: (data, num_rows, num_cols, stride).
506  return CuSubMatrix<BaseFloat>(
507  input_matrix.Data() + input_matrix.Stride() * row_offset,
508  num_output_rows,
509  input_matrix.NumCols(),
510  input_matrix.Stride() * row_stride);
511 }

513 void TdnnComponent::ModifyComputationIo(
514  time_height_convolution::ConvolutionComputationIo *io) {
515  if (io->t_step_out == 0) {
516  // the 't_step' values may be zero if there was only one (input or output)
517  // index so the time-stride could not be determined. This code fixes them
518  // up in that case. (If there was only one value, the stride is a
519  // don't-care actually).
520  if (io->t_step_in == 0)
521  io->t_step_in = 1;
522  io->t_step_out = io->t_step_in;
523  }
524  // At this point the t_step_{in,out} values will be nonzero.
525  KALDI_ASSERT(io->t_step_out % io->t_step_in == 0);
526  // The following affects the ordering of the input indexes; it allows us to
527  // reshape the input matrix in the way that we need to, in cases where there
528  // is subsampling. See the explanation where the variable was declared in
529  // class ConvolutionComputationIo.
530  io->reorder_t_in = io->t_step_out / io->t_step_in;
531 
532  // make sure that num_t_in is a multiple of io->reorder_t_in by rounding up.
533  int32 n = io->reorder_t_in;
534  io->num_t_in = n * ((io->num_t_in + n - 1) / n);
535 }

537 ComponentPrecomputedIndexes* TdnnComponent::PrecomputeIndexes(
538  const MiscComputationInfo &misc_info,
539  const std::vector<Index> &input_indexes,
540  const std::vector<Index> &output_indexes,
541  bool need_backprop) const {
542  using namespace time_height_convolution;
543  // The following figures out a regular structure for the input and
544  // output indexes, in case there were gaps (which is unlikely in typical
545  // situations).
546  ConvolutionComputationIo io;
547  GetComputationIo(input_indexes, output_indexes, &io);
548  ModifyComputationIo(&io);
549 
550  if (RandInt(0, 10) == 0) {
551  // Spot check that the provided indexes have the required properties;
552  // this is like calling this->ReorderIndexes() and checking that it
553  // doesn't change anything.
554  std::vector<Index> modified_input_indexes,
555  modified_output_indexes;
556  GetIndexesForComputation(io, input_indexes, output_indexes,
557  &modified_input_indexes,
558  &modified_output_indexes);
559  KALDI_ASSERT(modified_input_indexes == input_indexes &&
560  modified_output_indexes == output_indexes);
561  }
562 
563 
564  PrecomputedIndexes *ans = new PrecomputedIndexes();
565  ans->row_stride = io.reorder_t_in;
566  int32 num_offsets = time_offsets_.size();
567  ans->row_offsets.resize(num_offsets);
568  for (int32 i = 0; i < num_offsets; i++) {
569  // For each offset, work out which row of the input has the same t value as
570  // the first t value in the output plus that offset. That becomes the start
571  // row of the corresponding sub-part of the input.
572  int32 time_offset = time_offsets_[i],
573  required_input_t = io.start_t_out + time_offset,
574  input_t = (required_input_t - io.start_t_in) / io.t_step_in;
575 
576  KALDI_ASSERT(required_input_t == io.start_t_in + io.t_step_in * input_t);
577  // input_t is a kind of normalized time offset in the input, relative to the
578  // first 't' value in the input and divided by the t-step in the input, so
579  // it's the numbering "as if" the input 't' values were numbered from 0,1,2.
580  // To turn input_t into an input row we need to take account of 'reorder_t_in'.
581  // If this is 1 then the input row is input_t times io.num_images.
582  // Otherwise it's a little more complicated and to understand it you should
583  // read the comment where 'reorder_t_in' is declared in convolution.h.
584  // Briefly: the part that is an integer multiple of 'reorder_t_in' gets
585  // multiplied by io.num_images; the remainder does not.
586 
587  int32 n = io.reorder_t_in,
588  input_t_multiple = n * (input_t / n), input_t_remainder = input_t % n;
589  // note: input_t == input_t_multiple + input_t_remainder .
590  int32 input_row_offset = input_t_multiple * io.num_images +
591  input_t_remainder;
592  ans->row_offsets[i] = input_row_offset;
593  }
594  return ans;
595 }
