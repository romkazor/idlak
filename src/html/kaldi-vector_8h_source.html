484 template<typename Real>
485 class SubVector : public VectorBase<Real> {
486  public:
490  SubVector(const VectorBase<Real> &t, const MatrixIndexT origin,
491  const MatrixIndexT length) : VectorBase<Real>() {
492  // following assert equiv to origin>=0 && length>=0 &&
493  // origin+length <= rt.dim_
494  KALDI_ASSERT(static_cast<UnsignedMatrixIndexT>(origin)+
495  static_cast<UnsignedMatrixIndexT>(length) <=
496  static_cast<UnsignedMatrixIndexT>(t.Dim()));
497  VectorBase<Real>::data_ = const_cast<Real*> (t.Data()+origin);
498  VectorBase<Real>::dim_ = length;
499  }
500 
503  SubVector(const PackedMatrix<Real> &M) {
504  VectorBase<Real>::data_ = const_cast<Real*> (M.Data());
505  VectorBase<Real>::dim_ = (M.NumRows()*(M.NumRows()+1))/2;
506  }
507 
509  SubVector(const SubVector &other) : VectorBase<Real> () {
510  // this copy constructor needed for Range() to work in base class.
511  VectorBase<Real>::data_ = other.data_;
512  VectorBase<Real>::dim_ = other.dim_;
513  }
514 
517  SubVector(Real *data, MatrixIndexT length) : VectorBase<Real> () {
518  VectorBase<Real>::data_ = data;
519  VectorBase<Real>::dim_ = length;
520  }
521 
522 
524  SubVector(const MatrixBase<Real> &matrix, MatrixIndexT row) {
525  VectorBase<Real>::data_ = const_cast<Real*>(matrix.RowData(row));
